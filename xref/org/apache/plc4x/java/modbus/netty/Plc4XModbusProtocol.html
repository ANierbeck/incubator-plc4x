57      @Override
58      protected void encode(ChannelHandlerContext ctx, PlcRequestContainer<PlcRequest, PlcResponse> msg, List<Object> out) throws Exception {
59          LOGGER.trace("(<--OUT): {}, {}, {}", ctx, msg, out);
60          // Reset transactionId on overflow
61          transactionId.compareAndSet(Short.MAX_VALUE + 1, 0);
62          PlcRequest request = msg.getRequest();
63          if (request instanceof PlcReadRequest) {
64              encodeReadRequest(msg, out);
65          } else if (request instanceof PlcWriteRequest) {
66              encodeWriteRequest(msg, out);
67          }
68      }

70      private void encodeWriteRequest(PlcRequestContainer<PlcRequest, PlcResponse> msg, List<Object> out) throws PlcException {
71          PlcWriteRequest request = (PlcWriteRequest) msg.getRequest();
72          // TODO: support multiple requests
73          WriteRequestItem<?> writeRequestItem = request.getRequestItem().get();
74          // TODO: check if we can map like this. Implication is that we can only work with int, short, byte and boolean
75          // TODO: for higher datatypes float, double etc we might need to split the bytes into chunks
76          int quantity = writeRequestItem.getSize();
77          short unitId = 0;
78          checkSupportedDataType(writeRequestItem.getValues());
79  
80          ModbusAddress address = (ModbusAddress) writeRequestItem.getAddress();
81          ModbusPdu modbusRequest;
82          if (address instanceof RegisterAddress) {
83              RegisterAddress registerAddress = (RegisterAddress) address;
84              if (quantity > 1) {
85                  byte[] bytesToWrite = produceRegisterValue(writeRequestItem.getValues());
86                  modbusRequest = new WriteMultipleRegistersRequest(registerAddress.getAddress(), quantity, bytesToWrite);
87              } else {
88                  byte[] register = produceRegisterValue(writeRequestItem.getValues());
89                  int intToWrite = register[0] << 8 | register[1];
90                  modbusRequest = new WriteSingleRegisterRequest(registerAddress.getAddress(), intToWrite);
91              }
92          } else if (address instanceof CoilAddress) {
93              CoilAddress coilAddress = (CoilAddress) address;
94              if (quantity > 1) {
95                  byte[] bytesToWrite = produceCoilValue(writeRequestItem.getValues());
96                  modbusRequest = new WriteMultipleCoilsRequest(coilAddress.getAddress(), quantity, bytesToWrite);
97              } else {
98                  byte[] coil = produceCoilValue(writeRequestItem.getValues());
99                  boolean booleanToWrite = (coil[0] >> 8) == 1;
100                 modbusRequest = new WriteSingleCoilRequest(coilAddress.getAddress(), booleanToWrite);
101             }
102         } else {
103             throw new PlcProtocolException("Unsupported address type" + address.getClass());
104         }
105         short transactionId = (short) this.transactionId.getAndIncrement();
106         requestsMap.put(transactionId, msg);
107         out.add(new ModbusTcpPayload(transactionId, unitId, modbusRequest));
108     }

110     private void encodeReadRequest(PlcRequestContainer<PlcRequest, PlcResponse> msg, List<Object> out) throws PlcException {
111         PlcReadRequest request = (PlcReadRequest) msg.getRequest();
112         // TODO: support multiple requests
113         ReadRequestItem<?> readRequestItem = request.getRequestItem().get();
114         // TODO: check if we can map like this. Implication is that we can only work with int, short, byte and boolean
115         // TODO: for higher datatypes float, double etc we might need to split the bytes into chunks
116         int quantity = readRequestItem.getSize();
117         // TODO: the unit the should be used for multiple Requests
118         short unitId = 0;
119 
120         ModbusAddress address = (ModbusAddress) readRequestItem.getAddress();
121         ModbusPdu modbusRequest;
122         if (address instanceof CoilAddress) {
123             CoilAddress coilAddress = (CoilAddress) address;
124             modbusRequest = new ReadCoilsRequest(coilAddress.getAddress(), quantity);
125         } else if (address instanceof RegisterAddress) {
126             RegisterAddress registerAddress = (RegisterAddress) address;
127             modbusRequest = new ReadHoldingRegistersRequest(registerAddress.getAddress(), quantity);
128         } else if (address instanceof ReadDiscreteInputsModbusAddress) {
129             ReadDiscreteInputsModbusAddress readDiscreteInputsModbusAddress = (ReadDiscreteInputsModbusAddress) address;
130             modbusRequest = new ReadDiscreteInputsRequest(readDiscreteInputsModbusAddress.getAddress(), quantity);
131         } else if (address instanceof ReadHoldingRegistersModbusAddress) {
132             ReadHoldingRegistersModbusAddress readHoldingRegistersModbusAddress = (ReadHoldingRegistersModbusAddress) address;
133             modbusRequest = new ReadHoldingRegistersRequest(readHoldingRegistersModbusAddress.getAddress(), quantity);
134         } else if (address instanceof ReadInputRegistersModbusAddress) {
135             ReadInputRegistersModbusAddress readInputRegistersModbusAddress = (ReadInputRegistersModbusAddress) address;
136             modbusRequest = new ReadInputRegistersRequest(readInputRegistersModbusAddress.getAddress(), quantity);
137         } else {
138             throw new PlcProtocolException("Unsupported address type" + address.getClass());
139         }
140         short transactionId = (short) this.transactionId.getAndIncrement();
141         requestsMap.put(transactionId, msg);
142         out.add(new ModbusTcpPayload(transactionId, unitId, modbusRequest));
143     }

146     @Override
147     protected void decode(ChannelHandlerContext ctx, ModbusTcpPayload msg, List<Object> out) throws Exception {
148         LOGGER.trace("(-->IN): {}, {}, {}", ctx, msg, out);
149         LOGGER.debug("{}: transactionId: {}, unitId: {}, modbusPdu:{}", msg, msg.getTransactionId(), msg.getUnitId(), msg.getModbusPdu());
150         // TODO: implement me
151         short transactionId = msg.getTransactionId();
152         PlcRequestContainer<PlcRequest, PlcResponse> plcRequestContainer = requestsMap.get(transactionId);
153         if (plcRequestContainer == null) {
154             throw new PlcProtocolException("Unrelated payload received" + msg);
155         }
156 
157         // TODO: only single Item supported for now
158         PlcRequest request = plcRequestContainer.getRequest();
159         RequestItem requestItem = (RequestItem) request.getRequestItem().get();
160         Class datatype = requestItem.getDatatype();
161 
162         ModbusPdu modbusPdu = msg.getModbusPdu();
163         short unitId = msg.getUnitId();
164 
165         if (modbusPdu instanceof WriteMultipleCoilsResponse) {
166             // TODO: finish implementation
167             WriteMultipleCoilsResponse writeMultipleCoilsResponse = (WriteMultipleCoilsResponse) modbusPdu;
168             LOGGER.debug("{}: address:{}, quantity:{}", writeMultipleCoilsResponse, writeMultipleCoilsResponse.getAddress(), writeMultipleCoilsResponse.getQuantity());
169             plcRequestContainer.getResponseFuture().complete(new PlcWriteResponse((PlcWriteRequest) request, new WriteResponseItem<>((WriteRequestItem) requestItem, ResponseCode.OK)));
170         } else if (modbusPdu instanceof WriteMultipleRegistersResponse) {
171             // TODO: finish implementation
172             WriteMultipleRegistersResponse writeMultipleRegistersResponse = (WriteMultipleRegistersResponse) modbusPdu;
173             LOGGER.debug("{}: address:{}, quantity:{}", writeMultipleRegistersResponse, writeMultipleRegistersResponse.getAddress(), writeMultipleRegistersResponse.getQuantity());
174             plcRequestContainer.getResponseFuture().complete(new PlcWriteResponse((PlcWriteRequest) request, new WriteResponseItem<>((WriteRequestItem) requestItem, ResponseCode.OK)));
175         } else if (modbusPdu instanceof WriteSingleCoilResponse) {
176             // TODO: finish implementation
177             WriteSingleCoilResponse writeSingleCoilResponse = (WriteSingleCoilResponse) modbusPdu;
178             LOGGER.debug("{}: address:{}, value:{}", writeSingleCoilResponse, writeSingleCoilResponse.getAddress(), writeSingleCoilResponse.getValue());
179             plcRequestContainer.getResponseFuture().complete(new PlcWriteResponse((PlcWriteRequest) request, new WriteResponseItem<>((WriteRequestItem) requestItem, ResponseCode.OK)));
180         } else if (modbusPdu instanceof WriteSingleRegisterResponse) {
181             // TODO: finish implementation
182             WriteSingleRegisterResponse writeSingleRegisterResponse = (WriteSingleRegisterResponse) modbusPdu;
183             LOGGER.debug("{}: address:{}, value:{}", writeSingleRegisterResponse, writeSingleRegisterResponse.getAddress(), writeSingleRegisterResponse.getValue());
184             plcRequestContainer.getResponseFuture().complete(new PlcWriteResponse((PlcWriteRequest) request, new WriteResponseItem<>((WriteRequestItem) requestItem, ResponseCode.OK)));
185         } else if (modbusPdu instanceof ReadCoilsResponse) {
186             // TODO: finish implementation
187             ReadCoilsResponse readCoilsResponse = (ReadCoilsResponse) modbusPdu;
188             LOGGER.debug("{}: Nothing", readCoilsResponse);
189             ByteBuf byteBuf = readCoilsResponse.getCoilStatus();
190             List data = produceCoilValueList(requestItem, datatype, byteBuf);
191             plcRequestContainer.getResponseFuture().complete(new PlcReadResponse((PlcReadRequest) request, new ReadResponseItem((ReadRequestItem) requestItem, ResponseCode.OK, data)));
192         } else if (modbusPdu instanceof ReadDiscreteInputsResponse) {
193             // TODO: finish implementation
194             ReadDiscreteInputsResponse readDiscreteInputsResponse = (ReadDiscreteInputsResponse) modbusPdu;
195             LOGGER.debug("{}: Nothing", readDiscreteInputsResponse);
196             ByteBuf byteBuf = readDiscreteInputsResponse.getInputStatus();
197             List data = produceCoilValueList(requestItem, datatype, byteBuf);
198             plcRequestContainer.getResponseFuture().complete(new PlcReadResponse((PlcReadRequest) request, new ReadResponseItem((ReadRequestItem) requestItem, ResponseCode.OK, data)));
199         } else if (modbusPdu instanceof ReadHoldingRegistersResponse) {
200             // TODO: finish implementation
201             ReadHoldingRegistersResponse readHoldingRegistersResponse = (ReadHoldingRegistersResponse) modbusPdu;
202             LOGGER.debug("{}: Nothing", readHoldingRegistersResponse);
203             ByteBuf byteBuf = readHoldingRegistersResponse.getRegisters();
204             // TODO: use register method
205             List data = produceRegisterValueList(requestItem, datatype, byteBuf);
206             plcRequestContainer.getResponseFuture().complete(new PlcReadResponse((PlcReadRequest) request, new ReadResponseItem((ReadRequestItem) requestItem, ResponseCode.OK, data)));
207         } else if (modbusPdu instanceof ReadInputRegistersResponse) {
208             // TODO: finish implementation
209             ReadInputRegistersResponse readInputRegistersResponse = (ReadInputRegistersResponse) modbusPdu;
210             LOGGER.debug("{}: Nothing", readInputRegistersResponse);
211             ByteBuf byteBuf = readInputRegistersResponse.getRegisters();
212             // TODO: use register method
213             List data = produceRegisterValueList(requestItem, datatype, byteBuf);
214             plcRequestContainer.getResponseFuture().complete(new PlcReadResponse((PlcReadRequest) request, new ReadResponseItem((ReadRequestItem) requestItem, ResponseCode.OK, data)));
215         } else if (modbusPdu instanceof ExceptionResponse) {
216             ExceptionResponse exceptionResponse = (ExceptionResponse) modbusPdu;
217             throw new PlcProtocolException("Error received " + exceptionResponse.getExceptionCode());
218         } else {
219             throw new PlcProtocolException("Unsupported messageTyp type" + modbusPdu.getClass());
220         }
221     }

229     private void checkSupportedDataType(List<?> values) {
230         if (values == null || values.size() == 0) {
231             return;
232         }
233         for (Object value : values) {
234             if (
235                 !(value instanceof Boolean)
236                     && !(value instanceof Byte)
237                     && !(value instanceof byte[])
238                     && !(value instanceof Short)
239                     && !(value instanceof Integer)
240                 ) {
241                 throw new PlcRuntimeException("Unsupported datatype detected " + value.getClass());
242             }
243         }
244     }

249     private byte[] produceCoilValue(List<?> values) throws PlcProtocolException {
250         List<Byte> coils = new LinkedList<>();
251         Byte actualCoil = 0;
252         int i = 7;
253         for (Object value : values) {
254             boolean coilSet = false;
255             if (value.getClass() == Boolean.class) {
256                 coilSet = (Boolean) value;
257             } else if (value.getClass() == Byte.class) {
258                 if ((byte) value > 1) {
259                     throw new PlcProtocolException("Value to high to fit into Byte: " + value);
260                 }
261                 coilSet = (byte) value == 1;
262             } else if (value.getClass() == byte[].class) {
263                 byte[] bytes = (byte[]) value;
264                 if (bytes.length != 1) {
265                     throw new PlcProtocolException("Only exactly one byte is allowed: " + bytes.length);
266                 }
267                 byte byteValue = bytes[1];
268                 if (byteValue > 1) {
269                     throw new PlcProtocolException("Value to high to fit into Byte: " + value);
270                 }
271                 coilSet = byteValue == 1;
272             } else if (value.getClass() == Short.class) {
273                 if ((short) value > 1) {
274                     throw new PlcProtocolException("Value to high to fit into Byte: " + value);
275                 }
276                 coilSet = (Short) value == 1;
277             } else if (value.getClass() == Integer.class) {
278                 if ((int) value > 1) {
279                     throw new PlcProtocolException("Value to high to fit into Byte: " + value);
280                 }
281                 coilSet = (int) value == 1;
282             }
283             byte coilToSet = (coilSet ? (byte) 1 : (byte) 0);
284             actualCoil = (byte) (actualCoil | coilToSet << i);
285             i--;
286             if (i < 0) {
287                 coils.add(actualCoil);
288                 actualCoil = 0;
289                 i = 8;
290             }
291         }
292         // TODO: ensure we have a least (quantity + 7) / 8 = N bytes
293         return ArrayUtils.toPrimitive(coils.toArray(new Byte[0]));
294     }

296     private byte[] produceRegisterValue(List<?> values) throws PlcProtocolException {
297         ByteBuf buffer = Unpooled.buffer();
298         for (Object value : values) {
299             if (value.getClass() == Boolean.class) {
300                 buffer.writeByte(0);
301                 buffer.writeByte((boolean) value ? 1 : 0);
302             } else if (value.getClass() == Byte.class) {
303                 buffer.writeByte(0);
304                 buffer.writeByte((byte) value);
305             } else if (value.getClass() == byte[].class) {
306                 byte[] bytes = (byte[]) value;
307                 if (bytes.length != 2) {
308                     throw new PlcProtocolException("Only exactly two bytes are allowed: " + bytes.length);
309                 }
310                 buffer.writeBytes(bytes);
311             } else if (value.getClass() == Short.class) {
312                 buffer.writeShort((int) value);
313             } else if (value.getClass() == Integer.class) {
314                 if ((int) value > Integer.MAX_VALUE) {
315                     throw new PlcProtocolException("Value to high to fit into register: " + value);
316                 }
317                 buffer.writeShort((int) value);
318             }
319         }
320         // TODO: ensure we have a least quantity * 2 = N bytes
321         byte[] result = new byte[buffer.writerIndex()];
322         buffer.readBytes(result);
323         return result;
324     }

330     @SuppressWarnings("unchecked")
331     private List produceCoilValueList(RequestItem requestItem, Class datatype, ByteBuf byteBuf) {
332         ReadRequestItem readRequestItem = (ReadRequestItem) requestItem;
333         byte[] bytes = new byte[byteBuf.readableBytes()];
334         byteBuf.readBytes(bytes);
335         List data = new LinkedList();
336         for (int i = 0, j = 0; i < readRequestItem.getSize(); i++) {
337             if (i != 0 && i % 8 == 0) {
338                 // Every 8 Coils we need to increase the access
339                 j++;
340             }
341             Boolean coilValue = (1 << i & bytes[j]) == 1;
342             if (datatype == Boolean.class) {
343                 data.add(coilValue);
344             } else if (datatype == Byte.class) {
345                 data.add((byte) (coilValue ? 1 : 0));
346             } else if (datatype == byte[].class) {
347                 data.add(new byte[]{(byte) (coilValue ? 1 : 0)});
348             } else if (datatype == Short.class) {
349                 data.add((short) (coilValue ? 1 : 0));
350             } else if (datatype == Integer.class) {
351                 data.add(coilValue ? 1 : 0);
352             }
353         }
354         return data;
355     }

357     @SuppressWarnings("unchecked")
358     private List produceRegisterValueList(RequestItem requestItem, Class datatype, ByteBuf byteBuf) throws PlcProtocolException {
359         ReadRequestItem readRequestItem = (ReadRequestItem) requestItem;
360         int readableBytes = byteBuf.readableBytes();
361         if (readableBytes % 2 != 0) {
362             throw new PlcProtocolException("Readables bytes should even: " + readableBytes);
363         }
364         List data = new LinkedList();
365         for (int i = 0; i < readRequestItem.getSize(); i++) {
366             byte[] register = new byte[2];
367             byteBuf.readBytes(register);
368             int intValue = register[0] << 8 | register[1];
369             if (datatype == Boolean.class) {
370                 data.add(intValue == 1);
371             } else if (datatype == Byte.class) {
372                 if (intValue > Byte.MAX_VALUE) {
373                     throw new PlcProtocolException("Value to high to fit into Byte: " + intValue);
374                 }
375                 data.add((byte) intValue);
376             } else if (datatype == byte[].class) {
377                 data.add(register);
378             } else if (datatype == Short.class) {
379                 if (intValue > Short.MAX_VALUE) {
380                     throw new PlcProtocolException("Value to high to fit into Short: " + intValue);
381                 }
382                 data.add((short) intValue);
383             } else if (datatype == Integer.class) {
384                 data.add(intValue);
385             }
386         }
387         return data;
388     }
