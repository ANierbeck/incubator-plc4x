31  import static org.hamcrest.Matchers.equalTo;

34  public class Plc4XSupportedDataTypes {
35  
36      private final static Map<Class, DataTypePair<?>> littleEndianMap;
37      private final static Map<Class, DataTypePair<?>> bigEndianMap;
38  
39      static {
40          Calendar calenderInstance = Calendar.getInstance();
41          calenderInstance.setTime(new Date(283686951976960L));
42          littleEndianMap = new HashMap<>();
43          littleEndianMap.put(Boolean.class, DataTypePair.of(Boolean.TRUE, new byte[]{0x01}));
44          littleEndianMap.put(Byte.class, DataTypePair.of(Byte.valueOf("1"), new byte[]{0x1}));
45          littleEndianMap.put(Short.class, DataTypePair.of(Short.valueOf("1"), new byte[]{0x1, 0x0}));
46          littleEndianMap.put(Float.class, DataTypePair.of(Float.valueOf("1"), new byte[]{0x0, 0x0, (byte) 0x80, 0x3F}));
47          littleEndianMap.put(Integer.class, DataTypePair.of(Integer.valueOf("1"), new byte[]{0x1, 0x0, 0x0, 0x0}));
48          littleEndianMap.put(Double.class, DataTypePair.of(Double.valueOf("1"), new byte[]{0x0, 0x0, 0x0, 0x0, 0x0, 0x0, (byte) 0xF0, 0x3F}));
49          littleEndianMap.put(BigInteger.class, DataTypePair.of(BigInteger.valueOf(1), new byte[]{0x1, 0x0, 0x0, 0x0}));
50          littleEndianMap.put(Calendar.class, DataTypePair.of(calenderInstance, new byte[]{0x0, (byte) 0x80, 0x3E, 0x15, (byte) 0xAB, 0x47, (byte) 0xFC, 0x28}));
51          littleEndianMap.put(GregorianCalendar.class, littleEndianMap.get(Calendar.class));
52          littleEndianMap.put(String.class, DataTypePair.of(String.valueOf("Hello World!"), new byte[]{0x48, 0x65, 0x6c, 0x6c, 0x6f, 0x20, 0x57, 0x6f, 0x72, 0x6c, 0x64, 0x21, 0x00}));
53          littleEndianMap.put(byte[].class, DataTypePair.of(new byte[]{0x1, 0x2, 0x3, 0x4}, new byte[]{0x1, 0x2, 0x3, 0x4}));
54          littleEndianMap.put(Byte[].class, DataTypePair.of(new Byte[]{0x1, 0x2, 0x3, 0x4}, new byte[]{0x1, 0x2, 0x3, 0x4}));
55          bigEndianMap = new HashMap<>();
56          littleEndianMap.forEach((clazz, pair) -> {
57              Serializable serializable = pair.getValue();
58              byte[] littleEndianBytes = pair.getByteRepresentation();
59              byte[] bigEndianBytes = ArrayUtils.clone(littleEndianBytes);
60              ArrayUtils.reverse(bigEndianBytes);
61              bigEndianMap.put(clazz, DataTypePair.of(serializable, bigEndianBytes));
62          });
63      }
64  
65      /**
66       * A {@link Stream} of {@link Class}es plc4x can currently support.
67       *
68       * @return a stream of supported data types.
69       */
70      public static Stream<Class<? extends Serializable>> streamOfPlc4XSupportedDataTypes() {
71          return Stream.of(
72              Boolean.class,
73              Byte.class,
74              Short.class,
75              Float.class,
76              Integer.class,
77              Double.class,
78              BigInteger.class,
79              Calendar.class,
80              String.class,
81              byte[].class,
82              Byte[].class
83          );
84      }
85  
86      /**
87       * A {@link Stream} of instances of {@link Class}es plc4x can currently support with their according little endian byte representation.
88       *
89       * @return a stream of {@link DataTypePair}s of instances and their byte values.
90       * @see #streamOfPlc4XSupportedDataTypes
91       */
92      public static Stream<? extends DataTypePair<?>> streamOfLittleEndianDataTypePairs() {
93          return streamOfLittleEndianDataTypePairs(streamOfPlc4XSupportedDataTypes());
94      }
95  
96      /**
97       * A {@link Stream} of instances of {@link Class}es which are defined by {@code inputStream} can currently support with their according little endian byte representation.
98       *
99       * @param inputStream a stream of {@link DataTypePair}s of instances and their byte values.
100      * @see #streamOfPlc4XSupportedDataTypes
101      */
102     public static Stream<? extends DataTypePair<?>> streamOfLittleEndianDataTypePairs(Stream<Class<? extends Serializable>> inputStream) {
103         return inputStream
104             .map(littleEndianMap::get)
105             .peek(Objects::requireNonNull);
106     }
107 
108     /**
109      * A {@link Stream} of instances of {@link Class}es plc4x can currently support with their according big endian byte representation.
110      *
111      * @return a stream of {@link DataTypePair}s of instances and their byte values.
112      * @see #streamOfPlc4XSupportedDataTypes
113      */
114     public static Stream<? extends DataTypePair<?>> streamOfBigEndianDataTypePairs() {
115         return streamOfBigEndianDataTypePairs(streamOfPlc4XSupportedDataTypes());
116     }

118     /**
119      * A {@link Stream} of instances of {@link Class}es which are defined by {@code inputStream} can currently support with their according big endian byte representation.
120      *
121      * @param inputStream a stream of {@link DataTypePair}s of instances and their byte values.
122      * @see #streamOfPlc4XSupportedDataTypes
123      */
124     public static Stream<? extends DataTypePair<?>> streamOfBigEndianDataTypePairs(Stream<Class<? extends Serializable>> inputStream) {
125         return inputStream
126             .map(bigEndianMap::get)
127             .peek(Objects::requireNonNull);
128     }

130     /**
131      * Returns default value for supplied {@code clazz}.
132      *
133      * @param clazz the default value to get.
134      * @param <T>   the type of {@link Class}
135      * @return the found default.
136      */
137     @SuppressWarnings("unchecked")
138     public static <T> Optional<T> getDefaultForClass(Class<T> clazz) {
139         DataTypePair<?> pair = littleEndianMap.get(clazz);
140         if (pair == null) {
141             return Optional.empty();
142         }
143         return Optional.of((T) pair.getValue());
144     }
145 
146     /**
147      * A method which compares a value against a well known default.
148      *
149      * @param actualValue the value to check.
150      */
151     public static void defaultAssert(Object actualValue) {
152         littleEndianMap.values().forEach(pair -> assertPayloadDependentEquals(actualValue, pair.getValue()));
153     }
154 
155     private static void assertPayloadDependentEquals(Object actual, Object expected) {
156         if (actual.getClass() != expected.getClass()) {
157             return;
158         }
159         assertThat(actual, equalTo(expected));
160     }

162     /**
163      * An wrapper for {@link Pair} that make the usage a bit more readable downstream.
164      *
165      * @param <T> the type of the contained data type.
166      */
167     public static class DataTypePair<T extends Serializable> {
168         private final Pair<T, byte[]> dataTypePair;
169 
170         private DataTypePair(Pair<T, byte[]> dataTypePair) {
171             this.dataTypePair = dataTypePair;
172         }
173 
174         public static <T extends Serializable> DataTypePair<T> of(T value, byte[] bytes) {
175             return new DataTypePair<>(ImmutablePair.of(value, bytes));
176         }
177 
178         /**
179          * @return the value of the data type.
180          */
181         public T getValue() {
182             return SerializationUtils.clone(dataTypePair.getLeft());
183         }
184 
185         /**
186          * @return the {@link Class} of the data type.
187          */
188         public Class<?> getDataTypeClass() {
189             return dataTypePair.getLeft().getClass();
190         }
191 
192         /**
193          * @return The byte representation of the data type according to the endianness.
194          */
195         public byte[] getByteRepresentation() {
196             return ArrayUtils.clone(dataTypePair.getRight());
197         }
198 
199     }
