19  package org.apache.plc4x.java.test;
20  
21  import java.util.*;
22  
23  /**
24   * Test device storing its state in memory.
25   * Values are stored in a HashMap.
26   */
27  class TestDevice {
28      private final Random random = new Random();
29      private final String name;
30      private final Map<TestAddress, Object> state = new HashMap<>();
31  
32      TestDevice(String name) {
33          this.name = name;
34      }
35  
36      @SuppressWarnings("unchecked")
37      <T> Optional<T> get(Class<? extends T> type, TestAddress address) {
38          Objects.requireNonNull(address);
39          if (address.equals(TestAddress.RANDOM)) {
40              return Optional.of(randomValue(type));
41          } else {
42              return Optional.ofNullable((T) state.get(address));
43          }
44      }
45  
46      public void set(TestAddress address, Object value) {
47          state.put(address, value);
48      }
49  
50      @SuppressWarnings("unchecked")
51      private <T> T randomValue(Class<T> type) {
52          Object result = null;
53  
54          // TODO: implement for further data types
55  
56          if (type == Integer.class)
57              result = random.nextInt();
58  
59          if (type == Byte.class) {
60              byte[] bytes = new byte[1];
61              random.nextBytes(bytes);
62              result = bytes[0];
63          }
64  
65          if (type == Short.class) {
66              result = random.nextInt(1 << 16);
67          }
68  
69          return (T) result;
70      }
71  
72      @Override
73      public String toString() {
74          return name;
75      }
76  }
