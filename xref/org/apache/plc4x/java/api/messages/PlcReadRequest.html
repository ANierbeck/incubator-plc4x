19  package org.apache.plc4x.java.api.messages;
20  
21  import org.apache.plc4x.java.api.messages.items.ReadRequestItem;
22  import org.apache.plc4x.java.api.messages.specific.TypeSafePlcReadRequest;
23  import org.apache.plc4x.java.api.model.Address;
24  
25  import java.util.List;
26  import java.util.Objects;
27  
28  /**
29   * Request to read one or more values from a plc.
30   * The {@link PlcReadRequest} is a container for one or more {@link ReadRequestItem}s that are contained.
31   *
32   * By default this is NOT typesafe as it can contain {@link ReadRequestItem}s for different types.
33   * If there are only {@link ReadRequestItem}s of the same type one can use the {@link TypeSafePlcReadRequest} to enforce
34   * type safety.
35   *
36   * Provides a builder for object construction through {@link PlcReadRequest#builder()}.
37   *
38   * @see TypeSafePlcReadRequest
39   */
40  public class PlcReadRequest extends PlcRequest<ReadRequestItem<?>> {
41  
42      public PlcReadRequest() {
43          // Exists for construction of inherited classes
44      }
45  
46      public PlcReadRequest(ReadRequestItem<?> requestItem) {
47          addItem(requestItem);
48      }
49  
50      public PlcReadRequest(Class<?> dataType, Address address) {
51          addItem(new ReadRequestItem<>(dataType, address));
52      }
53  
54      public PlcReadRequest(Class<?> dataType, Address address, int size) {
55          Objects.requireNonNull(dataType, "Data type must not be null");
56          Objects.requireNonNull(address, "Address must not be null");
57          addItem(new ReadRequestItem<>(dataType, address, size));
58      }
59  
60      public PlcReadRequest(List<ReadRequestItem<?>> requestItems) {
61          super(requestItems);
62      }
63  
64      public static Builder builder() {
65          return new Builder();
66      }
67  
68      public static class Builder extends PlcRequest.Builder<ReadRequestItem<?>> {
69  
70          public final Builder addItem(Class<?> dataType, Address address) {
71              checkType(dataType);
72              requests.add(new ReadRequestItem<>(dataType, address));
73              return this;
74          }
75  
76          public final Builder addItem(Class<?> dataType, Address address, int size) {
77              checkType(dataType);
78              requests.add(new ReadRequestItem<>(dataType, address, size));
79              return this;
80          }
81  
82          public final Builder addItem(ReadRequestItem readRequestItem) {
83              checkType(readRequestItem.getDatatype());
84              requests.add(readRequestItem);
85              return this;
86          }
87  
88          public final PlcReadRequest build() {
89              if (requests.isEmpty()) {
90                  throw new IllegalStateException("No requests added");
91              }
92              PlcReadRequest plcReadRequest;
93              if (mixed) {
94                  plcReadRequest = new PlcReadRequest();
95              } else {
96                  plcReadRequest = new TypeSafePlcReadRequest<>(firstType);
97              }
98              for (ReadRequestItem request : requests) {
99                  plcReadRequest.addItem(request);
100             }
101             return plcReadRequest;
102         }
103 
104         @SuppressWarnings("unchecked")
105         public final <T> TypeSafePlcReadRequest<T> build(Class<T> type) {
106             if (firstType != type) {
107                 throw new ClassCastException("Incompatible type " + type + ". Required " + firstType);
108             }
109             if (mixed) {
110                 throw new IllegalStateException("Mixed types contained");
111             }
112             return (TypeSafePlcReadRequest<T>) build();
113         }
114 
115     }
116 
117 }
