54      private static final AtomicInteger tpduGenerator = new AtomicInteger(1);

56      private Map<Short, PlcRequestContainer> requests;

58      public Plc4XS7Protocol() {
59          this.requests = new HashMap<>();

62      @Override
63      protected void encode(ChannelHandlerContext ctx, PlcRequestContainer msg, List<Object> out) throws Exception {
64          PlcRequest request = msg.getRequest();
65          if (request instanceof PlcReadRequest) {
66              encodeReadRequest(msg, out);
67          } else if (request instanceof PlcWriteRequest) {
68              encodeWriteRequest(msg, out);
69          }
70      }

72      private void encodeWriteRequest(PlcRequestContainer msg, List<Object> out) throws PlcException {
73          List<VarParameterItem> parameterItems = new LinkedList<>();
74          List<VarPayloadItem> payloadItems = new LinkedList<>();
75  
76          PlcWriteRequest writeRequest = (PlcWriteRequest) msg.getRequest();
77          for (WriteRequestItem requestItem : writeRequest.getRequestItems()) {
78              // Try to get the correct S7 transport size for the given data type.
79              // (Map PLC4X data type to S7 data type)
80              TransportSize transportSize = encodeTransportSize(requestItem.getDatatype());
81              if (transportSize == null) {
82                  throw new PlcException("Unknown transport size for datatype " + requestItem.getDatatype());
83              }
84  
85              // Depending on the address type, generate the corresponding type of request item.
86              VarParameterItem varParameterItem = encodeVarParameterItem(
87                  requestItem.getAddress(), transportSize, requestItem.getValues().size());
88              parameterItems.add(varParameterItem);
89  
90              DataTransportSize dataTransportSize = encodeDataTransportSize(requestItem.getDatatype());
91              if (dataTransportSize == null) {
92                  throw new PlcException("Unknown data transport size for datatype " + requestItem.getDatatype());
93              }
94  
95              VarPayloadItem varPayloadItem = new VarPayloadItem(
96                  DataTransportErrorCode.RESERVED, dataTransportSize, encodeData(requestItem.getValues().toArray()));
97  
98              payloadItems.add(varPayloadItem);
99          }
100         VarParameter writeVarParameter = new VarParameter(ParameterType.WRITE_VAR, parameterItems);
101         VarPayload writeVarPayload = new VarPayload(ParameterType.WRITE_VAR, payloadItems);
102 
103         // Assemble the request.
104         S7RequestMessage s7WriteRequest = new S7RequestMessage(MessageType.JOB,
105             (short) tpduGenerator.getAndIncrement(), Collections.singletonList(writeVarParameter),
106             Collections.singletonList(writeVarPayload));
107 
108         requests.put(s7WriteRequest.getTpduReference(), msg);
109 
110         out.add(s7WriteRequest);
111     }

113     private void encodeReadRequest(PlcRequestContainer msg, List<Object> out) throws PlcException {
114         List<VarParameterItem> parameterItems = new LinkedList<>();
115 
116         PlcReadRequest readRequest = (PlcReadRequest) msg.getRequest();
117         encodeParameterItems(parameterItems, readRequest);
118         VarParameter readVarParameter = new VarParameter(ParameterType.READ_VAR, parameterItems);
119 
120         // Assemble the request.
121         S7RequestMessage s7ReadRequest = new S7RequestMessage(MessageType.JOB,
122             (short) tpduGenerator.getAndIncrement(), Collections.singletonList(readVarParameter),
123             Collections.emptyList());
124 
125         requests.put(s7ReadRequest.getTpduReference(), msg);
126 
127         out.add(s7ReadRequest);
128     }

130     private void encodeParameterItems(List<VarParameterItem> parameterItems, PlcReadRequest readRequest) throws PlcException {
131         for (ReadRequestItem requestItem : readRequest.getRequestItems()) {
132             // Try to get the correct S7 transport size for the given data type.
133             // (Map PLC4X data type to S7 data type)
134             TransportSize transportSize = encodeTransportSize(requestItem.getDatatype());
135             if (transportSize == null) {
136                 throw new PlcException("Unknown transport size for datatype " + requestItem.getDatatype());
137             }
138 
139             // Depending on the address type, generate the corresponding type of request item.
140             VarParameterItem varParameterItem = encodeVarParameterItem(requestItem.getAddress(), transportSize, requestItem.getSize());
141             parameterItems.add(varParameterItem);
142         }
143     }

145     @SuppressWarnings("unchecked")
146     @Override
147     protected void decode(ChannelHandlerContext ctx, S7Message msg, List<Object> out) throws Exception {
148         if (!(msg instanceof S7ResponseMessage)) {
149             return;
150         }
151         S7ResponseMessage responseMessage = (S7ResponseMessage) msg;
152         short tpduReference = responseMessage.getTpduReference();
153         if (requests.containsKey(tpduReference)) {
154             PlcRequestContainer requestContainer = requests.remove(tpduReference);
155             PlcRequest request = requestContainer.getRequest();
156             PlcResponse response = null;
157 
158             // Handle the response to a read request.
159             if (request instanceof PlcReadRequest) {
160                 response = decodeReadRequest(responseMessage, requestContainer);
161             }
162             else if (request instanceof PlcWriteRequest) {
163                 response = decodeWriteRequest(responseMessage, requestContainer);
164             }
165 
166             // Confirm the response being handled.
167             if (response != null) {
168                 requestContainer.getResponseFuture().complete(response);
169             }
170         }
171     }

173     @SuppressWarnings("unchecked")
174     private PlcResponse decodeWriteRequest(S7ResponseMessage responseMessage, PlcRequestContainer requestContainer) throws PlcProtocolException {
175         PlcResponse response;
176         PlcWriteRequest plcWriteRequest = (PlcWriteRequest) requestContainer.getRequest();
177         List<WriteResponseItem<?>> responseItems = new LinkedList<>();
178         VarPayload payload = responseMessage.getPayload(VarPayload.class)
179             .orElseThrow(() -> new PlcProtocolException("No VarPayload supplied"));
180         // If the numbers of items don't match, we're in big trouble as the only
181         // way to know how to interpret the responses is by aligning them with the
182         // items from the request as this information is not returned by the PLC.
183         if (plcWriteRequest.getRequestItems().size() != payload.getPayloadItems().size()) {
184             throw new PlcProtocolException(
185                 "The number of requested items doesn't match the number of returned items");
186         }
187         List<VarPayloadItem> payloadItems = payload.getPayloadItems();
188         final int noPayLoadItems = payloadItems.size();
189         for (int i = 0; i < noPayLoadItems; i++) {
190             VarPayloadItem payloadItem = payloadItems.get(i);
191 
192             // Get the request item for this payload item
193             WriteRequestItem requestItem = plcWriteRequest.getRequestItems().get(i);
194 
195             // A write response contains only the return code for every item.
196             ResponseCode responseCode = decodeResponseCode(payloadItem.getReturnCode());
197 
198             WriteResponseItem responseItem = new WriteResponseItem(requestItem, responseCode);
199             responseItems.add(responseItem);
200         }
201 
202         if (plcWriteRequest instanceof TypeSafePlcWriteRequest) {
203             response = new TypeSafePlcWriteResponse((TypeSafePlcWriteRequest) plcWriteRequest, responseItems);
204         } else {
205             response = new PlcWriteResponse(plcWriteRequest, responseItems);
206         }
207         return response;
208     }

210     @SuppressWarnings("unchecked")
211     private PlcResponse decodeReadRequest(S7ResponseMessage responseMessage, PlcRequestContainer requestContainer) throws PlcProtocolException {
212         PlcResponse response;
213         PlcReadRequest plcReadRequest = (PlcReadRequest) requestContainer.getRequest();
214 
215         List<ReadResponseItem<?>> responseItems = new LinkedList<>();
216         VarPayload payload = responseMessage.getPayload(VarPayload.class)
217             .orElseThrow(() -> new PlcProtocolException("No VarPayload supplied"));
218         // If the numbers of items don't match, we're in big trouble as the only
219         // way to know how to interpret the responses is by aligning them with the
220         // items from the request as this information is not returned by the PLC.
221         if (plcReadRequest.getRequestItems().size() != payload.getPayloadItems().size()) {
222             throw new PlcProtocolException(
223                 "The number of requested items doesn't match the number of returned items");
224         }
225         List<VarPayloadItem> payloadItems = payload.getPayloadItems();
226         final int noPayLoadItems = payloadItems.size();
227         for (int i = 0; i < noPayLoadItems; i++) {
228             VarPayloadItem payloadItem = payloadItems.get(i);
229 
230             // Get the request item for this payload item
231             ReadRequestItem requestItem = plcReadRequest.getRequestItems().get(i);
232 
233             ResponseCode responseCode = decodeResponseCode(payloadItem.getReturnCode());
234 
235             ReadResponseItem responseItem;
236             // Something went wrong.
237             if (responseCode != ResponseCode.OK) {
238                 responseItem = new ReadResponseItem<>(requestItem, responseCode, null);
239             }
240             // All Ok.
241             else {
242                 byte[] data = payloadItem.getData();
243                 Class<?> datatype = requestItem.getDatatype();
244                 List<?> value = decodeData(datatype, data);
245                 responseItem = new ReadResponseItem(requestItem, responseCode, value);
246             }
247             responseItems.add(responseItem);
248         }
249         if (plcReadRequest instanceof TypeSafePlcReadRequest) {
250             response = new TypeSafePlcReadResponse((TypeSafePlcReadRequest) plcReadRequest, responseItems);
251         } else {
252             response = new PlcReadResponse(plcReadRequest, responseItems);
253         }
254         return response;
255     }

261     private VarParameterItem encodeVarParameterItem(Address address, TransportSize transportSize, int size) throws PlcProtocolException {
262         // Depending on the address type, generate the corresponding type of request item.
263         if (!(address instanceof S7Address)) {
264             throw new PlcProtocolException("Can only use S7Address types on S7 connection");
265         }
266         S7Address s7Address = (S7Address) address;
267         if (s7Address instanceof S7DataBlockAddress) {
268             S7DataBlockAddress s7DataBlockAddress = (S7DataBlockAddress) s7Address;
269             return new S7AnyVarParameterItem(
270                 SpecificationType.VARIABLE_SPECIFICATION, s7Address.getMemoryArea(),
271                 transportSize, (short) size,
272                 s7DataBlockAddress.getDataBlockNumber(), s7DataBlockAddress.getByteOffset(), (byte) 0);
273         } else if (s7Address instanceof S7BitAddress) {
274             S7BitAddress s7BitAddress = (S7BitAddress) s7Address;
275             return new S7AnyVarParameterItem(
276                 SpecificationType.VARIABLE_SPECIFICATION, s7Address.getMemoryArea(),
277                 transportSize, (short) size, (short) 0,
278                 s7Address.getByteOffset(), s7BitAddress.getBitOffset());
279         } else {
280             return new S7AnyVarParameterItem(
281                 SpecificationType.VARIABLE_SPECIFICATION, s7Address.getMemoryArea(),
282                 transportSize, (short) size, (short) 0,
283                 s7Address.getByteOffset(), (byte) 0);
284         }
285     }

287     private TransportSize encodeTransportSize(Class<?> datatype) {
288         if (datatype == Boolean.class) {
289             return TransportSize.BIT;
290         } else if (datatype == Byte.class) {
291             return TransportSize.BYTE;
292         } else if (datatype == Short.class) {
293             return TransportSize.WORD;
294         } else if (datatype == Calendar.class) {
295             return TransportSize.DATE_AND_TIME;
296         } else if (datatype == Float.class) {
297             return TransportSize.REAL;
298         } else if (datatype == Integer.class) {
299             return TransportSize.DWORD;
300         } else if (datatype == String.class) {
301             return TransportSize.CHAR;
302         }
303         return null;
304     }

306     private DataTransportSize encodeDataTransportSize(Class<?> datatype) {
307         if (datatype == Boolean.class) {
308             return DataTransportSize.BIT;
309         } else if (datatype == Byte.class) {
310             return DataTransportSize.BYTE_WORD_DWORD;
311         } else if (datatype == Short.class) {
312             return DataTransportSize.BYTE_WORD_DWORD;
313         } else if (datatype == Calendar.class) {
314             // TODO: Decide what to do here ...
315             return null;
316         } else if (datatype == Float.class) {
317             return DataTransportSize.REAL;
318         } else if (datatype == Integer.class) {
319             return DataTransportSize.BYTE_WORD_DWORD;
320         } else if (datatype == String.class) {
321             return DataTransportSize.OCTET_STRING;
322         }
323         return null;
324     }

326     private byte[] encodeData(Object[] values) {
327         final int length = values.length;
328         if (length == 0) {
329             return new byte[]{};
330         }
331         byte[] result = null;
332         Class valueType = values[0].getClass();
333         if (valueType == Boolean.class) {
334             result = encodeBoolean(values, length);
335         } else if (valueType == Byte.class) {
336             result = encodeByte(values, length);
337         } else if (valueType == Short.class) {
338             result = encodeShort(values, length);
339         } else if (valueType == Integer.class) {
340             result = encodeInteger(values, length);
341         } else if (valueType == Calendar.class) {
342             // TODO: Decide what to do here ...
343             result = null;
344         } else if (valueType == Float.class) {
345             result = encodeFloat(values, length);
346         } else if (valueType == String.class) {
347             result = encodeString(values, length);
348         }
349         return result;
350     }

352     private byte[] encodeString(Object[] values, int length) {
353         byte[] result;
354         int size = 0;
355         for (Object value : values) {
356             size = size + ((String) value).length();
357         }
358         result = new byte[size + length];
359         int j = 0;
360         for (Object value : values) {
361             String str = (String) value;
362             for (int i = 0; i < str.length(); i++) {
363                 result[j++] = (byte) str.charAt(i);
364             }
365             result[j++] = (byte) 0x0;
366         }
367         return result;
368     }

370     private byte[] encodeFloat(Object[] values, int length) {
371         byte[] result;
372         result = new byte[length * 4];
373         for (int i = 0; i < length; i++) {
374             float floatValue = (float) values[i];
375             int intValue = Float.floatToIntBits(floatValue);
376             result[i * 4] = (byte) ((intValue & 0xff000000) >> 24);
377             result[(i * 4) + 1] = (byte) ((intValue & 0x00ff0000) >> 16);
378             result[(i * 4) + 2] = (byte) ((intValue & 0x0000ff00) >> 8);
379             result[(i * 4) + 3] = (byte) (intValue & 0xff);
380         }
381         return result;
382     }

384     private byte[] encodeInteger(Object[] values, int length) {
385         byte[] result;
386         result = new byte[length * 4];
387         for (int i = 0; i < length; i++) {
388             int intValue = (int) values[i];
389             result[i * 4] = (byte) ((intValue & 0xff000000) >> 24);
390             result[(i * 4) + 1] = (byte) ((intValue & 0x00ff0000) >> 16);
391             result[(i * 4) + 2] = (byte) ((intValue & 0x0000ff00) >> 8);
392             result[(i * 4) + 3] = (byte) (intValue & 0xff);
393         }
394         return result;
395     }

397     private byte[] encodeShort(Object[] values, int length) {
398         byte[] result;
399         result = new byte[length * 2];
400         for (int i = 0; i < length; i++) {
401             short intValue = (short) values[i];
402             result[i * 2] = (byte) ((intValue & 0xff00) >> 8);
403             result[(i * 2) + 1] = (byte) (intValue & 0xff);
404         }
405         return result;
406     }

408     private byte[] encodeByte(Object[] values, int length) {
409         byte[] result;
410         result = new byte[length];
411         for (int i = 0; i < length; i++) {
412             result[i] = (byte) values[i];
413         }
414         return result;
415     }

417     private byte[] encodeBoolean(Object[] values, int length) {
418         byte[] result;// TODO: Check if this is true and the result is not Math.ceil(values.lenght / 8)
419         result = new byte[length];
420         for (int i = 0; i < length; i++) {
421             result[i] = (byte) (((Boolean) values[i]) ? 0x01 : 0x00);
422         }
423         return result;

430     private ResponseCode decodeResponseCode(DataTransportErrorCode dataTransportErrorCode) {
431         if (dataTransportErrorCode == null) {
432             return ResponseCode.INTERNAL_ERROR;
433         }
434         switch (dataTransportErrorCode) {
435             case OK:
436                 return ResponseCode.OK;
437             case NOT_FOUND:
438                 return ResponseCode.NOT_FOUND;
439             case INVALID_ADDRESS:
440                 return ResponseCode.INVALID_ADDRESS;
441             default:
442                 return ResponseCode.INTERNAL_ERROR;
443         }
444     }

446     @SuppressWarnings("unchecked")
447     private <T> List<T> decodeData(Class<T> datatype, byte[] s7Data) throws PlcProtocolException {
448 
449         List<Object> result = new LinkedList<>();
450         int i = 0;
451         final int length = s7Data.length;
452         while (i < length) {
453             if (datatype == Boolean.class) {
454                 result.add((s7Data[i] & 0x01) == 0x01);
455                 i += 1;
456             } else if (datatype == Byte.class) {
457                 result.add(s7Data[i]);
458                 i += 1;
459             } else if (datatype == Short.class) {
460                 result.add((short) (((s7Data[i] & 0xff) << 8) | (s7Data[i + 1] & 0xff)));
461                 i += 2;
462             } else if (datatype == Integer.class) {
463                 result.add(((s7Data[i] & 0xff) << 24) | ((s7Data[i + 1] & 0xff) << 16) |
464                     ((s7Data[i + 2] & 0xff) << 8) | (s7Data[i + 3] & 0xff));
465                 i += 4;
466             } else if (datatype == Float.class) {
467                 // Description of the Real number format:
468                 // https://www.sps-lehrgang.de/zahlenformate-step7/#c144
469                 // https://de.wikipedia.org/wiki/IEEE_754
470                 int intValue = ((s7Data[i] & 0xff) << 24) | ((s7Data[i + 1] & 0xff) << 16) |
471                     ((s7Data[i + 2] & 0xff) << 8) | (s7Data[i + 3] & 0xff);
472                 result.add(Float.intBitsToFloat(intValue));
473                 i += 4;
474             } else if (datatype == String.class) {
475                 StringBuilder builder = new StringBuilder();
476                 while (s7Data[i] != (byte) 0x0 && i < length) {
477                     builder.append((char) s7Data[i]);
478                     i++;
479                 }
480                 i++; // skip terminating character
481                 result.add(builder.toString());
482             } else {
483                 throw new PlcProtocolException("Unsupported datatype " + datatype.getSimpleName());
484             }
485         }
486         return (List<T>) result;
487     }
