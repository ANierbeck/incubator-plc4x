19  package org.apache.plc4x.java.s7.netty.util;

21  import org.apache.plc4x.java.api.exceptions.PlcNotImplementedException;
22  import org.apache.plc4x.java.api.exceptions.PlcUnsupportedDataTypeException;
23  
24  import java.util.Calendar;
25  
26  public class S7TypeEncoder {
27  
28      private S7TypeEncoder() {
29          // Utility class
30      }
31  
32      public static byte[] encodeData(Object[] values) {
33          final int length = values.length;
34          if (length == 0) {
35              return new byte[]{};
36          }
37          final byte[] result;
38          Class valueType = values[0].getClass();
39          if (valueType == Boolean.class) {
40              result = encodeBoolean(values, length);
41          } else if (valueType == Byte.class) {
42              result = encodeByte(values, length);
43          } else if (valueType == Short.class) {
44              result = encodeShort(values, length);
45          } else if (valueType == Integer.class) {
46              result = encodeInteger(values, length);
47          } else if (valueType == Calendar.class) {
48              // TODO: Decide what to do here ...
49              throw new PlcNotImplementedException("calender not yet implemented in s7");
50          } else if (valueType == Float.class) {
51              result = encodeFloat(values, length);
52          } else if (valueType == Double.class) {
53              result = encodeDouble(values, length);
54          } else if (valueType == String.class) {
55              result = encodeString(values, length);
56          } else {
57              throw new PlcUnsupportedDataTypeException(valueType);
58          }
59          return result;
60      }
61  
62      public static byte[] encodeString(Object[] values, int length) {
63          byte[] result;
64          int size = 0;
65          for (Object value : values) {
66              size = size + ((String) value).length();
67          }
68          result = new byte[size + length];
69          int j = 0;
70          for (Object value : values) {
71              String str = (String) value;
72              for (int i = 0; i < str.length(); i++) {
73                  result[j++] = (byte) str.charAt(i);
74              }
75              result[j++] = (byte) 0x0;
76          }
77          return result;
78      }
79  
80      public static byte[] encodeFloat(Object[] values, int length) {
81          byte[] result;
82          result = new byte[length * 4];
83          for (int i = 0; i < length; i++) {
84              float floatValue = (float) values[i];
85              int intValue = Float.floatToIntBits(floatValue);
86              result[i * 4] = (byte) ((intValue & 0xff000000) >> 24);
87              result[(i * 4) + 1] = (byte) ((intValue & 0x00ff0000) >> 16);
88              result[(i * 4) + 2] = (byte) ((intValue & 0x0000ff00) >> 8);
89              result[(i * 4) + 3] = (byte) (intValue & 0xff);
90          }
91          return result;
92      }
93  
94      public static byte[] encodeDouble(Object[] values, int length) {
95          byte[] result;
96          result = new byte[length * 8];
97          for (int i = 0; i < length; i++) {
98              double doubleValue = (double) values[i];
99              long longValue = Double.doubleToLongBits(doubleValue);
100             result[i * 8] = (byte) ((longValue & 0xFF000000_00000000L) >> 56);
101             result[(i * 8) + 1] = (byte) ((longValue & 0x00FF0000_00000000L) >> 48);
102             result[(i * 8) + 2] = (byte) ((longValue & 0x0000FF00_00000000L) >> 40);
103             result[(i * 8) + 3] = (byte) ((longValue & 0x000000FF_00000000L) >> 32);
104             result[(i * 8) + 4] = (byte) ((longValue & 0x00000000_FF000000L) >> 24);
105             result[(i * 8) + 5] = (byte) ((longValue & 0x00000000_00FF0000L) >> 16);
106             result[(i * 8) + 6] = (byte) ((longValue & 0x00000000_0000FF00L) >> 8);
107             result[(i * 8) + 7] = (byte) (longValue & 0x00000000_000000FFL);
108         }
109         return result;
110     }
111 
112     public static byte[] encodeInteger(Object[] values, int length) {
113         byte[] result;
114         result = new byte[length * 4];
115         for (int i = 0; i < length; i++) {
116             int intValue = (int) values[i];
117             result[i * 4] = (byte) ((intValue & 0xff000000) >> 24);
118             result[(i * 4) + 1] = (byte) ((intValue & 0x00ff0000) >> 16);
119             result[(i * 4) + 2] = (byte) ((intValue & 0x0000ff00) >> 8);
120             result[(i * 4) + 3] = (byte) (intValue & 0xff);
121         }
122         return result;
123     }
124 
125     public static byte[] encodeShort(Object[] values, int length) {
126         byte[] result;
127         result = new byte[length * 2];
128         for (int i = 0; i < length; i++) {
129             short intValue = (short) values[i];
130             result[i * 2] = (byte) ((intValue & 0xff00) >> 8);
131             result[(i * 2) + 1] = (byte) (intValue & 0xff);
132         }
133         return result;
134     }
135 
136     public static byte[] encodeByte(Object[] values, int length) {
137         byte[] result;
138         result = new byte[length];
139         for (int i = 0; i < length; i++) {
140             result[i] = (byte) values[i];
141         }
142         return result;
143     }
144 
145     public static byte[] encodeBoolean(Object[] values, int length) {
146         byte[] result;// TODO: Check if this is true and the result is not Math.ceil(values.lenght / 8)
147         result = new byte[length];
148         for (int i = 0; i < length; i++) {
149             result[i] = (byte) (((Boolean) values[i]) ? 0x01 : 0x00);
150         }
151         return result;
152     }
153 }
