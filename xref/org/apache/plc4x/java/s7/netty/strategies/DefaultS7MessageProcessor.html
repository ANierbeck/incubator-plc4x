34  import java.util.*;
35  import java.util.concurrent.atomic.AtomicInteger;

37  /**
38   * While a SetupCommunication message is no problem, when reading or writing data,
39   * situations could arise that have to be handled. The following situations have to
40   * be handled:
41   * - The number of request items is so big, that the resulting PDU would exceed the
42   *   agreed upon PDU size: The request has to be split up into multiple requests.
43   * - If large blocks of data are requested by request items the result of a request
44   *   could exceed the PDU size: The requests has to be split up into multiple requests
45   *   where each requests response doesn't exceed the PDU size.
46   *
47   * The following optimizations should be implemented:
48   * - If blocks are read which are in near proximity to each other it could be better
49   *   to replace multiple requests by one that includes multiple blocks.
50   * - Rearranging the order of request items could reduce the number of needed PDUs.
51   */
52  public class DefaultS7MessageProcessor implements S7MessageProcessor {
53  
54      private AtomicInteger tpduRefGen;
55  
56      public DefaultS7MessageProcessor() {
57          this.tpduRefGen = new AtomicInteger(1);
58      }
59  
60      @Override
61      public Collection<? extends S7Message> process(S7Message s7Message, int pduSize) {
62          // The following considerations have to be taken into account:
63          // - The size of all parameters and payloads of a message cannot exceed the negotiated PDU size
64          // - When reading data, the size of the returned data cannot exceed the negotiated PDU size
65          //
66          // Examples:
67          // - Size of the request exceeds the maximum
68          //  When having a negotiated max PDU size of 256, the maximum size of individual addresses can be at most 18
69          //  If more are sent, the S7 will respond with a frame error.
70          // - Size of the response exceeds the maximum
71          //  When reading two Strings of each 200 bytes length, the size of the request is ok, however the PLC would
72          //  have to send back 400 bytes of String data, which would exceed the PDU size. In this case the first String
73          //  is correctly returned, but for the second item the PLC will return a code of 0x03 = Access Denied
74          // - A S7 device doesn't seem to accept more than one write item. So if we are doing write operations, we
75          //  have to split that up into one message per written item. This also seems to affect arrays. So if
76          //  an array of values is written, we have to also split up that array into single writes.
77  
78          if(s7Message instanceof S7RequestMessage) {
79              S7RequestMessage originalRequestMessage = (S7RequestMessage) s7Message;
80              Optional<VarParameter> varParameterOptional = originalRequestMessage.getParameter(VarParameter.class);
81              if (varParameterOptional.isPresent()) {
82                  VarParameter varParameter = varParameterOptional.get();
83  
84                  // Create a new composite request message.
85                  S7CompositeRequestMessage compositeRequestMessage =
86                      new S7CompositeRequestMessage(originalRequestMessage);
87  
88                  // If this is a read operation, try to get as many items in as possible.
89                  if(varParameter.getType() == ParameterType.READ_VAR) {
90                      // Create a var parameter without any items (yet).
91                      S7Parameter subVarParameter = new VarParameter(varParameter.getType(), new LinkedList<>());
92  
93                      // Create a sub message with only this empty parameter.
94                      S7RequestMessage subMessage = new S7RequestMessage(
95                          s7Message.getMessageType(), (short) tpduRefGen.getAndIncrement(),
96                          Collections.singletonList(subVarParameter), Collections.emptyList(), compositeRequestMessage);
97  
98                      // Add this sub-message to the composite.
99                      compositeRequestMessage.addRequestMessage(subMessage);
100 
101                     // This calculates the size of the header for the request and response.
102                     int curRequestSize = S7RequestSizeCalculator.getRequestMessageSize(subMessage);
103                     int curResponseSize = S7ResponseSizeEstimator.getEstimatedResponseMessageSize(subMessage);
104 
105                     // For each var item of the original request, try adding them to the current sub-message
106                     // as long as it or the resulting response does not exceed the max PDU size.
107                     for (VarParameterItem varParameterItem : varParameter.getItems()) {
108                         VarPayloadItem varPayloadItem = null;
109                         Optional<VarPayloadItem> payloadItem = s7Message.getPayload(VarPayloadItem.class);
110                         if (payloadItem.isPresent()) {
111                             varPayloadItem = payloadItem.get();
112                         }
113 
114                         // Use the S7RequestSizeCalculator to calculate the actual and estimated item sizes.
115                         int itemRequestSize = S7RequestSizeCalculator.getRequestItemTotalSize(
116                             varParameterItem, varPayloadItem);
117                         int itemResponseSize = S7ResponseSizeEstimator.getEstimatedResponseReadItemTotalSize(
118                             varParameterItem, varPayloadItem);
119 
120                         // When adding this item to the request we would exceed the pdu size in
121                         // the request or response, so we have to create a new sub-message.
122                         if ((curRequestSize + itemRequestSize > pduSize) || (curResponseSize + itemResponseSize > pduSize)) {
123                             // Create a new var parameter without any items (yet).
124                             subVarParameter = new VarParameter(varParameter.getType(), new LinkedList<>());
125 
126                             // Create a new sub message with only this empty parameter.
127                             subMessage = new S7RequestMessage(
128                                 s7Message.getMessageType(), (short) tpduRefGen.getAndIncrement(),
129                                 Collections.singletonList(subVarParameter),
130                                 Collections.emptyList(), compositeRequestMessage);
131 
132                             // Reset the message size
133                             curRequestSize = S7RequestSizeCalculator.getRequestMessageSize(subMessage);
134                             curResponseSize = S7ResponseSizeEstimator.getEstimatedResponseMessageSize(subMessage);
135 
136                             // Add this new sub-message to the composite.
137                             compositeRequestMessage.addRequestMessage(subMessage);
138                         } else {
139                             // Increment the current sizes.
140                             curRequestSize += itemRequestSize;
141                             curResponseSize += itemResponseSize;
142                         }
143 
144                         // Add the item to the current subVarParameter.
145                         ((VarParameter) subVarParameter).getItems().add(varParameterItem);
146                     }
147                 }
148 
149                 // If this is a write operation, split up every array item into single value items
150                 // and every item into a separate message.
151                 else if(varParameter.getType() == ParameterType.WRITE_VAR) {
152                     for (VarParameterItem varParameterItem : varParameter.getItems()) {
153                         if(varParameterItem instanceof S7AnyVarParameterItem) {
154                             S7AnyVarParameterItem s7AnyVarParameterItem = (S7AnyVarParameterItem) varParameterItem;
155                             short byteOffset = s7AnyVarParameterItem.getByteOffset();
156                             if(s7AnyVarParameterItem.getTransportSize() == TransportSize.BIT) {
157                                 byte bitOffset = 0;
158                                 for (int i = 0; i < s7AnyVarParameterItem.getNumElements(); i++) {
159                                     // Calculate the new memory and bit offset.
160                                     if((i > 0) && (i % 8) == 0) {
161                                         byteOffset++;
162                                         bitOffset = 0;
163                                     } else {
164                                         bitOffset++;
165                                     }
166 
167                                     // Create a new message with only one single value item in the var parameter.
168                                     VarParameterItem item = new S7AnyVarParameterItem(
169                                         s7AnyVarParameterItem.getSpecificationType(),
170                                         s7AnyVarParameterItem.getMemoryArea(),
171                                         s7AnyVarParameterItem.getTransportSize(), (short) 1,
172                                         s7AnyVarParameterItem.getDataBlockNumber(),
173                                         byteOffset, bitOffset);
174                                     S7Parameter subVarParameter = new VarParameter(varParameter.getType(),
175                                         Collections.singletonList(item));
176                                     S7RequestMessage subMessage = new S7RequestMessage(
177                                         s7Message.getMessageType(), (short) tpduRefGen.getAndIncrement(),
178                                         Collections.singletonList(subVarParameter), Collections.emptyList(),
179                                         compositeRequestMessage);
180 
181                                     // Add the new message to the composite.
182                                     compositeRequestMessage.addRequestMessage(subMessage);
183                                 }
184                             } else {
185                                 for (int i = 0; i < s7AnyVarParameterItem.getNumElements(); i++) {
186                                     byteOffset += s7AnyVarParameterItem.getTransportSize().getSizeInBytes();
187 
188                                     // Create a new message with only one single value item in the var parameter.
189                                     VarParameterItem item = new S7AnyVarParameterItem(
190                                         s7AnyVarParameterItem.getSpecificationType(),
191                                         s7AnyVarParameterItem.getMemoryArea(),
192                                         s7AnyVarParameterItem.getTransportSize(), (short) 1,
193                                         s7AnyVarParameterItem.getDataBlockNumber(),
194                                         byteOffset, (byte) 0);
195                                     S7Parameter subVarParameter = new VarParameter(varParameter.getType(),
196                                         Collections.singletonList(item));
197                                     S7RequestMessage subMessage = new S7RequestMessage(
198                                         s7Message.getMessageType(), (short) tpduRefGen.getAndIncrement(),
199                                         Collections.singletonList(subVarParameter), Collections.emptyList(),
200                                         compositeRequestMessage);
201 
202                                     // Add the new message to the composite.
203                                     compositeRequestMessage.addRequestMessage(subMessage);
204                                 }
205                             }
206                         } else {
207                             throw new NotImplementedException("Handling of other element types not implemented.");
208                         }
209                     }
210                 }
211                 return compositeRequestMessage.getRequestMessages();
212             }
213         }
214 
215         return Collections.singletonList(s7Message);
216     }
