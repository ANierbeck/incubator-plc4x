19  package org.apache.plc4x.java.ads.connection;
20  
21  import io.netty.channel.Channel;
22  import io.netty.channel.ChannelHandler;
23  import io.netty.channel.ChannelInitializer;
24  import io.netty.channel.ChannelPipeline;
25  import org.apache.plc4x.java.ads.api.generic.types.AmsNetId;
26  import org.apache.plc4x.java.ads.api.generic.types.AmsPort;
27  import org.apache.plc4x.java.ads.protocol.Ads2PayloadProtocol;
28  import org.apache.plc4x.java.ads.protocol.Payload2TcpProtocol;
29  import org.apache.plc4x.java.ads.protocol.Plc4x2AdsProtocol;
30  import org.apache.plc4x.java.base.connection.TcpSocketChannelFactory;
31  
32  import java.net.InetAddress;
33  import java.util.concurrent.CompletableFuture;
34  
35  public class AdsTcpPlcConnection extends AdsAbstractPlcConnection {
36  
37      private static final int TCP_PORT = 48898;
38  
39      private AdsTcpPlcConnection(InetAddress address, AmsNetId targetAmsNetId, AmsPort targetAmsPort) {
40          this(address, targetAmsNetId, targetAmsPort, generateAMSNetId(), generateAMSPort());
41      }
42  
43      private AdsTcpPlcConnection(InetAddress address, Integer port, AmsNetId targetAmsNetId, AmsPort targetAmsPort) {
44          this(address, port, targetAmsNetId, targetAmsPort, generateAMSNetId(), generateAMSPort());
45      }
46  
47      private AdsTcpPlcConnection(InetAddress address, AmsNetId targetAmsNetId, AmsPort targetAmsPort, AmsNetId sourceAmsNetId, AmsPort sourceAmsPort) {
48          this(address, null, targetAmsNetId, targetAmsPort, sourceAmsNetId, sourceAmsPort);
49      }
50  
51      private AdsTcpPlcConnection(InetAddress address, Integer port, AmsNetId targetAmsNetId, AmsPort targetAmsPort, AmsNetId sourceAmsNetId, AmsPort sourceAmsPort) {
52          super(new TcpSocketChannelFactory(address, port != null ? port : TCP_PORT), targetAmsNetId, targetAmsPort, sourceAmsNetId, sourceAmsPort);
53      }
54  
55      public static AdsTcpPlcConnection of(InetAddress address, AmsNetId targetAmsNetId, AmsPort targetAmsPort) {
56          return new AdsTcpPlcConnection(address, targetAmsNetId, targetAmsPort);
57      }
58  
59      public static AdsTcpPlcConnection of(InetAddress address, Integer port, AmsNetId targetAmsNetId, AmsPort targetAmsPort) {
60          return new AdsTcpPlcConnection(address, port, targetAmsNetId, targetAmsPort);
61      }
62  
63      public static AdsTcpPlcConnection of(InetAddress address, AmsNetId targetAmsNetId, AmsPort targetAmsPort, AmsNetId sourceAmsNetId, AmsPort sourceAmsPort) {
64          return new AdsTcpPlcConnection(address, null, targetAmsNetId, targetAmsPort, sourceAmsNetId, sourceAmsPort);
65      }
66  
67      public static AdsTcpPlcConnection of(InetAddress address, Integer port, AmsNetId targetAmsNetId, AmsPort targetAmsPort, AmsNetId sourceAmsNetId, AmsPort sourceAmsPort) {
68          return new AdsTcpPlcConnection(address, port, targetAmsNetId, targetAmsPort, sourceAmsNetId, sourceAmsPort);
69      }
70  
71      @Override
72      protected ChannelHandler getChannelHandler(CompletableFuture<Void> sessionSetupCompleteFuture) {
73          return new ChannelInitializer() {
74              @Override
75              protected void initChannel(Channel channel) {
76                  // Build the protocol stack for communicating with the ads protocol.
77                  ChannelPipeline pipeline = channel.pipeline();
78                  pipeline.addLast(new Payload2TcpProtocol());
79                  pipeline.addLast(new Ads2PayloadProtocol());
80                  pipeline.addLast(new Plc4x2AdsProtocol(targetAmsNetId, targetAmsPort, sourceAmsNetId, sourceAmsPort));
81              }
82          };
83      }
84  
85      protected static AmsPort generateAMSPort() {
86          return AmsPort.of(TCP_PORT);
87      }
88  
89  }
