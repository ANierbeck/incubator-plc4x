19  package org.apache.plc4x.java.ads.protocol;
20  
21  import io.netty.buffer.ByteBuf;
22  import io.netty.channel.ChannelHandlerContext;
23  import io.netty.handler.codec.MessageToMessageCodec;
24  import org.apache.plc4x.java.ads.api.serial.AmsSerialAcknowledgeFrame;
25  import org.apache.plc4x.java.ads.api.serial.AmsSerialFrame;
26  import org.apache.plc4x.java.ads.api.serial.AmsSerialResetFrame;
27  import org.apache.plc4x.java.ads.api.serial.types.*;
28  import org.apache.plc4x.java.ads.protocol.util.DigestUtil;
29  import org.apache.plc4x.java.api.exceptions.PlcProtocolException;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  import java.util.List;
34  
35  public class Payload2SerialProtocol extends MessageToMessageCodec<ByteBuf, ByteBuf> {
36  
37      private static final Logger LOGGER = LoggerFactory.getLogger(Payload2TcpProtocol.class);
38  
39      @Override
40      protected void encode(ChannelHandlerContext channelHandlerContext, ByteBuf amsPacket, List<Object> out) throws Exception {
41          out.add(AmsSerialFrame.of(FragmentNumber.of((byte) 0), UserData.of(amsPacket)).getByteBuf());
42          // TODO: we need to remember the fragment and maybe even need to spilt up the package
43          // TODO: if we exceed 255 byte
44          amsPacket.release();
45      }
46  
47      @Override
48      protected void decode(ChannelHandlerContext channelHandlerContext, ByteBuf byteBuf, List<Object> out) throws Exception {
49          MagicCookie magicCookie = MagicCookie.of(byteBuf);
50          TransmitterAddress transmitterAddress = TransmitterAddress.of(byteBuf);
51          ReceiverAddress receiverAddress = ReceiverAddress.of(byteBuf);
52          FragmentNumber fragmentNumber = FragmentNumber.of(byteBuf);
53          UserDataLength userDataLength = UserDataLength.of(byteBuf);
54          UserData userData;
55          byte userDataLengthAsByte = userDataLength.getAsByte();
56          if (userDataLengthAsByte > 0) {
57              byte[] userDataByteArray = new byte[userDataLengthAsByte];
58              byteBuf.readBytes(userDataByteArray);
59              userData = UserData.of(userDataByteArray);
60          } else {
61              userData = UserData.EMPTY;
62          }
63          CRC crc = CRC.of(byteBuf);
64  
65          switch (magicCookie.getAsInt()) {
66              case AmsSerialFrame.ID:
67                  AmsSerialFrame amsSerialFrame = AmsSerialFrame.of(magicCookie, transmitterAddress, receiverAddress, fragmentNumber, userDataLength, userData, crc);
68                  LOGGER.debug("Ams Serial Frame received {}", amsSerialFrame);
69                  out.add(userData.getByteBuf());
70                  break;
71              case AmsSerialAcknowledgeFrame.ID:
72                  AmsSerialAcknowledgeFrame amsSerialAcknowledgeFrame = AmsSerialAcknowledgeFrame.of(magicCookie, transmitterAddress, receiverAddress, fragmentNumber, userDataLength, crc);
73                  LOGGER.debug("Ams Serial ACK Frame received {}", amsSerialAcknowledgeFrame);
74                  break;
75              case AmsSerialResetFrame.ID:
76                  AmsSerialResetFrame amsSerialResetFrame = AmsSerialResetFrame.of(magicCookie, transmitterAddress, receiverAddress, fragmentNumber, userDataLength, crc);
77                  LOGGER.debug("Ams Serial Reset Frame received {}", amsSerialResetFrame);
78                  break;
79          }
80          int calculatedCrc16 = DigestUtil.calculateCrc16(magicCookie, transmitterAddress, receiverAddress, fragmentNumber, userDataLength, userData);
81          if (!crc.equals(CRC.of(calculatedCrc16))) {
82              throw new PlcProtocolException("CRC checksum wrong");
83          }
84  
85          if (byteBuf.readableBytes() > 0) {
86              byteBuf.release();
87              throw new IllegalStateException("Unread bytes left: " + byteBuf.readableBytes());
88          }
89  
90          byteBuf.release();
91      }
92  }
