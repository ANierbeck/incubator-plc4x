19  package org.apache.plc4x.java;
20  
21  import org.apache.plc4x.java.api.PlcDriver;
22  import org.apache.plc4x.java.api.authentication.PlcAuthentication;
23  import org.apache.plc4x.java.api.connection.PlcConnection;
24  import org.apache.plc4x.java.api.exceptions.PlcConnectionException;
25  
26  import java.net.URI;
27  import java.net.URISyntaxException;
28  import java.util.HashMap;
29  import java.util.Map;
30  import java.util.ServiceLoader;
31  
32  public class PlcDriverManager {
33  
34      private Map<String, PlcDriver> driverMap = null;
35  
36      public PlcDriverManager() {
37          this(Thread.currentThread().getContextClassLoader());
38      }
39  
40      public PlcDriverManager(ClassLoader classLoader) {
41          driverMap = new HashMap<>();
42          ServiceLoader<PlcDriver> plcDriverLoader = ServiceLoader.load(PlcDriver.class, classLoader);
43          for (PlcDriver driver : plcDriverLoader) {
44              if (driverMap.containsKey(driver.getProtocolCode())) {
45                  throw new IllegalStateException(
46                      "Multiple driver implementations available for protocol code '" +
47                          driver.getProtocolCode() + "'");
48              }
49              driverMap.put(driver.getProtocolCode(), driver);
50          }
51      }
52  
53      /**
54       * Connects to a PLC using the given plc connection string.
55       * @param url plc connection string.
56       * @return PlcConnection object.
57       * @throws PlcConnectionException an exception if the connection attempt failed.
58       */
59      public PlcConnection getConnection(String url) throws PlcConnectionException {
60          PlcDriver driver = getDriver(url);
61          PlcConnection connection = driver.connect(url);
62          connection.connect();
63          return connection;
64      }
65  
66      /**
67       * Connects to a PLC using the given plc connection string using given authentication credentials.
68       * @param url plc connection string.
69       * @param authentication authentication credentials.
70       * @return PlcConnection object.
71       * @throws PlcConnectionException an exception if the connection attempt failed.
72       */
73      public PlcConnection getConnection(String url, PlcAuthentication authentication) throws PlcConnectionException {
74          PlcDriver driver = getDriver(url);
75          return driver.connect(url, authentication);
76      }
77  
78      private PlcDriver getDriver(String url) throws PlcConnectionException {
79          try {
80              URI connectionUri = new URI(url);
81              String protocol = connectionUri.getScheme();
82              PlcDriver driver = driverMap.get(protocol);
83              if(driver ==  null) {
84                  throw new PlcConnectionException("Unable to find driver for protocol '" + protocol + "'");
85              }
86              return driver;
87          } catch (URISyntaxException e) {
88              throw new PlcConnectionException("Invalid plc4j connection string '" + url + "'");
89          }
90      }
91  
92  }
