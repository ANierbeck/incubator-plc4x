36  class PlcDriverManagerTest {
37  
38      /**
39       * Tries to get the mock plc driver which is part of this testsuite.
40       */
41      @Test
42      @Tag("fast")
43      void getExistingDriverTest() throws PlcException {
44          MockConnection mockConnection = (MockConnection) new PlcDriverManager().getConnection("mock://some-cool-url");
45          SoftAssertions softAssertions = new SoftAssertions();
46          softAssertions.assertThat(mockConnection.getAuthentication())
47              .as("check authentication object")
48              .isNull();
49          softAssertions.assertThat(mockConnection.isConnected())
50              .as("check connection state")
51              .isTrue();
52          softAssertions.assertThat(mockConnection.isClosed())
53              .as("check closed state")
54              .isFalse();
55          softAssertions.assertAll();
56      }
57  
58      /**
59       * Tries to get the mock plc driver with authentication which is part of this testsuite.
60       */
61      @Test
62      @Tag("fast")
63      void getExistingDriverWithAuthenticationTest() throws PlcException {
64          PlcUsernamePasswordAuthentication authentication =
65              new PlcUsernamePasswordAuthentication("user", "pass");
66          MockConnection mockConnection = (MockConnection) new PlcDriverManager().getConnection("mock://some-cool-url", authentication);
67  
68          SoftAssertions softAssertions = new SoftAssertions();
69          softAssertions.assertThat(mockConnection.getAuthentication())
70              .as("check authentication object")
71              .isNotNull();
72          softAssertions.assertThat(mockConnection.getAuthentication())
73              .as("check authentication object type")
74              .isInstanceOf(PlcUsernamePasswordAuthentication.class);
75          softAssertions.assertThat(mockConnection.isConnected())
76              .as("check connection state")
77              .isTrue();
78          softAssertions.assertThat(mockConnection.isClosed())
79              .as("check closed state")
80              .isFalse();
81          softAssertions.assertAll();
82      }

106     /**
107      * In this test the {@link PlcDriverManager} will be configured with a service list that
108      * contains multiple implementation instances of the same protocol. This should result in
109      * an error.
110      */
111     @Test
112     @Tag("fast")
113     void getDuplicateDriver() throws PlcException, MalformedURLException {
114         // Save and replace the context classloader as we need to force the ServiceLoader to
115         // use a different service file.
116         ClassLoader originalClassloader = Thread.currentThread().getContextClassLoader();
117         URL[] urls = new URL[1];
118         urls[0] = new File("src/test/resources/test").toURI().toURL();
119         ClassLoader fakeClassLoader = new URLClassLoader(urls, originalClassloader);
120 
121         assertThatThrownBy(() -> new PlcDriverManager(fakeClassLoader).getConnection("mock://some-cool-url"))
122             .as("check detection of duplicated driver detection")
123             .isInstanceOf(IllegalStateException.class);
124     }
