299     @Test
300     void accessResponseFromRequestMultiThreaded() throws Exception {
301         ReadRequestItem<Byte> readRequestItem = new ReadRequestItem<>(Byte.class, null, (byte) 0x0);
302         CountDownLatch successLatch = new CountDownLatch(10);
303         CountDownLatch failLatch = new CountDownLatch(10);
304         for (int i = 0; i < 10; i++) {
305             new Thread(() -> {
306                 try {
307                     readRequestItem.getResponseItem().get(10, TimeUnit.SECONDS);
308                     successLatch.countDown();
309                 } catch (Exception ignore) {
310                     failLatch.countDown();
311                 }
312             }).start();
313         }
314         assertEquals(10, successLatch.getCount());
315 
316         ReadResponseItem<Byte> readResponseItem = new ReadResponseItem<>(readRequestItem, ResponseCode.OK, Collections.singletonList((byte) 0x0));
317         ReadResponseItem<Byte> byteReadResponseItem = readRequestItem.getResponseItem().get(10, TimeUnit.MILLISECONDS);
318         assertEquals(readResponseItem, byteReadResponseItem);
319         assertEquals(0, successLatch.getCount());
320         assertEquals(10, failLatch.getCount());
321     }
