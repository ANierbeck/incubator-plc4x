107     @Override
108     protected ChannelHandler getChannelHandler(CompletableFuture<Void> sessionSetupCompleteFuture) {
109         return new ChannelInitializer() {
110             @Override
111             protected void initChannel(Channel channel) {
112                 // Build the protocol stack for communicating with the ads protocol.
113                 ChannelPipeline pipeline = channel.pipeline();
114                 pipeline.addLast(new Payload2TcpProtocol());
115                 pipeline.addLast(new Ads2PayloadProtocol());
116                 pipeline.addLast(new Plc4x2AdsProtocol(targetAmsNetId, targetAmsPort, sourceAmsNetId, sourceAmsPort, addressMapping));
117             }
118         };
119     }

121     public InetAddress getRemoteAddress() {
122         return ((TcpSocketChannelFactory) channelFactory).getAddress();
123     }

125     protected static AmsNetId generateAMSNetId() {
126         try {
127             return AmsNetId.of(Inet4Address.getLocalHost().getHostAddress() + ".1.1");
128         } catch (UnknownHostException e) {
129             throw new PlcRuntimeException(e);
130         }
131     }
132 
133     protected static AmsPort generateAMSPort() {
134         return AmsPort.of(localPorts.getAndIncrement());
135     }

137     @Override
138     public CompletableFuture<PlcSubscriptionResponse> subscribe(PlcSubscriptionRequest subscriptionRequest) {
139         // TODO: Make this multi-value
140         CompletableFuture<PlcSubscriptionResponse> future = new CompletableFuture<>();
141         if (subscriptionRequest.getNumberOfItems() != 1) {
142             throw new PlcNotImplementedException("Multirequest on subscribe not implemented yet");
143         }
144 
145         SubscriptionRequestItem<?> subscriptionRequestItem = subscriptionRequest.getRequestItem().orElseThrow(NullPointerException::new);
146 
147         Objects.requireNonNull(subscriptionRequestItem.getConsumer());
148         Objects.requireNonNull(subscriptionRequestItem.getAddress());
149         Objects.requireNonNull(subscriptionRequestItem.getDatatype());
150 
151         Address address = subscriptionRequestItem.getAddress();
152         Class<?> datatype = subscriptionRequestItem.getDatatype();
153 
154         IndexGroup indexGroup;
155         IndexOffset indexOffset;
156         // If this is a symbolic address, it has to be resolved first.
157         // TODO: This is blocking, should be changed to be async.
158         if (address instanceof SymbolicAdsAddress) {
159             mapAddress((SymbolicAdsAddress) address);
160             AdsAddress adsAddress = addressMapping.get(address);
161             if (adsAddress == null) {
162                 throw new PlcRuntimeException("Unresolvable address" + address);
163             }
164             indexGroup = IndexGroup.of(adsAddress.getIndexGroup());
165             indexOffset = IndexOffset.of(adsAddress.getIndexOffset());
166         }
167         // If it's no symbolic address, we can continue immediately
168         // without having to do any resolving.
169         else if (address instanceof AdsAddress) {
170             AdsAddress adsAddress = (AdsAddress) address;
171             indexGroup = IndexGroup.of(adsAddress.getIndexGroup());
172             indexOffset = IndexOffset.of(adsAddress.getIndexOffset());
173         } else {
174             throw new IllegalArgumentException("Unsupported address type " + address.getClass());
175         }
176 
177         // Prepare the subscription request itself.
178         AdsAddDeviceNotificationRequest adsAddDeviceNotificationRequest = AdsAddDeviceNotificationRequest.of(
179             targetAmsNetId,
180             targetAmsPort,
181             sourceAmsNetId,
182             sourceAmsPort,
183             Invoke.NONE,
184             indexGroup,
185             indexOffset,
186             LittleEndianDecoder.getLengthFor(datatype, 1),
187             TransmissionMode.DefinedValues.ADSTRANS_SERVERCYCLE,
188             MaxDelay.of(0),
189             CycleTime.of(4000000)
190         );
191 
192         // Send the request to the plc and wait for a response
193         // TODO: This is blocking, should be changed to be async.
194         CompletableFuture<PlcProprietaryResponse<AdsAddDeviceNotificationResponse>> addDeviceFuture = new CompletableFuture<>();
195         channel.writeAndFlush(new PlcRequestContainer<>(new PlcProprietaryRequest<>(adsAddDeviceNotificationRequest), addDeviceFuture));
196         PlcProprietaryResponse<AdsAddDeviceNotificationResponse> addDeviceResponse = getFromFuture(addDeviceFuture, ADD_DEVICE_TIMEOUT);
197         AdsAddDeviceNotificationResponse response = addDeviceResponse.getResponse();
198 
199         // Abort if we got anything but a successful response.
200         if (response.getResult().toAdsReturnCode() != AdsReturnCode.ADS_CODE_0) {
201             throw new PlcRuntimeException("Error code received " + response.getResult());
202         }
203         AdsSubscriptionHandle adsSubscriptionHandle = new AdsSubscriptionHandle(response.getNotificationHandle());
204         future.complete(new PlcSubscriptionResponse(subscriptionRequest, Collections.singletonList(
205             new SubscriptionResponseItem<>(subscriptionRequestItem, adsSubscriptionHandle, ResponseCode.OK))));
206 
207         Consumer<AdsDeviceNotificationRequest> adsDeviceNotificationRequestConsumer =
208             adsDeviceNotificationRequest -> adsDeviceNotificationRequest.getAdsStampHeaders().forEach(adsStampHeader -> {
209                 Calendar timeStamp = Calendar.getInstance();
210                 timeStamp.setTime(adsStampHeader.getTimeStamp().getAsDate());
211 
212                 adsStampHeader.getAdsNotificationSamples()
213                     .forEach(adsNotificationSample -> {
214                         Data data = adsNotificationSample.getData();
215                         try {
216                             @SuppressWarnings("unchecked")
217                             List<?> decodeData = LittleEndianDecoder.decodeData(datatype, data.getBytes());
218                             SubscriptionEventItem subscriptionEventItem =
219                                 new SubscriptionEventItem(subscriptionRequestItem, timeStamp, decodeData);
220                             subscriptionRequestItem.getConsumer().accept(subscriptionEventItem);
221                         } catch (PlcProtocolException | RuntimeException e) {
222                             LOGGER.error("Can't decode {}", data, e);
223                         }
224                     });
225             });
226         // TODO: What's this for? Is this still needed if we use the consumers in the subscriptions?
227         getChannel().pipeline().get(Plc4x2AdsProtocol.class).addConsumer(adsDeviceNotificationRequestConsumer);
228         return future;
229     }

231     @Override
232     public CompletableFuture<PlcUnsubscriptionResponse> unsubscribe(PlcUnsubscriptionRequest unsubscriptionRequest) {
233         for (UnsubscriptionRequestItem unsubscriptionRequestItem : unsubscriptionRequest.getRequestItems()) {
234             Objects.requireNonNull(unsubscriptionRequestItem);
235             if (unsubscriptionRequestItem.getSubscriptionHandle() instanceof AdsSubscriptionHandle) {
236                 AdsSubscriptionHandle adsSubscriptionHandle =
237                     (AdsSubscriptionHandle) unsubscriptionRequestItem.getSubscriptionHandle();
238                 AdsDeleteDeviceNotificationRequest adsDeleteDeviceNotificationRequest =
239                     AdsDeleteDeviceNotificationRequest.of(
240                         targetAmsNetId,
241                         targetAmsPort,
242                         sourceAmsNetId,
243                         sourceAmsPort,
244                         Invoke.NONE,
245                         adsSubscriptionHandle.getNotificationHandle()
246                     );
247                 CompletableFuture<PlcProprietaryResponse<AdsDeleteDeviceNotificationResponse>> deleteDeviceFuture =
248                     new CompletableFuture<>();
249                 channel.writeAndFlush(new PlcRequestContainer<>(new PlcProprietaryRequest<>(
250                     adsDeleteDeviceNotificationRequest), deleteDeviceFuture));
251 
252                 PlcProprietaryResponse<AdsDeleteDeviceNotificationResponse> deleteDeviceResponse =
253                     getFromFuture(deleteDeviceFuture, DEL_DEVICE_TIMEOUT);
254                 AdsDeleteDeviceNotificationResponse response = deleteDeviceResponse.getResponse();
255                 if (response.getResult().toAdsReturnCode() != AdsReturnCode.ADS_CODE_0) {
256                     throw new PlcRuntimeException("Non error code received " + response.getResult());
257                 }
258             }
259         }
260         CompletableFuture<PlcUnsubscriptionResponse> future = new CompletableFuture<>();
261         future.complete(new PlcUnsubscriptionResponse());
262         return future;
263     }
