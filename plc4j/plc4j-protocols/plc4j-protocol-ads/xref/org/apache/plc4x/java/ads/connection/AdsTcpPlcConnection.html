102     @Override
103     protected ChannelHandler getChannelHandler(CompletableFuture<Void> sessionSetupCompleteFuture) {
104         return new ChannelInitializer() {
105             @Override
106             protected void initChannel(Channel channel) {
107                 // Build the protocol stack for communicating with the ads protocol.
108                 ChannelPipeline pipeline = channel.pipeline();
109                 pipeline.addLast(new Payload2TcpProtocol());
110                 pipeline.addLast(new Ads2PayloadProtocol());
111                 pipeline.addLast(new Plc4x2AdsProtocol(targetAmsNetId, targetAmsPort, sourceAmsNetId, sourceAmsPort, addressMapping));
112             }
113         };
114     }
115 
116     public InetAddress getRemoteAddress() {
117         return ((TcpSocketChannelFactory) channelFactory).getAddress();
118     }
119 
120     protected static AmsNetId generateAMSNetId() {
121         try {
122             return AmsNetId.of(Inet4Address.getLocalHost().getHostAddress() + ".1.1");
123         } catch (UnknownHostException e) {
124             throw new PlcRuntimeException(e);
125         }
126     }
127 
128     protected static AmsPort generateAMSPort() {
129         return AmsPort.of(localPorts.getAndIncrement());
130     }

132     @Override
133     public <T extends R, R> void subscribe(Consumer<PlcNotification<R>> consumer, Address address, Class<T> dataType) {
134         Objects.requireNonNull(consumer);
135         Objects.requireNonNull(address);
136         IndexGroup indexGroup;
137         IndexOffset indexOffset;
138         if (address instanceof SymbolicAdsAddress) {
139             mapAddresses(new PlcRequest() {
140                 {
141                     requestItems.add(new RequestItem(Void.class, address) {
142                         // Not relevant
143                     });
144                 }
145             });
146             AdsAddress adsAddress = addressMapping.get(address);
147             if (adsAddress == null) {
148                 throw new PlcRuntimeException("Unresolvable address" + address);
149             }
150             indexGroup = IndexGroup.of(adsAddress.getIndexGroup());
151             indexOffset = IndexOffset.of(adsAddress.getIndexOffset());
152         } else if (address instanceof AdsAddress) {
153             AdsAddress adsAddress = (AdsAddress) address;
154             indexGroup = IndexGroup.of(adsAddress.getIndexGroup());
155             indexOffset = IndexOffset.of(adsAddress.getIndexOffset());
156         } else {
157             throw new IllegalArgumentException("Unssuported address type " + address.getClass());
158         }
159         AdsAddDeviceNotificationRequest adsAddDeviceNotificationRequest = AdsAddDeviceNotificationRequest.of(
160             targetAmsNetId,
161             targetAmsPort,
162             sourceAmsNetId,
163             sourceAmsPort,
164             Invoke.NONE,
165             indexGroup,
166             indexOffset,
167             Length.of(1),
168             TransmissionMode.of(3),
169             MaxDelay.of(0),
170             CycleTime.of(4000000)
171         );
172 
173         CompletableFuture<PlcProprietaryResponse<AdsAddDeviceNotificationResponse>> addDeviceFuture = new CompletableFuture<>();
174         channel.writeAndFlush(new PlcRequestContainer<>(new PlcProprietaryRequest<>(adsAddDeviceNotificationRequest), addDeviceFuture));
175         PlcProprietaryResponse<AdsAddDeviceNotificationResponse> addDeviceResponse = getFromFuture(addDeviceFuture, ADD_DEVICE_TIMEOUT);
176         AdsAddDeviceNotificationResponse response = addDeviceResponse.getResponse();
177         if (response.getResult().toAdsReturnCode() != AdsReturnCode.ADS_CODE_0) {
178             throw new PlcRuntimeException("Non error code received " + response.getResult());
179         }
180         NotificationHandle notificationHandle = response.getNotificationHandle();
181         handleConsumerMap.put(notificationHandle, consumer);
182 
183         Consumer<AdsDeviceNotificationRequest> adsDeviceNotificationRequestConsumer =
184             adsDeviceNotificationRequest -> adsDeviceNotificationRequest.getAdsStampHeaders().forEach(adsStampHeader -> {
185                 Date timeStamp = adsStampHeader.getTimeStamp().getAsDate();
186 
187                 adsStampHeader.getAdsNotificationSamples()
188                     .forEach(adsNotificationSample -> {
189                         Consumer<? extends PlcNotification> plcNotificationConsumer = handleConsumerMap.get(adsNotificationSample.getNotificationHandle());
190                         if (plcNotificationConsumer == null) {
191                             LOGGER.warn("Unmapped notification received {}", adsNotificationSample);
192                             return;
193                         }
194                         Data data = adsNotificationSample.getData();
195                         try {
196                             @SuppressWarnings("unchecked")
197                             List<R> decodeData = (List<R>) LittleEndianDecoder.decodeData(dataType, data.getBytes());
198                             consumer.accept(new PlcNotification<>(timeStamp, address, decodeData));
199                         } catch (PlcProtocolException | RuntimeException e) {
200                             LOGGER.error("Can't decode {}", data, e);
201                         }
202                     });
203             });
204         subscriberMap.put(Pair.of(consumer, address), Pair.of(adsDeviceNotificationRequestConsumer, notificationHandle));
205         getChannel().pipeline().get(Plc4x2AdsProtocol.class).addConsumer(adsDeviceNotificationRequestConsumer);
206     }
207 
208     @Override
209     public <R> void unsubscribe(Consumer<PlcNotification<R>> consumer, Address address) {
210         Pair<Consumer<AdsDeviceNotificationRequest>, NotificationHandle> handlePair = subscriberMap.remove(Pair.of(consumer, address));
211         if (handlePair != null) {
212             NotificationHandle notificationHandle = handlePair.getRight();
213             AdsDeleteDeviceNotificationRequest adsDeleteDeviceNotificationRequest = AdsDeleteDeviceNotificationRequest.of(
214                 targetAmsNetId,
215                 targetAmsPort,
216                 sourceAmsNetId,
217                 sourceAmsPort,
218                 Invoke.NONE,
219                 notificationHandle
220             );
221             CompletableFuture<PlcProprietaryResponse<AdsDeleteDeviceNotificationResponse>> deleteDeviceFuture = new CompletableFuture<>();
222             channel.writeAndFlush(new PlcRequestContainer<>(new PlcProprietaryRequest<>(adsDeleteDeviceNotificationRequest), deleteDeviceFuture));
223 
224             PlcProprietaryResponse<AdsDeleteDeviceNotificationResponse> deleteDeviceResponse = getFromFuture(deleteDeviceFuture, DEL_DEVICE_TIMEOUT);
225             AdsDeleteDeviceNotificationResponse response = deleteDeviceResponse.getResponse();
226             if (response.getResult().toAdsReturnCode() != AdsReturnCode.ADS_CODE_0) {
227                 throw new PlcRuntimeException("Non error code received " + response.getResult());
228             }
229 
230             getChannel().pipeline().get(Plc4x2AdsProtocol.class).removeConsumer(handlePair.getLeft());
231             handleConsumerMap.remove(notificationHandle);
232         }
233     }
