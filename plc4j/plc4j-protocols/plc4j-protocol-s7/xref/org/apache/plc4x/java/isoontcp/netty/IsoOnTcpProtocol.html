19  package org.apache.plc4x.java.isoontcp.netty;
20  
21  import io.netty.buffer.ByteBuf;
22  import io.netty.buffer.ByteBufUtil;
23  import io.netty.buffer.Unpooled;
24  import io.netty.channel.ChannelHandlerContext;
25  import io.netty.handler.codec.MessageToMessageCodec;
26  import org.apache.plc4x.java.api.exceptions.PlcProtocolException;
27  import org.apache.plc4x.java.isoontcp.netty.model.IsoOnTcpMessage;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import java.util.List;
32  
33  public class IsoOnTcpProtocol extends MessageToMessageCodec<ByteBuf, IsoOnTcpMessage> {
34  
35      public static final byte ISO_ON_TCP_MAGIC_NUMBER = 0x03;
36  
37      private static final Logger logger = LoggerFactory.getLogger(IsoOnTcpProtocol.class);
38  
39      @Override
40      protected void encode(ChannelHandlerContext ctx, IsoOnTcpMessage in, List<Object> out) throws Exception {
41          logger.debug("ISO on TCP Message sent");
42          // At this point of processing all higher levels have already serialized their payload.
43          // This data is passed to the lower levels in form of an IoBuffer.
44          final ByteBuf userData = in.getUserData();
45  
46          int packetSize = userData.readableBytes() + 4;
47  
48          ByteBuf buf = Unpooled.buffer();
49          // Version (is always constant 0x03)
50          buf.writeByte(ISO_ON_TCP_MAGIC_NUMBER);
51          // Reserved (is always constant 0x00)
52          buf.writeByte((byte) 0x00);
53          // Packet length (including ISOonTCP header)
54          // ("remaining" returns the number of bytes left to read in this buffer.
55          // It is usually set to a read position of 0 and a limit at the end.
56          // So in general remaining is equivalent to a non-existing
57          // "userData.size()" method.)
58          buf.writeShort((short) packetSize);
59  
60          // Output the payload.
61          buf.writeBytes(userData);
62  
63          out.add(buf);
64      }
65  
66      @Override
67      protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
68          if(logger.isTraceEnabled()) {
69              logger.trace("Got Data: {}", ByteBufUtil.hexDump(in));
70          }
71          // If at least 4 bytes are readable, peek into them (without changing the read position)
72          // and get the packet length. Only if the available amount of readable bytes is larger or
73          // equal to this, continue processing the rest.
74          if(in.readableBytes() >= 4) {
75              logger.debug("ISO on TCP Message received");
76              // The ISO on TCP protocol is really simple and in this case the buffer length
77              // will take care of the higher levels not reading more than is in the packet.
78              // So we just gobble up the header and continue reading in higher levels.
79              if (in.getByte(0) != ISO_ON_TCP_MAGIC_NUMBER) {
80                  logger.warn("Expecting ISO on TCP magic number: {}", ISO_ON_TCP_MAGIC_NUMBER);
81                  logger.debug("Got Data: " + ByteBufUtil.hexDump(in));
82                  exceptionCaught(ctx, new PlcProtocolException(
83                      String.format("Expecting ISO on TCP magic number: %02X", ISO_ON_TCP_MAGIC_NUMBER)));
84                  return;
85              }
86              // Byte 1 is a reserved byte set to 0x00
87              short packetLength = in.getShort(2);
88              if(in.readableBytes() >= packetLength) {
89                  // Skip the 4 bytes we peeked into manually.
90                  in.skipBytes(4);
91                  // Simply place the current buffer to the output ... the next handler will continue.
92                  ByteBuf payload = in.readBytes(packetLength - 4);
93                  out.add(new IsoOnTcpMessage(payload));
94              }
95          }
96      }
97  
98  }
