63      @Override
64      protected void encode(ChannelHandlerContext ctx, PlcRequestContainer msg, List<Object> out) throws Exception {
65          if (msg.getRequest() instanceof PlcReadRequest) {
66              List<VarParameterItem> parameterItems = new LinkedList<>();
67  
68              PlcReadRequest readRequest = (PlcReadRequest) msg.getRequest();
69              for (ReadRequestItem requestItem : readRequest.getReadRequestItems()) {
70                  // Try to get the correct S7 transport size for the given data type.
71                  // (Map PLC4X data type to S7 data type)
72                  TransportSize transportSize = encodeTransportSize(requestItem.getDatatype());
73                  if (transportSize == null) {
74                      throw new PlcException("Unknown transport size for datatype " + requestItem.getDatatype());
75                  }
76  
77                  // Depending on the address type, generate the corresponding type of request item.
78                  VarParameterItem varParameterItem = encodeVarParameterItem(requestItem.getAddress(), transportSize, requestItem.getSize());
79                  parameterItems.add(varParameterItem);
80              }
81              VarParameter readVarParameter =  new VarParameter(ParameterType.READ_VAR, parameterItems);
82  
83              // Assemble the request.
84              S7RequestMessage s7ReadRequest = new S7RequestMessage(MessageType.JOB,
85                  (short) tpduGenerator.getAndIncrement(), Collections.singletonList(readVarParameter),
86                  Collections.emptyList());
87  
88              requests.put(s7ReadRequest.getTpduReference(), msg);
89  
90              out.add(s7ReadRequest);
91          } else if(msg.getRequest() instanceof PlcWriteRequest) {
92              List<VarParameterItem> parameterItems = new LinkedList<>();
93              List<VarPayloadItem> payloadItems = new LinkedList<>();
94  
95              PlcWriteRequest writeRequest = (PlcWriteRequest) msg.getRequest();
96              for (WriteRequestItem requestItem : writeRequest.getRequestItems()) {
97                  // Try to get the correct S7 transport size for the given data type.
98                  // (Map PLC4X data type to S7 data type)
99                  TransportSize transportSize = encodeTransportSize(requestItem.getDatatype());
100                 if (transportSize == null) {
101                     throw new PlcException("Unknown transport size for datatype " + requestItem.getDatatype());
102                 }
103 
104                 // Depending on the address type, generate the corresponding type of request item.
105                 VarParameterItem varParameterItem = encodeVarParameterItem(
106                     requestItem.getAddress(), transportSize, requestItem.getValues().length);
107                 parameterItems.add(varParameterItem);
108 
109                 DataTransportSize dataTransportSize = encodeDataTransportSize(requestItem.getDatatype());
110                 if (dataTransportSize == null) {
111                     throw new PlcException("Unknown data transport size for datatype " + requestItem.getDatatype());
112                 }
113 
114                 VarPayloadItem varPayloadItem = new VarPayloadItem(
115                     DataTransportErrorCode.RESERVED, dataTransportSize, encodeData(requestItem.getValues()));
116 
117                 payloadItems.add(varPayloadItem);
118             }
119             VarParameter writeVarParameter =  new VarParameter(ParameterType.WRITE_VAR, parameterItems);
120             VarPayload writeVarPayload = new VarPayload(ParameterType.WRITE_VAR, payloadItems);
121 
122             // Assemble the request.
123             S7RequestMessage s7WriteRequest = new S7RequestMessage(MessageType.JOB,
124                 (short) tpduGenerator.getAndIncrement(), Collections.singletonList(writeVarParameter),
125                 Collections.singletonList(writeVarPayload));
126 
127             requests.put(s7WriteRequest.getTpduReference(), msg);
128 
129             out.add(s7WriteRequest);
130         }
131     }

133     @Override
134     protected void decode(ChannelHandlerContext ctx, S7Message msg, List<Object> out) throws Exception {
135         if(msg instanceof S7ResponseMessage) {
136             S7ResponseMessage responseMessage = (S7ResponseMessage) msg;
137             short tpduReference = responseMessage.getTpduReference();
138             if (requests.containsKey(tpduReference)) {
139                 PlcRequestContainer requestContainer = requests.remove(tpduReference);
140 
141                 PlcResponse response = null;
142 
143                 // Handle the response to a read request.
144                 if (requestContainer.getRequest() instanceof PlcReadRequest) {
145                     PlcReadRequest plcReadRequest = (PlcReadRequest) requestContainer.getRequest();
146 
147                     List<ReadResponseItem> responseItems = new LinkedList<>();
148                     VarPayload payload = responseMessage.getPayload(VarPayload.class);
149                     // If the numbers of items don't match, we're in big trouble as the only
150                     // way to know how to interpret the responses is by aligning them with the
151                     // items from the request as this information is not returned by the PLC.
152                     if(plcReadRequest.getReadRequestItems().size() != payload.getPayloadItems().size()) {
153                         throw new PlcProtocolException(
154                             "The number of requested items doesn't match the number of returned items");
155                     }
156                     List<VarPayloadItem> payloadItems = payload.getPayloadItems();
157                     for (int i = 0; i < payloadItems.size(); i++) {
158                         VarPayloadItem payloadItem = payloadItems.get(i);
159 
160                         // Get the request item for this payload item
161                         ReadRequestItem requestItem = plcReadRequest.getReadRequestItems().get(i);
162 
163                         ResponseCode responseCode = decodeResponseCode(payloadItem.getReturnCode());
164 
165                         ReadResponseItem responseItem;
166                         // Something went wrong.
167                         if(responseCode != ResponseCode.OK) {
168                             responseItem = new ReadResponseItem(requestItem, responseCode, null);
169                         }
170                         // All Ok.
171                         else {
172                             byte[] data = payloadItem.getData();
173                             Class<?> datatype = requestItem.getDatatype();
174                             List<Object> value = decodeData(datatype, data);
175                             responseItem = new ReadResponseItem(requestItem, responseCode, value);
176                         }
177                         responseItems.add(responseItem);
178                     }
179                     response = new PlcReadResponse(plcReadRequest, responseItems);
180                 }
181 
182                 // Handle the response to a write request.
183                 else if (requestContainer.getRequest() instanceof PlcWriteRequest) {
184                     PlcWriteRequest plcWriteRequest = (PlcWriteRequest) requestContainer.getRequest();
185                     List<WriteResponseItem> responseItems = new LinkedList<>();
186                     VarPayload payload = responseMessage.getPayload(VarPayload.class);
187                     // If the numbers of items don't match, we're in big trouble as the only
188                     // way to know how to interpret the responses is by aligning them with the
189                     // items from the request as this information is not returned by the PLC.
190                     if(plcWriteRequest.getRequestItems().size() != payload.getPayloadItems().size()) {
191                         throw new PlcProtocolException(
192                             "The number of requested items doesn't match the number of returned items");
193                     }
194                     List<VarPayloadItem> payloadItems = payload.getPayloadItems();
195                     for (int i = 0; i < payloadItems.size(); i++) {
196                         VarPayloadItem payloadItem = payloadItems.get(i);
197 
198                         // Get the request item for this payload item
199                         WriteRequestItem requestItem = plcWriteRequest.getRequestItems().get(i);
200 
201                         // A write response contains only the return code for every item.
202                         ResponseCode responseCode = decodeResponseCode(payloadItem.getReturnCode());
203 
204                         WriteResponseItem responseItem = new WriteResponseItem(requestItem, responseCode);
205                         responseItems.add(responseItem);
206                     }
207 
208                     response = new PlcWriteResponse(plcWriteRequest, responseItems);
209                 }
210 
211                 // Confirm the response being handled.
212                 if (response != null) {
213                     requestContainer.getResponseFuture().complete(response);
214                 }
215             }
216         }
217     }

223     private VarParameterItem encodeVarParameterItem(Address address, TransportSize transportSize, int size) throws PlcProtocolException {
224         // Depending on the address type, generate the corresponding type of request item.
225         if (!(address instanceof S7Address)) {
226             throw new PlcProtocolException("Can only use S7Address types on S7 connection");
227         }
228         S7Address s7Address = (S7Address) address;
229         if (s7Address instanceof S7DataBlockAddress) {
230             S7DataBlockAddress s7DataBlockAddress = (S7DataBlockAddress) s7Address;
231             return new S7AnyVarParameterItem(
232                 SpecificationType.VARIABLE_SPECIFICATION, s7Address.getMemoryArea(),
233                 transportSize, (short) size,
234                 s7DataBlockAddress.getDataBlockNumber(), s7DataBlockAddress.getByteOffset(), (byte) 0);
235         } else if (s7Address instanceof S7BitAddress) {
236             S7BitAddress s7BitAddress = (S7BitAddress) s7Address;
237             return new S7AnyVarParameterItem(
238                 SpecificationType.VARIABLE_SPECIFICATION, s7Address.getMemoryArea(),
239                 transportSize, (short) size, (short) 0,
240                 s7Address.getByteOffset(), s7BitAddress.getBitOffset());
241         } else {
242             return new S7AnyVarParameterItem(
243                 SpecificationType.VARIABLE_SPECIFICATION, s7Address.getMemoryArea(),
244                 transportSize, (short) size, (short) 0,
245                 s7Address.getByteOffset(), (byte) 0);
246         }
247     }

249     private TransportSize encodeTransportSize(Class<?> datatype) {
250         if (datatype == Boolean.class) {
251             return TransportSize.BIT;
252         } else if (datatype == Byte.class) {
253             return TransportSize.BYTE;
254         } else if (datatype == Short.class) {
255             return TransportSize.WORD;
256         } else if (datatype == Calendar.class) {
257             return TransportSize.DATE_AND_TIME;
258         } else if (datatype == Float.class) {
259             return TransportSize.REAL;
260         } else if (datatype == Integer.class) {
261             return TransportSize.DWORD;
262         } else if (datatype == String.class) {
263             return TransportSize.CHAR;
264         }
265         return null;
266     }

268     private DataTransportSize encodeDataTransportSize(Class<?> datatype) {
269         if (datatype == Boolean.class) {
270             return DataTransportSize.BIT;
271         } else if (datatype == Byte.class) {
272             return DataTransportSize.BYTE_WORD_DWORD;
273         } else if (datatype == Short.class) {
274             return DataTransportSize.BYTE_WORD_DWORD;
275         } else if (datatype == Calendar.class) {
276             // TODO: Decide what to do here ...
277             return null;
278         } else if (datatype == Float.class) {
279             return DataTransportSize.REAL;
280         } else if (datatype == Integer.class) {
281             return DataTransportSize.BYTE_WORD_DWORD;
282         } else if (datatype == String.class) {
283             return DataTransportSize.OCTET_STRING;
284         }
285         return null;
286     }

288     private byte[] encodeData(Object[] values) {
289         if(values.length == 0) {
290             return null;
291         }
292         byte[] result = null;
293         Class valueType = values[0].getClass();
294         if (valueType == Boolean.class) {
295             // TODO: Check if this is true and the result is not Math.ceil(values.lenght / 8)
296             result = new byte[values.length * 1];
297             for(int i = 0; i < values.length; i++) {
298                 result[i] = (byte) (((Boolean) values[i]) ? 0x01 : 0x00);
299             }
300         } else if (valueType == Byte[].class) {
301             result = new byte[values.length * 1];
302             for(int i = 0; i < values.length; i++) {
303                 result[i] = (byte) values[i];
304             }
305         } else if (valueType == Short.class) {
306             result = new byte[values.length * 2];
307             for(int i = 0; i < values.length; i++) {
308                 short intValue = (short) values[i];
309                 result[i * 2] = (byte) ((intValue & 0xff00) >> 8);
310                 result[(i * 2) + 1] = (byte) (intValue & 0xff);
311             }
312         } else if (valueType == Integer.class) {
313             result = new byte[values.length * 4];
314             for(int i = 0; i < values.length; i++) {
315                 int intValue = (int) values[i];
316                 result[i * 4] = (byte) ((intValue & 0xff000000) >> 24);
317                 result[(i * 4) + 1] = (byte) ((intValue & 0x00ff0000) >> 16);
318                 result[(i * 4) + 2] = (byte) ((intValue & 0x0000ff00) >> 8);
319                 result[(i * 4) + 3] = (byte) (intValue & 0xff);
320             }
321         } else if (valueType == Calendar.class) {
322             result = null;
323         } else if (valueType == Float.class) {
324             result = new byte[values.length * 4];
325             for(int i = 0; i < values.length; i++) {
326                 float floatValue = (float) values[i];
327                 int intValue = Float.floatToIntBits(floatValue);
328                 result[i * 4] = (byte) ((intValue & 0xff000000) >> 24);
329                 result[(i * 4) + 1] = (byte) ((intValue & 0x00ff0000) >> 16);
330                 result[(i * 4) + 2] = (byte) ((intValue & 0x0000ff00) >> 8);
331                 result[(i * 4) + 3] = (byte) (intValue & 0xff);
332             }
333         } else if (valueType == String.class) {
334             result = null;
335         }
336         return result;
337     }

343     private ResponseCode decodeResponseCode(DataTransportErrorCode dataTransportErrorCode) {
344         if(dataTransportErrorCode != null) {
345             switch (dataTransportErrorCode) {
346                 case OK:
347                     return ResponseCode.OK;
348                 case NOT_FOUND:
349                     return ResponseCode.NOT_FOUND;
350                 case INVALID_ADDRESS:
351                     return ResponseCode.INVALID_ADDRESS;
352             }
353         }
354         return ResponseCode.INTERNAL_ERROR;
355     }

357     private List<Object> decodeData(Class<?> datatype, byte[] s7Data) {
358         if(s7Data.length == 0) {
359             return null;
360         }
361         List<Object> result = new LinkedList<>();
362         for(int i = 0; i < s7Data.length; i++) {
363             if (datatype == Boolean.class) {
364                 result.add((s7Data[i] & 0x01) == 0x01);
365                 i+=1;
366             } else if (datatype == Byte.class) {
367                 result.add(s7Data[i]);
368                 i+=1;
369             } else if (datatype == Short.class) {
370                 result.add((short) (((s7Data[i] & 0xff) << 8) | (s7Data[i+1] & 0xff)));
371                 i+=2;
372             } else if (datatype == Integer.class) {
373                 result.add((((s7Data[i] & 0xff) << 24) | ((s7Data[i + 1] & 0xff) << 16) |
374                     ((s7Data[i + 2] & 0xff) << 8) | (s7Data[i + 3] & 0xff)));
375                 i+=4;
376             } else if (datatype == Float.class) {
377                 // Description of the Real number format:
378                 // https://www.sps-lehrgang.de/zahlenformate-step7/#c144
379                 // https://de.wikipedia.org/wiki/IEEE_754
380                 int intValue = (((s7Data[i] & 0xff) << 24) | ((s7Data[i + 1] & 0xff) << 16) |
381                     ((s7Data[i + 2] & 0xff) << 8) | (s7Data[i + 3] & 0xff));
382                 result.add(Float.intBitsToFloat(intValue));
383                 i+=4;
384             }
385         }
386         return result;
387     }
