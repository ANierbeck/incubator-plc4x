63      @Override
64      protected void encode(ChannelHandlerContext ctx, PlcRequestContainer msg, List<Object> out) throws Exception {
65          if (msg.getRequest() instanceof PlcReadRequest) {
66              List<VarParameterItem> parameterItems = new LinkedList<>();
67  
68              PlcReadRequest readRequest = (PlcReadRequest) msg.getRequest();
69              for (ReadRequestItem requestItem : readRequest.getReadRequestItems()) {
70                  // Try to get the correct S7 transport size for the given data type.
71                  // (Map PLC4X data type to S7 data type)
72                  TransportSize transportSize = encodeTransportSize(requestItem.getDatatype());
73                  if (transportSize == null) {
74                      throw new PlcException("Unknown transport size for datatype " + requestItem.getDatatype());
75                  }
76  
77                  // Depending on the address type, generate the corresponding type of request item.
78                  VarParameterItem varParameterItem = encodeVarParameterItem(requestItem.getAddress(), transportSize, requestItem.getSize());
79                  parameterItems.add(varParameterItem);
80              }
81              VarParameter readVarParameter =  new VarParameter(ParameterType.READ_VAR, parameterItems);
82  
83              // Assemble the request.
84              S7RequestMessage s7ReadRequest = new S7RequestMessage(MessageType.JOB,
85                  (short) tpduGenerator.getAndIncrement(), Collections.singletonList(readVarParameter),
86                  Collections.emptyList());
87  
88              requests.put(s7ReadRequest.getTpduReference(), msg);
89  
90              out.add(s7ReadRequest);
91          } else if (msg.getRequest() instanceof PlcWriteRequest) {
92              List<VarParameterItem> parameterItems = new LinkedList<>();
93              List<VarPayloadItem> payloadItems = new LinkedList<>();
94  
95              PlcWriteRequest writeRequest = (PlcWriteRequest) msg.getRequest();
96              for (WriteRequestItem requestItem : writeRequest.getRequestItems()) {
97                  // Try to get the correct S7 transport size for the given data type.
98                  // (Map PLC4X data type to S7 data type)
99                  TransportSize transportSize = encodeTransportSize(requestItem.getDatatype());
100                 if (transportSize == null) {
101                     throw new PlcException("Unknown transport size for datatype " + requestItem.getDatatype());
102                 }
103 
104                 // Depending on the address type, generate the corresponding type of request item.
105                 VarParameterItem varParameterItem = encodeVarParameterItem(
106                     requestItem.getAddress(), transportSize, requestItem.getValues().length);
107                 parameterItems.add(varParameterItem);
108 
109                 DataTransportSize dataTransportSize = encodeDataTransportSize(requestItem.getDatatype());
110                 if (dataTransportSize == null) {
111                     throw new PlcException("Unknown data transport size for datatype " + requestItem.getDatatype());
112                 }
113 
114                 VarPayloadItem varPayloadItem = new VarPayloadItem(
115                     DataTransportErrorCode.RESERVED, dataTransportSize, encodeData(requestItem.getValues()));
116 
117                 payloadItems.add(varPayloadItem);
118             }
119             VarParameter writeVarParameter =  new VarParameter(ParameterType.WRITE_VAR, parameterItems);
120             VarPayload writeVarPayload = new VarPayload(ParameterType.WRITE_VAR, payloadItems);
121 
122             // Assemble the request.
123             S7RequestMessage s7WriteRequest = new S7RequestMessage(MessageType.JOB,
124                 (short) tpduGenerator.getAndIncrement(), Collections.singletonList(writeVarParameter),
125                 Collections.singletonList(writeVarPayload));
126 
127             requests.put(s7WriteRequest.getTpduReference(), msg);
128 
129             out.add(s7WriteRequest);
130         }
131     }

133     @Override
134     protected void decode(ChannelHandlerContext ctx, S7Message msg, List<Object> out) throws Exception {
135         if(msg instanceof S7ResponseMessage) {
136             S7ResponseMessage responseMessage = (S7ResponseMessage) msg;
137             short tpduReference = responseMessage.getTpduReference();
138             if (requests.containsKey(tpduReference)) {
139                 PlcRequestContainer requestContainer = requests.remove(tpduReference);
140 
141                 PlcResponse response = null;
142 
143                 // Handle the response to a read request.
144                 if (requestContainer.getRequest() instanceof PlcReadRequest) {
145                     PlcReadRequest plcReadRequest = (PlcReadRequest) requestContainer.getRequest();
146 
147                     List<ReadResponseItem> responseItems = new LinkedList<>();
148                     VarPayload payload = responseMessage.getPayload(VarPayload.class);
149                     // If the numbers of items don't match, we're in big trouble as the only
150                     // way to know how to interpret the responses is by aligning them with the
151                     // items from the request as this information is not returned by the PLC.
152                     if(plcReadRequest.getReadRequestItems().size() != payload.getPayloadItems().size()) {
153                         throw new PlcProtocolException(
154                             "The number of requested items doesn't match the number of returned items");
155                     }
156                     List<VarPayloadItem> payloadItems = payload.getPayloadItems();
157                     final int noPayLoadItems = payloadItems.size();
158                     for (int i = 0; i < noPayLoadItems; i++) {
159                         VarPayloadItem payloadItem = payloadItems.get(i);
160 
161                         // Get the request item for this payload item
162                         ReadRequestItem requestItem = plcReadRequest.getReadRequestItems().get(i);
163 
164                         ResponseCode responseCode = decodeResponseCode(payloadItem.getReturnCode());
165 
166                         ReadResponseItem responseItem;
167                         // Something went wrong.
168                         if(responseCode != ResponseCode.OK) {
169                             responseItem = new ReadResponseItem(requestItem, responseCode, null);
170                         }
171                         // All Ok.
172                         else {
173                             byte[] data = payloadItem.getData();
174                             Class<?> datatype = requestItem.getDatatype();
175                             List<Object> value = decodeData(datatype, data);
176                             responseItem = new ReadResponseItem(requestItem, responseCode, value);
177                         }
178                         responseItems.add(responseItem);
179                     }
180                     response = new PlcReadResponse(plcReadRequest, responseItems);
181                 }
182 
183                 // Handle the response to a write request.
184                 else if (requestContainer.getRequest() instanceof PlcWriteRequest) {
185                     PlcWriteRequest plcWriteRequest = (PlcWriteRequest) requestContainer.getRequest();
186                     List<WriteResponseItem> responseItems = new LinkedList<>();
187                     VarPayload payload = responseMessage.getPayload(VarPayload.class);
188                     // If the numbers of items don't match, we're in big trouble as the only
189                     // way to know how to interpret the responses is by aligning them with the
190                     // items from the request as this information is not returned by the PLC.
191                     if(plcWriteRequest.getRequestItems().size() != payload.getPayloadItems().size()) {
192                         throw new PlcProtocolException(
193                             "The number of requested items doesn't match the number of returned items");
194                     }
195                     List<VarPayloadItem> payloadItems = payload.getPayloadItems();
196                     final int noPayLoadItems = payloadItems.size();
197                     for (int i = 0; i < noPayLoadItems; i++) {
198                         VarPayloadItem payloadItem = payloadItems.get(i);
199 
200                         // Get the request item for this payload item
201                         WriteRequestItem requestItem = plcWriteRequest.getRequestItems().get(i);
202 
203                         // A write response contains only the return code for every item.
204                         ResponseCode responseCode = decodeResponseCode(payloadItem.getReturnCode());
205 
206                         WriteResponseItem responseItem = new WriteResponseItem(requestItem, responseCode);
207                         responseItems.add(responseItem);
208                     }
209 
210                     response = new PlcWriteResponse(plcWriteRequest, responseItems);
211                 }
212 
213                 // Confirm the response being handled.
214                 if (response != null) {
215                     requestContainer.getResponseFuture().complete(response);
216                 }
217             }
218         }
219     }

225     private VarParameterItem encodeVarParameterItem(Address address, TransportSize transportSize, int size) throws PlcProtocolException {
226         // Depending on the address type, generate the corresponding type of request item.
227         if (!(address instanceof S7Address)) {
228             throw new PlcProtocolException("Can only use S7Address types on S7 connection");
229         }
230         S7Address s7Address = (S7Address) address;
231         if (s7Address instanceof S7DataBlockAddress) {
232             S7DataBlockAddress s7DataBlockAddress = (S7DataBlockAddress) s7Address;
233             return new S7AnyVarParameterItem(
234                 SpecificationType.VARIABLE_SPECIFICATION, s7Address.getMemoryArea(),
235                 transportSize, (short) size,
236                 s7DataBlockAddress.getDataBlockNumber(), s7DataBlockAddress.getByteOffset(), (byte) 0);
237         } else if (s7Address instanceof S7BitAddress) {
238             S7BitAddress s7BitAddress = (S7BitAddress) s7Address;
239             return new S7AnyVarParameterItem(
240                 SpecificationType.VARIABLE_SPECIFICATION, s7Address.getMemoryArea(),
241                 transportSize, (short) size, (short) 0,
242                 s7Address.getByteOffset(), s7BitAddress.getBitOffset());
243         } else {
244             return new S7AnyVarParameterItem(
245                 SpecificationType.VARIABLE_SPECIFICATION, s7Address.getMemoryArea(),
246                 transportSize, (short) size, (short) 0,
247                 s7Address.getByteOffset(), (byte) 0);
248         }
249     }

251     private TransportSize encodeTransportSize(Class<?> datatype) {
252         if (datatype == Boolean.class) {
253             return TransportSize.BIT;
254         } else if (datatype == Byte.class) {
255             return TransportSize.BYTE;
256         } else if (datatype == Short.class) {
257             return TransportSize.WORD;
258         } else if (datatype == Calendar.class) {
259             return TransportSize.DATE_AND_TIME;
260         } else if (datatype == Float.class) {
261             return TransportSize.REAL;
262         } else if (datatype == Integer.class) {
263             return TransportSize.DWORD;
264         } else if (datatype == String.class) {
265             return TransportSize.CHAR;
266         }
267         return null;
268     }

270     private DataTransportSize encodeDataTransportSize(Class<?> datatype) {
271         if (datatype == Boolean.class) {
272             return DataTransportSize.BIT;
273         } else if (datatype == Byte.class) {
274             return DataTransportSize.BYTE_WORD_DWORD;
275         } else if (datatype == Short.class) {
276             return DataTransportSize.BYTE_WORD_DWORD;
277         } else if (datatype == Calendar.class) {
278             // TODO: Decide what to do here ...
279             return null;
280         } else if (datatype == Float.class) {
281             return DataTransportSize.REAL;
282         } else if (datatype == Integer.class) {
283             return DataTransportSize.BYTE_WORD_DWORD;
284         } else if (datatype == String.class) {
285             return DataTransportSize.OCTET_STRING;
286         }
287         return null;
288     }

290     private byte[] encodeData(Object[] values) {
291         final int length = values.length;
292         if (length == 0) {
293             return new byte[]{};
294         }
295         byte[] result = null;
296         Class valueType = values[0].getClass();
297         if (valueType == Boolean.class) {
298             // TODO: Check if this is true and the result is not Math.ceil(values.lenght / 8)
299             result = new byte[length];
300             for (int i = 0; i < length; i++) {
301                 result[i] = (byte) (((Boolean) values[i]) ? 0x01 : 0x00);
302             }
303         } else if (valueType == Byte.class) {
304             result = new byte[length];
305             for (int i = 0; i < length; i++) {
306                 result[i] = (byte) values[i];
307             }
308         } else if (valueType == Short.class) {
309             result = new byte[length * 2];
310             for (int i = 0; i < length; i++) {
311                 short intValue = (short) values[i];
312                 result[i * 2] = (byte) ((intValue & 0xff00) >> 8);
313                 result[(i * 2) + 1] = (byte) (intValue & 0xff);
314             }
315         } else if (valueType == Integer.class) {
316             result = new byte[length * 4];
317             for (int i = 0; i < length; i++) {
318                 int intValue = (int) values[i];
319                 result[i * 4] = (byte) ((intValue & 0xff000000) >> 24);
320                 result[(i * 4) + 1] = (byte) ((intValue & 0x00ff0000) >> 16);
321                 result[(i * 4) + 2] = (byte) ((intValue & 0x0000ff00) >> 8);
322                 result[(i * 4) + 3] = (byte) (intValue & 0xff);
323             }
324         } else if (valueType == Calendar.class) {
325             result = null;
326         } else if (valueType == Float.class) {
327             result = new byte[length * 4];
328             for (int i = 0; i < length; i++) {
329                 float floatValue = (float) values[i];
330                 int intValue = Float.floatToIntBits(floatValue);
331                 result[i * 4] = (byte) ((intValue & 0xff000000) >> 24);
332                 result[(i * 4) + 1] = (byte) ((intValue & 0x00ff0000) >> 16);
333                 result[(i * 4) + 2] = (byte) ((intValue & 0x0000ff00) >> 8);
334                 result[(i * 4) + 3] = (byte) (intValue & 0xff);
335             }
336         } else if (valueType == String.class) {
337             int size = 0;
338             for (Object value : values) {
339                 size = size + ((String) value).length();
340             }
341             result = new byte[size + length];
342             int j = 0;
343             for (Object value : values) {
344                 String str = (String) value;
345                 for (int i = 0; i < str.length(); i++) {
346                     result[j++] = (byte) str.charAt(i);
347                 }
348                 result[j++] = (byte) 0x0;
349             }
350         }
351         return result;
352     }

358     private ResponseCode decodeResponseCode(DataTransportErrorCode dataTransportErrorCode) {
359         if (dataTransportErrorCode == null) {
360             return ResponseCode.INTERNAL_ERROR;
361         }
362         switch (dataTransportErrorCode) {
363             case OK:
364                 return ResponseCode.OK;
365             case NOT_FOUND:
366                 return ResponseCode.NOT_FOUND;
367             case INVALID_ADDRESS:
368                 return ResponseCode.INVALID_ADDRESS;
369             default:
370                 return ResponseCode.INTERNAL_ERROR;
371         }
372     }

374     private List<Object> decodeData(Class<?> datatype, byte[] s7Data) throws PlcProtocolException {
375         List<Object> result = new LinkedList<>();
376         int i = 0;
377         final int length = s7Data.length;
378         
379         while (i < length) {
380             if (datatype == Boolean.class) {
381                 result.add((s7Data[i] & 0x01) == 0x01);
382                 i+=1;
383             } else if (datatype == Byte.class) {
384                 result.add(s7Data[i]);
385                 i+=1;
386             } else if (datatype == Short.class) {
387                 result.add((short) (((s7Data[i] & 0xff) << 8) | (s7Data[i+1] & 0xff)));
388                 i+=2;
389             } else if (datatype == Integer.class) {
390                 result.add(((s7Data[i] & 0xff) << 24) | ((s7Data[i + 1] & 0xff) << 16) |
391                     ((s7Data[i + 2] & 0xff) << 8) | (s7Data[i + 3] & 0xff));
392                 i+=4;
393             } else if (datatype == Float.class) {
394                 // Description of the Real number format:
395                 // https://www.sps-lehrgang.de/zahlenformate-step7/#c144
396                 // https://de.wikipedia.org/wiki/IEEE_754
397                 int intValue = ((s7Data[i] & 0xff) << 24) | ((s7Data[i + 1] & 0xff) << 16) |
398                     ((s7Data[i + 2] & 0xff) << 8) | (s7Data[i + 3] & 0xff);
399                 result.add(Float.intBitsToFloat(intValue));
400                 i+=4;
401             } else if (datatype == String.class) {
402                 StringBuilder builder = new StringBuilder();
403                 while (s7Data[i] != (byte) 0x0) {
404                     builder.append(s7Data[i]);
405                     i++;
406                 }
407                 i++; // skip terminating character
408                 result.add(builder.toString());
409             } else {
410                 throw new PlcProtocolException("Unsupported datatype " + datatype.getSimpleName());
411             }
412         }
413         return result;
414     }
