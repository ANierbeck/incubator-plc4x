19  package org.apache.plc4x.java.isotp.netty.model.types;

21  import java.util.HashMap;
22  import java.util.Map;

24  /**
25   * Generally based upon RFC 905, but there were some inconsistencies in the spec:
26   * - RESIDUAL_ERROR_RATE was defined twice with different codes
27   * - The code 0x89 was defined for two types.
28   * This version was extracted from information generated by the PcapGenerator.
29   */
30  public enum ParameterCode {
31      ATN_EXTENDED_CHECKSUM_32BIT((byte) 0x08),
32      ATN_EXTENDED_CHECKSUM_16BIT((byte) 0x09),
33      ACKNOWLEDGE_TIME((byte) 0x85),
34      RESIDUAL_ERROR_RATE((byte) 0x86),
35      PRIORITY((byte) 0x87),
36      TRANSIT_DELAY((byte) 0x88),
37      THROUGHPUT((byte) 0x89),
38      SEQUENCE_NUMBER((byte) 0x8A),
39      REASSIGNMENT((byte) 0x8B),
40      FLOW_CONTROL((byte) 0x8C),
41      TPDU_SIZE((byte) 0xC0),
42      CALLING_TSAP((byte) 0xC1),
43      CALLED_TSAP((byte) 0xC2),
44      CHECKSUM((byte) 0xC3),
45      VERSION((byte) 0xC4),
46      SECURITY_PARAMETERS((byte) 0xC5),
47      ADDITIONAL_OPTION_SELECTION((byte) 0xC6),
48      ALTERNATIVE_PROTOCOL_CLASS((byte) 0xC7),
49      DISCONNECT_ADDITIONAL_INFORMATION((byte) 0xE0),
50      PREFERRED_MAX_PDU_SIZE((byte) 0xF0),
51      INACTIVITY_TIMER((byte) 0xF2);
52  
53      private byte code;
54  
55      ParameterCode(byte code) {
56          this.code = code;
57      }
58  
59      public byte getCode() {
60          return code;
61      }
62  
63      private static Map<Byte, ParameterCode> map = null;
64  
65      public static ParameterCode valueOf(byte code) {
66          if (map == null) {
67              map = new HashMap<>();
68              for (ParameterCode parameterCode : ParameterCode.values()) {
69                  map.put(parameterCode.code, parameterCode);
70              }
71          }
72          return map.get(code);
73      }
