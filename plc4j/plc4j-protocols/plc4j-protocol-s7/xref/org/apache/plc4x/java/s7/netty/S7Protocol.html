62          this.maxAmqCallee = requestedMaxAmqCallee;

66      @Override
67      public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
68          if (evt instanceof S7ConnectionEvent &&
69              ((S7ConnectionEvent) evt).getState() == S7ConnectionState.ISO_TP_CONNECTION_RESPONSE_RECEIVED) {
70              // Setup Communication
71              SetupCommunicationRequestMessage setupCommunicationRequest =
72                  new SetupCommunicationRequestMessage((short) 7, maxAmqCaller, maxAmqCallee, pduSize);
73  
74              ctx.channel().writeAndFlush(setupCommunicationRequest);
75          } else {
76              super.userEventTriggered(ctx, evt);
77          }
78      }

80      @Override
81      protected void encode(ChannelHandlerContext ctx, S7Message in, List<Object> out) {
82          logger.debug("S7 Message sent");
83  
84          ByteBuf buf = Unpooled.buffer();
85  
86          encodeHeader(in, buf);
87          encodeParameters(in, buf);
88          encodePayloads(in, buf);
89  
90          out.add(new DataTpdu(true, (byte) 1, Collections.emptyList(), buf));
91      }

93      private void encodePayloads(S7Message in, ByteBuf buf) {
94          for (S7Payload payload : in.getPayloads()) {
95              ParameterType parameterType = payload.getType();
96              if (parameterType == ParameterType.READ_VAR || parameterType == ParameterType.WRITE_VAR) {
97                  VarPayload varPayload = (VarPayload) payload;
98                  for (VarPayloadItem payloadItem : varPayload.getPayloadItems()) {
99                      buf.writeByte(payloadItem.getReturnCode().getCode());
100                     buf.writeByte(payloadItem.getDataTransportSize().getCode());
101                     buf.writeShort(payloadItem.getData().length);
102                     buf.writeBytes(payloadItem.getData());
103                 }
104             }
105         }
106     }

108     private void encodeParameters(S7Message in, ByteBuf buf) {
109         for (S7Parameter s7Parameter : in.getParameters()) {
110             buf.writeByte(s7Parameter.getType().getCode());
111             switch (s7Parameter.getType()) {
112                 case READ_VAR:
113                 case WRITE_VAR:
114                     encodeReadWriteVar(buf, (VarParameter) s7Parameter);
115                     break;
116                 case SETUP_COMMUNICATION:
117                     encodeSetupCommunication(buf, (SetupCommunicationParameter) s7Parameter);
118                     break;
119                 default:
120                     logger.error("writing this parameter type not implemented");
121             }
122         }
123     }

125     private void encodeHeader(S7Message in, ByteBuf buf) {
126         buf.writeByte(S7_PROTOCOL_MAGIC_NUMBER);
127         buf.writeByte(in.getMessageType().getCode());
128         // Reserved (is always constant 0x0000)
129         buf.writeShort((short) 0x0000);
130         // PDU Reference (Request Id, generated by the initiating node)
131         buf.writeShort(in.getTpduReference());
132         // S7 message parameters length
133         buf.writeShort(getParametersLength(in.getParameters()));
134         // Data field length
135         buf.writeShort(getPayloadsLength(in.getPayloads()));
136         if (in instanceof S7ResponseMessage) {
137             S7ResponseMessage s7ResponseMessage = (S7ResponseMessage) in;
138             buf.writeByte(s7ResponseMessage.getErrorClass());
139             buf.writeByte(s7ResponseMessage.getErrorCode());
140         }
141     }

143     private void encodeSetupCommunication(ByteBuf buf, SetupCommunicationParameter s7Parameter) {
144         // Reserved (is always constant 0x00)
145         buf.writeByte((byte) 0x00);
146         buf.writeShort(s7Parameter.getMaxAmqCaller());
147         buf.writeShort(s7Parameter.getMaxAmqCallee());
148         buf.writeShort(s7Parameter.getPduLength());
149     }

151     private void encodeReadWriteVar(ByteBuf buf, VarParameter s7Parameter) {
152         List<VarParameterItem> items = s7Parameter.getItems();
153         // ReadRequestItem count (Read one variable at a time)
154         buf.writeByte((byte) items.size());
155         for (VarParameterItem item : items) {
156             VariableAddressingMode addressMode = item.getAddressingMode();
157             if (addressMode == VariableAddressingMode.S7ANY) {
158                 encodeS7AnyParameterItem(buf, (S7AnyVarParameterItem) item);
159             } else {
160                 logger.error("writing this item type not implemented");
161             }
162         }
163     }

165     private void encodeS7AnyParameterItem(ByteBuf buf, S7AnyVarParameterItem s7AnyRequestItem) {
166         buf.writeByte(s7AnyRequestItem.getSpecificationType().getCode());
167         // Length of this item (excluding spec type and length)
168         buf.writeByte((byte) 0x0a);
169         buf.writeByte(s7AnyRequestItem.getAddressingMode().getCode());
170         buf.writeByte(s7AnyRequestItem.getTransportSize().getCode());
171         buf.writeShort(s7AnyRequestItem.getNumElements());
172         buf.writeShort(s7AnyRequestItem.getDataBlockNumber());
173         buf.writeByte(s7AnyRequestItem.getMemoryArea().getCode());
174         // A S7 address is 3 bytes long. Unfortunately the byte-offset is NOT located in
175         // byte 1 and byte 2 and the bit offset in byte 3. Siemens used the last 3 bits of
176         // byte 3 for the bit-offset and the remaining 5 bits of byte 3 to contain the lowest
177         // 5 bits of the byte-offset. The highest 5 bits of byte 1 are probably left unused
178         // for future extensions.
179         buf.writeShort((short) (s7AnyRequestItem.getByteOffset() >> 5));
180         buf.writeByte((byte) ((
181                 (s7AnyRequestItem.getByteOffset() & 0x1F) << 3)
182                 | (s7AnyRequestItem.getBitOffset() & 0x07)));
183     }

185     @Override
186     protected void decode(ChannelHandlerContext ctx, IsoTPMessage in, List<Object> out) {
187         if (logger.isTraceEnabled()) {
188             logger.trace("Got Data: {}", ByteBufUtil.hexDump(in.getUserData()));
189         }
190         ByteBuf userData = in.getUserData();
191         if (userData.readableBytes() == 0) {
192             return;
193         }
194         logger.debug("S7 Message received");
195 
196         if (userData.readByte() != S7_PROTOCOL_MAGIC_NUMBER) {
197             logger.warn("Expecting S7 protocol magic number.");
198             logger.debug("Got Data: {}", ByteBufUtil.hexDump(userData));
199             return;
200         }
201 
202         MessageType messageType = MessageType.valueOf(userData.readByte());
203         boolean isResponse = messageType == MessageType.ACK_DATA;
204         userData.readShort();  // Reserved (is always constant 0x0000)
205         short tpduReference = userData.readShort();
206         short headerParametersLength = userData.readShort();
207         short userDataLength = userData.readShort();
208         byte errorClass = 0;
209         byte errorCode = 0;
210         if (isResponse) {
211             errorClass = userData.readByte();
212             errorCode = userData.readByte();
213         }
214 
215         List<S7Parameter> s7Parameters = new LinkedList<>();
216         SetupCommunicationParameter setupCommunicationParameter = null;
217         VarParameter readWriteVarParameter = null;
218         int i = 0;
219 
220         while (i < headerParametersLength) {
221             S7Parameter parameter = parseParameter(userData, isResponse, headerParametersLength - i);
222             s7Parameters.add(parameter);
223             if (parameter instanceof SetupCommunicationParameter) {
224                 setupCommunicationParameter = (SetupCommunicationParameter) parameter;
225             }
226             if (readWriteVarParameter == null)  {
227                 readWriteVarParameter = decodeReadWriteParameter(parameter);
228             }
229             i += getParameterLength(parameter);
230         }
231 
232         List<S7Payload> s7Payloads = decodePayloads(userData, isResponse, userDataLength, readWriteVarParameter);
233 
234         if (isResponse) {
235             setupCommunications(ctx, setupCommunicationParameter);
236             out.add(new S7ResponseMessage(messageType, tpduReference, s7Parameters, s7Payloads, errorClass, errorCode));
237         } else {
238             out.add(new S7RequestMessage(messageType, tpduReference, s7Parameters, s7Payloads));
239         }
240     }

242     private void setupCommunications(ChannelHandlerContext ctx, SetupCommunicationParameter setupCommunicationParameter) {
243         // If we got a SetupCommunicationParameter as part of the response
244         // we are currently in the process of establishing a connection with
245         // the PLC, so save some of the information in the session and tell
246         // the next layer to negotiate the connection parameters.
247         if (setupCommunicationParameter != null) {
248             maxAmqCaller = setupCommunicationParameter.getMaxAmqCaller();
249             maxAmqCallee = setupCommunicationParameter.getMaxAmqCallee();
250             pduSize = setupCommunicationParameter.getPduLength();
251 
252             // Send an event that setup is complete.
253             ctx.channel().pipeline().fireUserEventTriggered(
254                 new S7ConnectionEvent(S7ConnectionState.SETUP_COMPLETE));
255         }
256     }

258     private VarParameter decodeReadWriteParameter(S7Parameter parameter) {
259         VarParameter readWriteVarParameter = null;
260 
261         if (parameter instanceof VarParameter) {
262             ParameterType paramType = parameter.getType();
263             if (paramType == ParameterType.READ_VAR || paramType == ParameterType.WRITE_VAR) {
264                 readWriteVarParameter = (VarParameter) parameter;
265             }
266         }
267         return readWriteVarParameter;
268     }

270     private List<S7Payload> decodePayloads(ByteBuf userData, boolean isResponse, short userDataLength, VarParameter readWriteVarParameter) {
271         int i = 0;
272         List<S7Payload> s7Payloads = new LinkedList<>();
273         if (readWriteVarParameter != null) {
274             List<VarPayloadItem> payloadItems = new LinkedList<>();
275 
276             while (i < userDataLength) {
277                 DataTransportErrorCode dataTransportErrorCode = DataTransportErrorCode.valueOf(userData.readByte());
278                 // This is a response to a WRITE_VAR request (It only contains the return code for every sent item.
279                 if ((readWriteVarParameter.getType() == ParameterType.WRITE_VAR) && isResponse) {
280                     // Initialize a rudimentary payload (This is updated in the Plc4XS7Protocol class
281                     VarPayloadItem payload = new VarPayloadItem(dataTransportErrorCode, null, null);
282                     payloadItems.add(payload);
283                     i += 1;
284                 }
285                 // This is a response to a READ_VAR request.
286                 else if ((readWriteVarParameter.getType() == ParameterType.READ_VAR) && isResponse) {
287                     DataTransportSize dataTransportSize = DataTransportSize.valueOf(userData.readByte());
288                     short length = (dataTransportSize.isSizeInBits()) ?
289                         (short) Math.ceil(userData.readShort() / 8.0) : userData.readShort();
290                     byte[] data = new byte[length];
291                     userData.readBytes(data);
292                     // Initialize a rudimentary payload (This is updated in the Plc4XS7Protocol class
293                     VarPayloadItem payload = new VarPayloadItem(dataTransportErrorCode, dataTransportSize, data);
294                     payloadItems.add(payload);
295                     i += getPayloadLength(payload);
296                 }
297             }
298 
299             VarPayload varPayload = new VarPayload(readWriteVarParameter.getType(), payloadItems);
300             s7Payloads.add(varPayload);
301         }
302         return s7Payloads;
303     }

305     private S7Parameter parseParameter(ByteBuf in, boolean isResponse, int restLength) {
306         ParameterType parameterType = ParameterType.valueOf(in.readByte());
307         if (parameterType == null) {
308             logger.error("Could not find parameter type");
309             return null;
310         }
311         switch (parameterType) {
312             case CPU_SERVICES:
313                 // Just read in the rest of the header as content of this parameter.
314                 // Will have to do a lot more investigation on how this parameter is
315                 // constructed.
316                 byte[] cpuServices = new byte[restLength - 1];
317                 in.readBytes(cpuServices);
318                 return null;
319             case READ_VAR:
320             case WRITE_VAR:
321                 List<VarParameterItem> varParamameter = null;
322                 if (isResponse) {
323                     varParamameter = parseReadWriteVarParameter(in);
324                 }
325                 return new VarParameter(parameterType, varParamameter);
326             case SETUP_COMMUNICATION:
327                 // Reserved (is always constant 0x00)
328                 in.readByte();
329                 short callingMaxAmq = in.readShort();
330                 short calledMaxAmq = in.readShort();
331                 short pduLength = in.readShort();
332                 return new SetupCommunicationParameter(callingMaxAmq, calledMaxAmq, pduLength);
333             default:
334                 logger.error("Unimplemented parameter type: " + parameterType.name());
335         }
336         return null;
337     }

339     private List<VarParameterItem> parseReadWriteVarParameter(ByteBuf in) {
340         List<VarParameterItem> items = new LinkedList<>();
341         byte numItems = in.readByte();
342 
343         for (int i = 0; i < numItems; i++) {
344             SpecificationType specificationType = SpecificationType.valueOf(in.readByte());
345             // Length of the rest of this item.
346             byte itemLength = in.readByte();
347             if (itemLength != 0x0a) {
348                 logger.warn("Expecting a length of 10 here.");
349                 return items;
350             }
351             VariableAddressingMode variableAddressingMode = VariableAddressingMode.valueOf(in.readByte());
352             if (variableAddressingMode == VariableAddressingMode.S7ANY) {
353                 TransportSize transportSize = TransportSize.valueOf(in.readByte());
354                 short length = in.readShort();
355                 short dbNumber = in.readShort();
356                 MemoryArea memoryArea = MemoryArea.valueOf(in.readByte());
357                 short byteAddress = (short) (in.readShort() << 5);
358                 byte tmp = in.readByte();
359                 // Only the least 3 bits are the bit address, the
360                 byte bitAddress = (byte) (tmp & 0x07);
361                 // Bits 4-8 belong to the byte address
362                 byteAddress = (short) (byteAddress | (tmp >> 3));
363                 S7AnyVarParameterItem item = new S7AnyVarParameterItem(
364                         specificationType, memoryArea, transportSize,
365                         length, dbNumber, byteAddress, bitAddress);
366                 items.add(item);
367             } else {
368                 logger.error("Error parsing item type");
369                 return items;
370             }
371         }
372 
373         return items;
374     }

376     private short getParametersLength(List<S7Parameter> parameters) {
377         short l = 0;
378         if (parameters != null) {
379             for (S7Parameter parameter : parameters) {
380                 l += getParameterLength(parameter);
381             }
382         }
383         return l;
384     }

386     private short getPayloadsLength(List<S7Payload> payloads) {
387         short l = 0;
388         if (payloads == null) {
389             return 0;
390         }
391 
392         for (S7Payload payload : payloads) {
393             if(payload instanceof VarPayload) {
394                 VarPayload varPayload = (VarPayload) payload;
395                 for (VarPayloadItem payloadItem : varPayload.getPayloadItems()) {
396                     l += getPayloadLength(payloadItem);
397                 }
398             }
399         }
400         return l;
401     }

403     private short getParameterLength(S7Parameter parameter) {
404         if (parameter == null) {
405             return 0;
406         }
407 
408         switch (parameter.getType()) {
409             case READ_VAR:
410             case WRITE_VAR:
411                 return getReadWriteVarParameterLength((VarParameter) parameter);
412             case SETUP_COMMUNICATION:
413                 return 8;
414             default:
415                 logger.error("Not implemented");
416                 return 0;
417         }
418     }
419 
420     private short getReadWriteVarParameterLength(VarParameter parameter) {
421         short length = 2;
422         for (VarParameterItem varParameterItem : parameter.getItems()) {
423             VariableAddressingMode addressMode = varParameterItem.getAddressingMode();
424 
425             if (addressMode == VariableAddressingMode.S7ANY) {
426                 length += 12;
427             } else {
428                 logger.error("Not implemented");
429             }
430         }
431         return length;
432     }

434     private short getPayloadLength(VarPayloadItem payloadItem) {
435         if (payloadItem == null) {
436             return 0;
437         }
438         return (short) (4 + payloadItem.getData().length);
