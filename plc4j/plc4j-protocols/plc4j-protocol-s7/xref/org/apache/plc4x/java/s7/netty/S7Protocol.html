52      private static final byte S7_PROTOCOL_MAGIC_NUMBER = 0x32;
53  
54      private static final Logger logger = LoggerFactory.getLogger(S7Protocol.class);
55  
56      private short maxAmqCaller;
57      private short maxAmqCallee;
58      private short pduSize;
59  
60      public S7Protocol(short requestedMaxAmqCaller, short requestedMaxAmqCallee, short requestedPduSize) {
61          this.maxAmqCaller = requestedMaxAmqCaller;
62          this.maxAmqCallee = requestedMaxAmqCallee;
63          this.pduSize = requestedPduSize;
64      }
65  
66      @Override
67      public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
68          if (evt instanceof S7ConnectionEvent &&
69              ((S7ConnectionEvent) evt).getState() == S7ConnectionState.ISO_TP_CONNECTION_RESPONSE_RECEIVED) {
70              // Setup Communication
71              SetupCommunicationRequestMessage setupCommunicationRequest =
72                  new SetupCommunicationRequestMessage((short) 7, maxAmqCaller, maxAmqCallee, pduSize);
73  
74              ctx.channel().writeAndFlush(setupCommunicationRequest);
75          } else {
76              super.userEventTriggered(ctx, evt);
77          }
78      }

80      @Override
81      protected void encode(ChannelHandlerContext ctx, S7Message in, List<Object> out) {
82          logger.debug("S7 Message sent");
83  
84          ByteBuf buf = Unpooled.buffer();
85  
86          encodeHeader(in, buf);
87          encodeParameters(in, buf);
88          encodePayloads(in, buf);
89  
90          out.add(new DataTpdu(true, (byte) 1, Collections.emptyList(), buf));
91      }
92  
93      private void encodePayloads(S7Message in, ByteBuf buf) {
94          for (S7Payload payload : in.getPayloads()) {
95              ParameterType parameterType = payload.getType();
96              if (parameterType == ParameterType.READ_VAR || parameterType == ParameterType.WRITE_VAR) {
97                  VarPayload varPayload = (VarPayload) payload;
98                  for (VarPayloadItem payloadItem : varPayload.getPayloadItems()) {
99                      buf.writeByte(payloadItem.getReturnCode().getCode());
100                     buf.writeByte(payloadItem.getDataTransportSize().getCode());
101                     buf.writeShort(payloadItem.getData().length);
102                     buf.writeBytes(payloadItem.getData());
103                 }
104             }
105         }
106     }
107 
108     private void encodeParameters(S7Message in, ByteBuf buf) {
109         for (S7Parameter s7Parameter : in.getParameters()) {
110             buf.writeByte(s7Parameter.getType().getCode());
111             switch (s7Parameter.getType()) {
112                 case READ_VAR:
113                 case WRITE_VAR:
114                     encodeReadWriteVar(buf, (VarParameter) s7Parameter);
115                     break;
116                 case SETUP_COMMUNICATION:
117                     encodeSetupCommunication(buf, (SetupCommunicationParameter) s7Parameter);
118                     break;
119                 default:
120                     logger.error("writing this parameter type not implemented");
121             }
122         }
123     }
124 
125     private void encodeHeader(S7Message in, ByteBuf buf) {
126         buf.writeByte(S7_PROTOCOL_MAGIC_NUMBER);
127         buf.writeByte(in.getMessageType().getCode());
128         // Reserved (is always constant 0x0000)
129         buf.writeShort((short) 0x0000);
130         // PDU Reference (Request Id, generated by the initiating node)
131         buf.writeShort(in.getTpduReference());
132         // S7 message parameters length
133         buf.writeShort(getParametersLength(in.getParameters()));
134         // Data field length
135         buf.writeShort(getPayloadsLength(in.getPayloads()));
136         if (in instanceof S7ResponseMessage) {
137             S7ResponseMessage s7ResponseMessage = (S7ResponseMessage) in;
138             buf.writeByte(s7ResponseMessage.getErrorClass());
139             buf.writeByte(s7ResponseMessage.getErrorCode());
140         }
141     }
142 
143     private void encodeSetupCommunication(ByteBuf buf, SetupCommunicationParameter s7Parameter) {
144         // Reserved (is always constant 0x00)
145         buf.writeByte((byte) 0x00);
146         buf.writeShort(s7Parameter.getMaxAmqCaller());
147         buf.writeShort(s7Parameter.getMaxAmqCallee());
148         buf.writeShort(s7Parameter.getPduLength());
149     }
150 
151     private void encodeReadWriteVar(ByteBuf buf, VarParameter s7Parameter) {
152         List<VarParameterItem> items = s7Parameter.getItems();
153         // ReadRequestItem count (Read one variable at a time)
154         buf.writeByte((byte) items.size());
155         for (VarParameterItem item : items) {
156             VariableAddressingMode addressMode = item.getAddressingMode();
157             if (addressMode == VariableAddressingMode.S7ANY) {
158                 encodeS7AnyParameterItem(buf, (S7AnyVarParameterItem) item);
159             } else {
160                 logger.error("writing this item type not implemented");
161             }
162         }
163     }
164 
165     private void encodeS7AnyParameterItem(ByteBuf buf, S7AnyVarParameterItem s7AnyRequestItem) {
166         buf.writeByte(s7AnyRequestItem.getSpecificationType().getCode());
167         // Length of this item (excluding spec type and length)
168         buf.writeByte((byte) 0x0a);
169         buf.writeByte(s7AnyRequestItem.getAddressingMode().getCode());
170         buf.writeByte(s7AnyRequestItem.getTransportSize().getCode());
171         buf.writeShort(s7AnyRequestItem.getNumElements());
172         buf.writeShort(s7AnyRequestItem.getDataBlockNumber());
173         buf.writeByte(s7AnyRequestItem.getMemoryArea().getCode());
174         // A S7 address is 3 bytes long. Unfortunately the byte-offset is NOT located in
175         // byte 1 and byte 2 and the bit offset in byte 3. Siemens used the last 3 bits of
176         // byte 3 for the bit-offset and the remaining 5 bits of byte 3 to contain the lowest
177         // 5 bits of the byte-offset. The highest 5 bits of byte 1 are probably left unused
178         // for future extensions.
179         buf.writeShort((short) (s7AnyRequestItem.getByteOffset() >> 5));
180         buf.writeByte((byte) ((
181                 (s7AnyRequestItem.getByteOffset() & 0x1F) << 3)
182                 | (s7AnyRequestItem.getBitOffset() & 0x07)));
183     }

185     @Override
186     protected void decode(ChannelHandlerContext ctx, IsoTPMessage in, List<Object> out) {
187         if (logger.isTraceEnabled()) {
188             logger.trace("Got Data: {}", ByteBufUtil.hexDump(in.getUserData()));
189         }
190         ByteBuf userData = in.getUserData();
191         if (userData.readableBytes() == 0) {
192             return;
193         }
194         logger.debug("S7 Message received");
195 
196         if (userData.readByte() != S7_PROTOCOL_MAGIC_NUMBER) {
197             logger.warn("Expecting S7 protocol magic number.");
198             if (logger.isDebugEnabled()) {
199                 logger.debug("Got Data: {}", ByteBufUtil.hexDump(userData));
200             }
201             return;
202         }
203 
204         MessageType messageType = MessageType.valueOf(userData.readByte());
205         boolean isResponse = messageType == MessageType.ACK_DATA;
206         userData.readShort();  // Reserved (is always constant 0x0000)
207         short tpduReference = userData.readShort();
208         short headerParametersLength = userData.readShort();
209         short userDataLength = userData.readShort();
210         byte errorClass = 0;
211         byte errorCode = 0;
212         if (isResponse) {
213             errorClass = userData.readByte();
214             errorCode = userData.readByte();
215         }
216 
217         List<S7Parameter> s7Parameters = new LinkedList<>();
218         SetupCommunicationParameter setupCommunicationParameter = null;
219         VarParameter readWriteVarParameter = null;
220         int i = 0;
221 
222         while (i < headerParametersLength) {
223             S7Parameter parameter = parseParameter(userData, isResponse, headerParametersLength - i);
224             s7Parameters.add(parameter);
225             if (parameter instanceof SetupCommunicationParameter) {
226                 setupCommunicationParameter = (SetupCommunicationParameter) parameter;
227             }
228             if (readWriteVarParameter == null)  {
229                 readWriteVarParameter = decodeReadWriteParameter(parameter);
230             }
231             i += getParameterLength(parameter);
232         }
233 
234         List<S7Payload> s7Payloads = decodePayloads(userData, isResponse, userDataLength, readWriteVarParameter);
235 
236         if (isResponse) {
237             setupCommunications(ctx, setupCommunicationParameter);
238             out.add(new S7ResponseMessage(messageType, tpduReference, s7Parameters, s7Payloads, errorClass, errorCode));
239         } else {
240             out.add(new S7RequestMessage(messageType, tpduReference, s7Parameters, s7Payloads));
241         }
242     }

244     private void setupCommunications(ChannelHandlerContext ctx, SetupCommunicationParameter setupCommunicationParameter) {
245         // If we got a SetupCommunicationParameter as part of the response
246         // we are currently in the process of establishing a connection with
247         // the PLC, so save some of the information in the session and tell
248         // the next layer to negotiate the connection parameters.
249         if (setupCommunicationParameter != null) {
250             maxAmqCaller = setupCommunicationParameter.getMaxAmqCaller();
251             maxAmqCallee = setupCommunicationParameter.getMaxAmqCallee();
252             pduSize = setupCommunicationParameter.getPduLength();
253 
254             // Send an event that setup is complete.
255             ctx.channel().pipeline().fireUserEventTriggered(
256                 new S7ConnectionEvent(S7ConnectionState.SETUP_COMPLETE));
257         }
258     }

260     private VarParameter decodeReadWriteParameter(S7Parameter parameter) {
261         VarParameter readWriteVarParameter = null;
262 
263         if (parameter instanceof VarParameter) {
264             ParameterType paramType = parameter.getType();
265             if (paramType == ParameterType.READ_VAR || paramType == ParameterType.WRITE_VAR) {
266                 readWriteVarParameter = (VarParameter) parameter;
267             }
268         }
269         return readWriteVarParameter;
270     }

272     private List<S7Payload> decodePayloads(ByteBuf userData, boolean isResponse, short userDataLength, VarParameter readWriteVarParameter) {
273         int i = 0;
274         List<S7Payload> s7Payloads = new LinkedList<>();
275         if (readWriteVarParameter != null) {
276             List<VarPayloadItem> payloadItems = new LinkedList<>();
277 
278             while (i < userDataLength) {
279                 DataTransportErrorCode dataTransportErrorCode = DataTransportErrorCode.valueOf(userData.readByte());
280                 // This is a response to a WRITE_VAR request (It only contains the return code for every sent item.
281                 if ((readWriteVarParameter.getType() == ParameterType.WRITE_VAR) && isResponse) {
282                     // Initialize a rudimentary payload (This is updated in the Plc4XS7Protocol class
283                     VarPayloadItem payload = new VarPayloadItem(dataTransportErrorCode, null, null);
284                     payloadItems.add(payload);
285                     i += 1;
286                 }
287                 // This is a response to a READ_VAR request.
288                 else if ((readWriteVarParameter.getType() == ParameterType.READ_VAR) && isResponse) {
289                     DataTransportSize dataTransportSize = DataTransportSize.valueOf(userData.readByte());
290                     short length = (dataTransportSize.isSizeInBits()) ?
291                         (short) Math.ceil(userData.readShort() / 8.0) : userData.readShort();
292                     byte[] data = new byte[length];
293                     userData.readBytes(data);
294                     // Initialize a rudimentary payload (This is updated in the Plc4XS7Protocol class
295                     VarPayloadItem payload = new VarPayloadItem(dataTransportErrorCode, dataTransportSize, data);
296                     payloadItems.add(payload);
297                     i += getPayloadLength(payload);
298                 }
299             }
300 
301             VarPayload varPayload = new VarPayload(readWriteVarParameter.getType(), payloadItems);
302             s7Payloads.add(varPayload);
303         }
304         return s7Payloads;
305     }

307     private S7Parameter parseParameter(ByteBuf in, boolean isResponse, int restLength) {
308         ParameterType parameterType = ParameterType.valueOf(in.readByte());
309         if (parameterType == null) {
310             logger.error("Could not find parameter type");
311             return null;
312         }
313         switch (parameterType) {
314             case CPU_SERVICES:
315                 // Just read in the rest of the header as content of this parameter.
316                 // Will have to do a lot more investigation on how this parameter is
317                 // constructed.
318                 byte[] cpuServices = new byte[restLength - 1];
319                 in.readBytes(cpuServices);
320                 return null;
321             case READ_VAR:
322             case WRITE_VAR:
323                 List<VarParameterItem> varParamameter;
324                 byte numItems = in.readByte();
325                 if (!isResponse) {
326                     varParamameter = parseReadWriteVarParameter(in, numItems);
327                 } else {
328                     varParamameter = Collections.emptyList();
329                 }
330                 return new VarParameter(parameterType, varParamameter);
331             case SETUP_COMMUNICATION:
332                 // Reserved (is always constant 0x00)
333                 in.readByte();
334                 short callingMaxAmq = in.readShort();
335                 short calledMaxAmq = in.readShort();
336                 short pduLength = in.readShort();
337                 return new SetupCommunicationParameter(callingMaxAmq, calledMaxAmq, pduLength);
338             default:
339                 if (logger.isErrorEnabled()) {
340                     logger.error("Unimplemented parameter type: {}", parameterType.name());
341                 }
342         }
343         return null;
344     }

346     private List<VarParameterItem> parseReadWriteVarParameter(ByteBuf in, byte numItems) {
347         List<VarParameterItem> items = new LinkedList<>();
348         for (int i = 0; i < numItems; i++) {
349             SpecificationType specificationType = SpecificationType.valueOf(in.readByte());
350             // Length of the rest of this item.
351             byte itemLength = in.readByte();
352             if (itemLength != 0x0a) {
353                 logger.warn("Expecting a length of 10 here.");
354                 return items;
355             }
356             VariableAddressingMode variableAddressingMode = VariableAddressingMode.valueOf(in.readByte());
357             if (variableAddressingMode == VariableAddressingMode.S7ANY) {
358                 TransportSize transportSize = TransportSize.valueOf(in.readByte());
359                 short length = in.readShort();
360                 short dbNumber = in.readShort();
361                 MemoryArea memoryArea = MemoryArea.valueOf(in.readByte());
362                 short byteAddress = (short) (in.readShort() << 5);
363                 byte tmp = in.readByte();
364                 // Only the least 3 bits are the bit address, the
365                 byte bitAddress = (byte) (tmp & 0x07);
366                 // Bits 4-8 belong to the byte address
367                 byteAddress = (short) (byteAddress | (tmp >> 3));
368                 S7AnyVarParameterItem item = new S7AnyVarParameterItem(
369                         specificationType, memoryArea, transportSize,
370                         length, dbNumber, byteAddress, bitAddress);
371                 items.add(item);
372             } else {
373                 logger.error("Error parsing item type");
374                 return items;
375             }
376         }
377 
378         return items;
379     }

381     private short getParametersLength(List<S7Parameter> parameters) {
382         short l = 0;
383         if (parameters != null) {
384             for (S7Parameter parameter : parameters) {
385                 l += getParameterLength(parameter);
386             }
387         }
388         return l;
389     }
390 
391     private short getPayloadsLength(List<S7Payload> payloads) {
392         short l = 0;
393         if (payloads == null) {
394             return 0;
395         }
396 
397         for (S7Payload payload : payloads) {
398             if(payload instanceof VarPayload) {
399                 VarPayload varPayload = (VarPayload) payload;
400                 for (VarPayloadItem payloadItem : varPayload.getPayloadItems()) {
401                     l += getPayloadLength(payloadItem);
402                 }
403             }
404         }
405         return l;
406     }
407 
408     private short getParameterLength(S7Parameter parameter) {
409         if (parameter == null) {
410             return 0;
411         }
412 
413         switch (parameter.getType()) {
414             case READ_VAR:
415             case WRITE_VAR:
416                 return getReadWriteVarParameterLength((VarParameter) parameter);
417             case SETUP_COMMUNICATION:
418                 return 8;
419             default:
420                 logger.error("Not implemented");
421                 return 0;
422         }
423     }
424 
425     private short getReadWriteVarParameterLength(VarParameter parameter) {
426         short length = 2;
427         for (VarParameterItem varParameterItem : parameter.getItems()) {
428             VariableAddressingMode addressMode = varParameterItem.getAddressingMode();
429 
430             if (addressMode == VariableAddressingMode.S7ANY) {
431                 length += 12;
432             } else {
433                 logger.error("Not implemented");
434             }
435         }
436         return length;
437     }
438 
439     private short getPayloadLength(VarPayloadItem payloadItem) {
440         if (payloadItem == null) {
441             return 0;
442         }
443         return (short) (4 + payloadItem.getData().length);
444     }
