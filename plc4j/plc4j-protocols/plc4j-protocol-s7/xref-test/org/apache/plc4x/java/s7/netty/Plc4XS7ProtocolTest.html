159     private <T> T fakeValueFor(Class<T> type) {
160         if (type == Boolean.class) {
161             return (T) Boolean.TRUE;
162         } else if (type == Byte.class) {
163             return (T) Byte.valueOf((byte) 0x0000_0000);
164         } else if (type == Short.class) {
165             return (T) Short.valueOf((short) 123);
166         } else if (type == Calendar.class) {
167             return (T) Calendar.getInstance();
168         } else if (type == Float.class) {
169             return (T) Float.valueOf(123f);
170         } else if (type == Integer.class) {
171             return (T) Integer.valueOf(123);
172         } else if (type == String.class) {
173             return (T) "string";
174         } else {
175             throw new IllegalArgumentException("Type t not supported " + type);
176         }
177     }

179     private VarPayloadItem varPayloadItemFor(Class type) {
180         // TODO: fix example
181         final DataTransportSize size;
182         final byte[] data;
183         if (type == Boolean.class) {
184             size = DataTransportSize.BIT;
185             data = new byte[]{(byte) 0b0};
186         } else if (type == Byte.class) {
187             size = DataTransportSize.BYTE_WORD_DWORD;
188             data = new byte[]{(byte) 0b0000_0000};
189         } else if (type == Short.class) {
190             size = DataTransportSize.BYTE_WORD_DWORD;
191             data = new byte[]{(byte) 0b0000_0000, (byte) 0b0000_0000};
192         } else if (type == Calendar.class) {
193             size = DataTransportSize.BYTE_WORD_DWORD;
194             // TODO: what size is calender?
195             data = new byte[]{(byte) 0b0000_0000};
196         } else if (type == Float.class) {
197             size = DataTransportSize.BYTE_WORD_DWORD;
198             data = new byte[]{(byte) 0b0000_0000, (byte) 0b0000_0000, (byte) 0b0000_0000, (byte) 0b0000_0000};
199         } else if (type == Integer.class) {
200             size = DataTransportSize.INTEGER;
201             data = new byte[]{(byte) 0b0000_0000, (byte) 0b0000_0000, (byte) 0b0000_0000, (byte) 0b0000_0000};
202         } else if (type == String.class) {
203             size = DataTransportSize.BYTE_WORD_DWORD;
204             // TODO: what size is string?
205             data = new byte[]{(byte) 0b0000_0000};
206         } else {
207             throw new IllegalArgumentException("Type t not supported " + type);
208         }
209         return new VarPayloadItem(DataTransportErrorCode.OK, size, data);
210     }

212     private <T extends PlcRequest> PlcRequestContainer createMockedContainer(T initialRequest) {
213         return createMockedContainer(initialRequest, null);
214     }

216     private <T extends PlcRequest> PlcRequestContainer createMockedContainer(T initialRequest, Consumer<T> requestEnricher) {
217         Objects.requireNonNull(initialRequest);
218         PlcRequestContainer mock = mock(PlcRequestContainer.class, RETURNS_DEEP_STUBS);
219         if (requestEnricher != null) {
220             requestEnricher.accept(initialRequest);
221         }
222         when(mock.getRequest()).thenReturn(initialRequest);
223         if (initialRequest.getClass() == PlcReadRequest.class) {
224             return mock;
225         } else if (initialRequest.getClass() == PlcWriteRequest.class) {
226             return mock;
227         } else {
228             throw new IllegalArgumentException("Unsupported Type: " + initialRequest.getClass());
229         }
230     }
