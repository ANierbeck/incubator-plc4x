19  package org.apache.plc4x.java.s7;
20  
21  import org.apache.plc4x.java.PlcDriverManager;
22  import org.apache.plc4x.java.api.connection.PlcConnection;
23  import org.apache.plc4x.java.api.connection.PlcReader;
24  import org.apache.plc4x.java.api.messages.PlcReadRequest;
25  import org.apache.plc4x.java.api.messages.PlcReadResponse;
26  import org.apache.plc4x.java.api.model.Address;
27  import org.apache.plc4x.java.s7.netty.model.types.MemoryArea;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import java.util.Optional;
32  
33  public class S7PlcScanner {
34  
35      private final static Logger logger = LoggerFactory.getLogger(S7PlcScanner.class);
36  
37      /**
38       * Example code do demonstrate using the S7 Plc Driver.
39       *
40       * @param args ignored.
41       * @throws Exception something went wrong.
42       */
43      public static void main(String[] args) throws Exception {
44          // Create a connection to the S7 PLC (s7://{hostname/ip}/{racknumber}/{slotnumber})
45          logger.info("Connecting");
46          try (PlcConnection plcConnection = new PlcDriverManager().getConnection("s7://192.168.0.1/0/0")){
47              logger.info("Connected");
48  
49              Optional<PlcReader> reader = plcConnection.getReader();
50              // Check if this connection support reading of data.
51              if (reader.isPresent()) {
52                  PlcReader plcReader = reader.get();
53  
54                  for (MemoryArea memoryArea : MemoryArea.values()) {
55                      System.out.println(memoryArea);
56                      System.out.println("------------------------------------------");
57                      for(int i = 0; i < 8959; i++) {
58                          try {
59                              Address address;
60                              if(memoryArea == MemoryArea.DATA_BLOCKS) {
61                                  address = plcConnection.parseAddress("DATA_BLOCKS/1/" + i);
62                              } else {
63                                  address = plcConnection.parseAddress(memoryArea.name() + "/" + i);
64                              }
65                              PlcReadResponse plcReadResponse = plcReader.read(
66                                  new PlcReadRequest(Byte.class, address)).get();
67                              Byte data = (Byte) plcReadResponse.getResponseItems().get(0).getValues().get(0);
68                              if(data != null && data != 0) {
69                                  System.out.println(String.format(
70                                      "Response: Memory Area: %s Index: %d Value: %02X", memoryArea.name(), i, data));
71                              }
72                          } catch (Exception e) {
73                              e.printStackTrace();
74                          }
75                      }
76                  }
77              }
78          }
79          // Catch any exception or the application won't be able to finish if something goes wrong.
80          catch (Exception e) {
81              e.printStackTrace();
82          }
83          // The application would cleanly terminate after several seconds ... this just speeds things up.
84          System.exit(0);
85      }
