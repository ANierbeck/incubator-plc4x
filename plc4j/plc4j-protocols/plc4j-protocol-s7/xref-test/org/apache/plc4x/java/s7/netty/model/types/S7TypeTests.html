30      @Test
31      @Category(FastTests.class)
32      public void dataTransportErrorCode() {
33          DataTransportErrorCode dataTransportErrorCode = DataTransportErrorCode.INVALID_ADDRESS;
34  
35          assertThat(DataTransportErrorCode.valueOf((byte)5)).isEqualTo(DataTransportErrorCode.INVALID_ADDRESS).withFailMessage("5 incorrectly mapped");
36          assertThat(dataTransportErrorCode.getCode()).isEqualTo((byte)5).withFailMessage("code is not 5");
37      }
38  
39      @Test
40      @Category(FastTests.class)
41      public void unknownDataTransportErrorCode() {
42          DataTransportErrorCode dataTransportErrorCode = DataTransportErrorCode.INVALID_ADDRESS;
43  
44          assertThat(DataTransportErrorCode.valueOf((byte)0xFE)).isNull();
45      }

47      @Test
48      @Category(FastTests.class)
49      public void dataTransportSize() {
50          DataTransportSize dataTransportSize = DataTransportSize.DINTEGER;
51  
52          assertThat(DataTransportSize.valueOf((byte)6)).isEqualTo(DataTransportSize.DINTEGER).withFailMessage("6 incorrectly mapped");
53          assertThat(dataTransportSize.getCode()).isEqualTo((byte)6).withFailMessage("code is not 6");
54          assertThat(dataTransportSize.isSizeInBits()).isFalse().withFailMessage("Unexpected return from bit size");
55      }
