305     @Test
306     @Tag("fast")
307     public void encodeCallingParameter() throws Exception {
308         ArrayList<Parameter> parmameters = new ArrayList<>();
309         CallingTsapParameter callingParameter = new CallingTsapParameter(DeviceGroup.PG_OR_PC, (byte) 0x7, (byte) 0xe1); // slot number too big and overflows into rack
310         parmameters.add(callingParameter);
311         ErrorTpdu tpdu = new ErrorTpdu((short) 0x1, RejectCause.REASON_NOT_SPECIFIED, parmameters, buf);
312 
313         isoTPProtocol.encode(ctx, tpdu, out);
314 
315         assertTrue(out.size() == 1, "Message not decoded");
316 
317         ByteBuf userData = ((IsoOnTcpMessage) out.get(0)).getUserData();
318 
319         assertTrue(userData.writerIndex() == 9, "Incorrect message length");
320         assertTrue(userData.readByte() == (byte) 0x8, "Incorrect header length");
321         assertTrue(userData.readByte() == TpduCode.TPDU_ERROR.getCode(), "Incorrect Tpdu code");
322         assertTrue(userData.readShort() == (short) 0x1, "Incorrect destination reference code");
323         assertTrue(userData.readByte() == RejectCause.REASON_NOT_SPECIFIED.getCode(), "Incorrect reject cause code");
324         assertTrue(userData.readByte() == ParameterCode.CALLING_TSAP.getCode(), "Incorrect parameter code");
325         assertTrue(userData.readByte() == (byte) 0x2, "Incorrect parameter length");
326         assertTrue(userData.readByte() == DeviceGroup.PG_OR_PC.getCode(), "Incorrect device group code");
327         byte rackAndSlot = userData.readByte();
328         assertTrue((rackAndSlot & 0xf0) >> 4 == 0x7, "Incorrect rack number");
329         assertTrue((rackAndSlot & 0x0f) == (0xe1 & 0x0f), "Incorrect slot number");
330     }
