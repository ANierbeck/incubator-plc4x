342     @Tag("fast")
343     public void encodeCallingParameter() throws Exception {
344         ChannelHandlerContext ctx = new MockChannelHandlerContext();
345         ByteBuf buf = Unpooled.buffer();
346         ArrayList<Parameter> parmameters = new ArrayList<>();
347         CallingTsapParameter callingParameter = new CallingTsapParameter(DeviceGroup.PG_OR_PC, (byte) 0x7, (byte) 0xe1); // slot number too big and overflows into rack
348         parmameters.add(callingParameter);
349         ErrorTpdu tpdu = new ErrorTpdu((short)0x1, RejectCause.REASON_NOT_SPECIFIED, parmameters, buf);
350         ArrayList<Object> out = new ArrayList<>();
351 
352         isoTPProtocol.encode(ctx, tpdu, out);
353 
354         assertTrue(out.size() == 1, "Message not decoded");
355 
356         ByteBuf userData = ((IsoOnTcpMessage)out.get(0)).getUserData();
357 
358         assertTrue(userData.writerIndex() == 9, "Incorrect message length");
359         assertTrue(userData.readByte() == (byte)0x8, "Incorrect header length");
360         assertTrue(userData.readByte() == TpduCode.TPDU_ERROR.getCode(), "Incorrect Tpdu code");
361         assertTrue(userData.readShort() == (short)0x1, "Incorrect destination reference code");
362         assertTrue(userData.readByte() == RejectCause.REASON_NOT_SPECIFIED.getCode(), "Incorrect reject cause code");
363         assertTrue(userData.readByte() == ParameterCode.CALLING_TSAP.getCode(), "Incorrect parameter code");
364         assertTrue(userData.readByte() == (byte)0x2, "Incorrect parameter length");
365         assertTrue(userData.readByte() == DeviceGroup.PG_OR_PC.getCode(), "Incorrect device group code");
366         byte rackAndSlot = userData.readByte();
367         assertTrue((rackAndSlot & 0xf0) >> 4 == 0x7, "Incorrect rack number");
368         assertTrue((rackAndSlot & 0x0f) == (0xe1 & 0x0f), "Incorrect slot number");
369     }
