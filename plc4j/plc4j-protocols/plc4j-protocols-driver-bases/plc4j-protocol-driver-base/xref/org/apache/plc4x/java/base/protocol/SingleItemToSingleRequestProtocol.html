43  /**
44   * This layer can be used to split a {@link org.apache.plc4x.java.api.messages.PlcRequest} which addresses multiple {@link PlcField}s into multiple subsequent {@link org.apache.plc4x.java.api.messages.PlcRequest}s.
45   */
46  public class SingleItemToSingleRequestProtocol extends ChannelDuplexHandler {

87      public SingleItemToSingleRequestProtocol(PlcReader reader, PlcWriter writer, Timer timer) {
88          this(reader, writer, timer, true);
89      }

91      public SingleItemToSingleRequestProtocol(PlcReader reader, PlcWriter writer, Timer timer, boolean betterImplementationPossible) {
92          this(reader, writer, timer, TimeUnit.SECONDS.toMillis(30), betterImplementationPossible);
93      }

95      public SingleItemToSingleRequestProtocol(PlcReader reader, PlcWriter writer, Timer timer, long defaultReceiveTimeout, boolean betterImplementationPossible) {
96          this.reader = reader;
97          this.writer = writer;
98          this.timer = timer;
99          this.defaultReceiveTimeout = defaultReceiveTimeout;
100         if (betterImplementationPossible) {
101             String callStack = Arrays.stream(Thread.currentThread().getStackTrace())
102                 .map(StackTraceElement::toString)
103                 .collect(Collectors.joining("\n"));
104             LOGGER.warn("Unoptimized Usage of {} detected at:\n{}", this.getClass(), callStack);
105             LOGGER.info("Consider implementing item splitting native to the protocol.");
106         }
107     }

109     @Override
110     public void channelRegistered(ChannelHandlerContext ctx) throws Exception {
111         this.queue = new PendingWriteQueue(ctx);
112         this.scheduledTimeouts = new ConcurrentHashMap<>();
113         this.sentButUnacknowledgedSubContainer = new ConcurrentHashMap<>();
114         this.correlationToParentContainer = new ConcurrentHashMap<>();
115         this.containerCorrelationIdMap = new ConcurrentHashMap<>();
116         this.responsesToBeDelivered = new ConcurrentHashMap<>();
117         this.correlationIdGenerator = new AtomicInteger();
118         this.deliveredItems = new AtomicLong();
119         this.erroredItems = new AtomicLong();
120         this.deliveredContainers = new AtomicLong();
121         this.erroredContainers = new AtomicLong();
122         super.channelRegistered(ctx);
123     }

125     @Override
126     public void channelUnregistered(ChannelHandlerContext ctx) throws Exception {
127         this.queue.removeAndWriteAll();
128         this.scheduledTimeouts.clear();
129         this.sentButUnacknowledgedSubContainer.clear();
130         this.correlationToParentContainer.clear();
131         this.containerCorrelationIdMap.clear();
132         this.responsesToBeDelivered.clear();
133         this.correlationIdGenerator.set(0);
134         this.deliveredItems.set(0);
135         this.erroredItems.set(0);
136         this.deliveredContainers.set(0);
137         this.erroredContainers.set(0);
138         super.channelUnregistered(ctx);
139     }

141     @Override
142     public void channelInactive(ChannelHandlerContext ctx) throws Exception {
143         // Send everything so we get a proper failure for those pending writes
144         this.queue.removeAndWriteAll();
145         this.timer.stop();
146         this.scheduledTimeouts.clear();
147         this.sentButUnacknowledgedSubContainer.clear();
148         this.correlationToParentContainer.clear();
149         this.containerCorrelationIdMap.clear();
150         this.responsesToBeDelivered.clear();
151         this.correlationIdGenerator.set(0);
152         this.deliveredItems.set(0);
153         this.erroredItems.set(0);
154         this.deliveredContainers.set(0);
155         this.erroredContainers.set(0);
156         super.channelInactive(ctx);
157     }

163     protected void tryFinish(Integer currentTdpu, InternalPlcResponse msg, CompletableFuture<InternalPlcResponse> originalResponseFuture) {
164         deliveredItems.incrementAndGet();
165         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> subPlcRequestContainer = sentButUnacknowledgedSubContainer.remove(currentTdpu);
166         LOGGER.info("{} got acknowledged", subPlcRequestContainer);
167         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> originalPlcRequestContainer = correlationToParentContainer.remove(currentTdpu);
168         if (originalPlcRequestContainer == null) {
169             LOGGER.warn("Unrelated package received {}", msg);
170             return;
171         }
172         Queue<InternalPlcResponse> correlatedResponseItems = responsesToBeDelivered.computeIfAbsent(originalPlcRequestContainer, ignore -> new ConcurrentLinkedQueue<>());
173         correlatedResponseItems.add(msg);
174         Set<Integer> integers = containerCorrelationIdMap.get(originalPlcRequestContainer);
175         integers.remove(currentTdpu);
176         if (integers.isEmpty()) {
177             deliveredContainers.incrementAndGet();
178             Timeout timeout = scheduledTimeouts.remove(originalPlcRequestContainer);
179             if (timeout != null) {
180                 timeout.cancel();
181             }
182 
183             InternalPlcResponse plcResponse;
184             if (originalPlcRequestContainer.getRequest() instanceof InternalPlcReadRequest) {
185                 InternalPlcReadRequest internalPlcReadRequest = (InternalPlcReadRequest) originalPlcRequestContainer.getRequest();
186                 HashMap<String, Pair<PlcResponseCode, BaseDefaultFieldItem>> fields = new HashMap<>();
187 
188                 correlatedResponseItems.stream()
189                     .map(InternalPlcReadResponse.class::cast)
190                     .map(InternalPlcReadResponse::getValues)
191                     .forEach(stringPairMap -> stringPairMap.forEach(fields::put));
192 
193                 plcResponse = new DefaultPlcReadResponse(internalPlcReadRequest, fields);
194             } else if (originalPlcRequestContainer.getRequest() instanceof InternalPlcWriteRequest) {
195                 InternalPlcWriteRequest internalPlcWriteRequest = (InternalPlcWriteRequest) originalPlcRequestContainer.getRequest();
196                 HashMap<String, PlcResponseCode> values = new HashMap<>();
197 
198                 correlatedResponseItems.stream()
199                     .map(InternalPlcWriteResponse.class::cast)
200                     .map(InternalPlcWriteResponse::getValues)
201                     .forEach(stringPairMap -> stringPairMap.forEach(values::put));
202 
203                 plcResponse = new DefaultPlcWriteResponse(internalPlcWriteRequest, values);
204             } else {
205                 errored(currentTdpu, new PlcProtocolException("Unknown type detected " + originalPlcRequestContainer.getRequest().getClass()), originalResponseFuture);
206                 return;
207             }
208             responsesToBeDelivered.remove(originalPlcRequestContainer);
209             containerCorrelationIdMap.remove(originalPlcRequestContainer);
210             originalResponseFuture.complete(plcResponse);
211         }
212     }
213 
214     protected void errored(Integer currentTdpu, Throwable throwable, CompletableFuture<InternalPlcResponse> originalResponseFuture) {
215         erroredItems.incrementAndGet();
216         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> subPlcRequestContainer = sentButUnacknowledgedSubContainer.remove(currentTdpu);
217         LOGGER.info("{} got errored", subPlcRequestContainer);
218 
219 
220         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> originalPlcRequestContainer = correlationToParentContainer.remove(currentTdpu);
221         if (originalPlcRequestContainer == null) {
222             LOGGER.warn("Unrelated error received tdpu:{}", currentTdpu, throwable);
223         } else {
224             erroredContainers.incrementAndGet();
225             Timeout timeout = scheduledTimeouts.remove(originalPlcRequestContainer);
226             if (timeout != null) {
227                 timeout.cancel();
228             }
229             responsesToBeDelivered.remove(originalPlcRequestContainer);
230 
231             Set<Integer> tdpus = containerCorrelationIdMap.remove(originalPlcRequestContainer);
232             if (tdpus != null) {
233                 tdpus.forEach(tdpu -> {
234                     // TODO: technically the other items didn't error so do we increment?
235                     //erroredItems.incrementAndGet();
236                     sentButUnacknowledgedSubContainer.remove(tdpu);
237                     correlationToParentContainer.remove(tdpu);
238                 });
239             }
240 
241             LOGGER.warn("PlcRequestContainer {} and correlationId {} failed ", correlationToParentContainer, currentTdpu, throwable);
242             originalResponseFuture.completeExceptionally(throwable);
243         }
244     }

250     @Override
251     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) throws Exception {
252         if (msg instanceof PlcRequestContainer) {
253             @SuppressWarnings("unchecked")
254             PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> in = (PlcRequestContainer<InternalPlcRequest, InternalPlcResponse>) msg;
255             Set<Integer> tdpus = containerCorrelationIdMap.computeIfAbsent(in, plcRequestContainer -> ConcurrentHashMap.newKeySet());
256 
257             Timeout timeout = timer.newTimeout(timeout_ -> handleTimeout(timeout_, in, tdpus, System.nanoTime()), defaultReceiveTimeout, TimeUnit.MILLISECONDS);
258             scheduledTimeouts.put(in, timeout);
259 
260             // Create a promise that has to be called multiple times.
261             PromiseCombiner promiseCombiner = new PromiseCombiner();
262             InternalPlcRequest request = in.getRequest();
263             if (request instanceof InternalPlcFieldRequest) {
264                 InternalPlcFieldRequest internalPlcFieldRequest = (InternalPlcFieldRequest) request;
265 
266                 if (internalPlcFieldRequest instanceof InternalPlcReadRequest) {
267                     InternalPlcReadRequest internalPlcReadRequest = (InternalPlcReadRequest) internalPlcFieldRequest;
268                     internalPlcReadRequest.getNamedFields().forEach(field -> {
269                         ChannelPromise subPromise = new DefaultChannelPromise(promise.channel());
270 
271                         Integer tdpu = correlationIdGenerator.getAndIncrement();
272                         CompletableFuture<InternalPlcResponse> correlatedCompletableFuture = new CompletableFuture<>();
273                         // Important: don't chain to above as we want the above to be completed not the result of when complete
274                         correlatedCompletableFuture
275                             .thenApply(InternalPlcResponse.class::cast)
276                             .whenComplete((internalPlcResponse, throwable) -> {
277                                 if (throwable != null) {
278                                     errored(tdpu, throwable, in.getResponseFuture());
279                                 } else {
280                                     tryFinish(tdpu, internalPlcResponse, in.getResponseFuture());
281                                 }
282                             });
283                         PlcRequestContainer<CorrelatedPlcReadRequest, InternalPlcResponse> correlatedPlcRequestContainer = new PlcRequestContainer<>(CorrelatedPlcReadRequest.of(reader, field, tdpu), correlatedCompletableFuture);
284                         correlationToParentContainer.put(tdpu, in);
285                         queue.add(correlatedPlcRequestContainer, subPromise);
286                         if (!tdpus.add(tdpu)) {
287                             throw new IllegalStateException("AtomicInteger should not create duplicated ids: " + tdpu);
288                         }
289                         promiseCombiner.add((Future) subPromise);
290                     });
291                 } else if (internalPlcFieldRequest instanceof InternalPlcWriteRequest) {
292                     InternalPlcWriteRequest internalPlcWriteRequest = (InternalPlcWriteRequest) internalPlcFieldRequest;
293                     internalPlcWriteRequest.getNamedFieldTriples().forEach(fieldItemTriple -> {
294                         ChannelPromise subPromise = new DefaultChannelPromise(promise.channel());
295 
296                         Integer tdpu = correlationIdGenerator.getAndIncrement();
297                         CompletableFuture<InternalPlcResponse> correlatedCompletableFuture = new CompletableFuture<>()
298                             .thenApply(InternalPlcResponse.class::cast)
299                             .whenComplete((internalPlcResponse, throwable) -> {
300                                 if (throwable != null) {
301                                     errored(tdpu, throwable, in.getResponseFuture());
302                                 } else {
303                                     tryFinish(tdpu, internalPlcResponse, in.getResponseFuture());
304                                 }
305                             });
306                         PlcRequestContainer<CorrelatedPlcWriteRequest, InternalPlcResponse> correlatedPlcRequestContainer = new PlcRequestContainer<>(CorrelatedPlcWriteRequest.of(writer, fieldItemTriple, tdpu), correlatedCompletableFuture);
307                         correlationToParentContainer.put(tdpu, in);
308                         queue.add(correlatedPlcRequestContainer, subPromise);
309                         if (!tdpus.add(tdpu)) {
310                             throw new IllegalStateException("AtomicInteger should not create duplicated ids: " + tdpu);
311                         }
312                         promiseCombiner.add((Future) subPromise);
313                     });
314                     // TODO: add sub/unsub
315                 } else {
316                     throw new PlcProtocolException("Unmapped request type " + request.getClass());
317                 }
318             } else {
319                 ChannelPromise subPromise = new DefaultChannelPromise(promise.channel());
320                 queue.add(msg, subPromise);
321                 promiseCombiner.add((Future) subPromise);
322             }
323 
324             promiseCombiner.finish(promise);
325 
326             // Start sending the queue content.
327             trySendingMessages(ctx);
328         } else {
329             super.write(ctx, msg, promise);
330         }
331     }

337     @SuppressWarnings("unchecked")
338     protected synchronized void trySendingMessages(ChannelHandlerContext ctx) {
339         while (queue.size() > 0) {
340             // Get the RequestItem that is up next in the queue.
341             PlcRequestContainer currentItem = (PlcRequestContainer) queue.current();
342             InternalPlcRequest request = currentItem.getRequest();
343 
344             // Send the TPDU.
345             try {
346                 ChannelFuture channelFuture = queue.removeAndWrite();
347                 ctx.flush();
348                 if (channelFuture == null) {
349                     break;
350                 }
351             } catch (Exception e) {
352                 LOGGER.error("Error sending more queues messages", e);
353                 ctx.fireExceptionCaught(e);
354             }
355 
356             if (request instanceof CorrelatedPlcRequest) {
357                 CorrelatedPlcRequest correlatedPlcRequest = (CorrelatedPlcRequest) request;
358 
359                 // Add it to the list of sentButUnacknowledgedSubContainer.
360                 sentButUnacknowledgedSubContainer.put(correlatedPlcRequest.getTdpu(), currentItem);
361 
362                 LOGGER.debug("container with id {} sent: ", correlatedPlcRequest.getTdpu(), currentItem);
363             }
364         }
365         ctx.flush();
366     }

368     private void handleTimeout(Timeout timeout, PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> in, Set<Integer> tdpus, long scheduledAt) {
369         if (timeout.isCancelled()) {
370             LOGGER.debug("container {} with timeout {} got canceled", in, timeout);
371             return;
372         }
373         LOGGER.warn("container {} timed out:{}", in, timeout);
374         erroredContainers.incrementAndGet();
375         responsesToBeDelivered.remove(in);
376         containerCorrelationIdMap.remove(in);
377         tdpus.forEach(tdpu -> {
378             erroredItems.incrementAndGet();
379             sentButUnacknowledgedSubContainer.remove(tdpu);
380             correlationToParentContainer.remove(tdpu);
381         });
382         in.getResponseFuture().completeExceptionally(new PlcTimeoutException(System.nanoTime() - scheduledAt));
383     }

390     protected static class CorrelatedPlcReadRequest extends DefaultPlcReadRequest implements CorrelatedPlcRequest {
391 
392         protected final int tdpu;
393 
394         protected CorrelatedPlcReadRequest(PlcReader reader, LinkedHashMap<String, PlcField> fields, int tdpu) {
395             super(reader, fields);
396             this.tdpu = tdpu;
397         }
398 
399         protected static CorrelatedPlcReadRequest of(PlcReader reader, Pair<String, PlcField> stringPlcFieldPair, int tdpu) {
400             LinkedHashMap<String, PlcField> fields = new LinkedHashMap<>();
401             fields.put(stringPlcFieldPair.getKey(), stringPlcFieldPair.getValue());
402             return new CorrelatedPlcReadRequest(reader, fields, tdpu);
403         }
404 
405         @Override
406         public int getTdpu() {
407             return tdpu;
408         }
409     }

411     protected static class CorrelatedPlcWriteRequest extends DefaultPlcWriteRequest implements CorrelatedPlcRequest {
412 
413         private final int tdpu;
414 
415         public CorrelatedPlcWriteRequest(PlcWriter writer, LinkedHashMap<String, Pair<PlcField, BaseDefaultFieldItem>> fields, int tdpu) {
416             super(writer, fields);
417             this.tdpu = tdpu;
418         }
419 
420         public static CorrelatedPlcWriteRequest of(PlcWriter writer, Triple<String, PlcField, BaseDefaultFieldItem> fieldItemTriple, int tdpu) {
421             LinkedHashMap<String, Pair<PlcField, BaseDefaultFieldItem>> fields = new LinkedHashMap<>();
422             fields.put(fieldItemTriple.getLeft(), Pair.of(fieldItemTriple.getMiddle(), fieldItemTriple.getRight()));
423             return new CorrelatedPlcWriteRequest(writer, fields, tdpu);
424         }
425 
426         @Override
427         public int getTdpu() {
428             return tdpu;
429         }

432     // TODO: maybe export to jmx
433     public Map<String, Number> getStatistics() {
434         HashMap<String, Number> statistics = new HashMap<>();
435         statistics.put("queue", queue.size());
436         statistics.put("sentButUnacknowledgedSubContainer", sentButUnacknowledgedSubContainer.size());
437         statistics.put("correlationToParentContainer", correlationToParentContainer.size());
438         statistics.put("containerCorrelationIdMap", containerCorrelationIdMap.size());
439         statistics.put("responsesToBeDelivered", responsesToBeDelivered.size());
440         statistics.put("correlationIdGenerator", correlationIdGenerator.get());
441         statistics.put("deliveredItems", deliveredItems.get());
442         statistics.put("erroredItems", erroredItems.get());
443         statistics.put("deliveredContainers", deliveredContainers.get());
444         statistics.put("erroredContainers", erroredContainers.get());
445         return statistics;
446     }
