274     private static void generateFiles(File parentDir, String testName, int byteIndex, byte[] template, String matchXPath)
275         throws IOException, ParserConfigurationException, XPathExpressionException, InterruptedException, SAXException {
276         File testDir = new File(parentDir, testName);
277         if (!testDir.exists() && !testDir.mkdirs()) {
278             throw new FileExistsException("Error creating directory " + testDir.getAbsolutePath());
279         }
280 
281         // Initialize the heavy weight XML stuff.
282         DocumentBuilderFactory domFactory = DocumentBuilderFactory.newInstance();
283         domFactory.setNamespaceAware(true);
284         DocumentBuilder builder = domFactory.newDocumentBuilder();
285         XPathFactory factory = XPathFactory.newInstance();
286         XPath xpath = factory.newXPath();
287         XPathExpression expr = xpath.compile(matchXPath);
288 
289         // Iterate over all possible value the current byte could have.
290         for (int i = 0; i <= 255; i++) {
291 
292             // Generate a packet.
293             byte[] copy = Arrays.copyOf(template, template.length);
294             copy[byteIndex] = (byte) i;
295             File output = new File(testDir, i + ".pcapng");
296             FileUtils.writeByteArrayToFile(output, copy);
297 
298             // Use WireShark to decode the packet to an xml form.
299             File decodedOutput = new File(testDir, i + ".xml");
300             ProcessBuilder pb = new ProcessBuilder("tshark", "-T", "pdml", "-r", output.getAbsolutePath());
301             pb.redirectOutput(ProcessBuilder.Redirect.to(decodedOutput));
302             Process process = pb.start();
303             process.waitFor();
304 
305             // Check if a given XPath is found -> A valid parameter was found.
306             Document document = builder.parse(decodedOutput);
307             NodeList list = (NodeList) expr.evaluate(document, XPathConstants.NODESET);
308 
309             // If a valid parameter is found, output that to the console.
310             if (list.getLength() > 0) {
311                 String name = list.item(0).getAttributes().getNamedItem("showname").getNodeValue();
312                 String value = list.item(0).getAttributes().getNamedItem("value").getNodeValue();
313                 System.out.println("found option for " + testName + ": " + name + " = 0x" + value);
314             }
315         }
316     }
