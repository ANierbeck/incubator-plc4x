113   private <T> void checkRead(MockConnection connection, PlcReadRequest request, T value) throws InterruptedException, ExecutionException {
114     // this is really a tests of our mock tooling but knowing it's behaving as expected
115     // will help identify problems in the adapter/supplier/consumer
116     connection.setDataValue(request.getReadRequestItems().get(0).getAddress(), value);
117     
118     CompletableFuture<PlcReadResponse> cf = connection.read(request);
119     
120     Assertions.assertTrue(cf.isDone());
121     PlcReadResponse response = cf.get();
122     Assertions.assertEquals(value, response.getResponseItems().get(0).getValues().get(0));
123   }

125   private <T> void checkWrite(MockConnection connection, PlcWriteRequest request, T value) throws InterruptedException, ExecutionException {
126     // this is really a tests of our mock tooling but knowing it's behaving as expected
127     // will help identify problems in the adapter/supplier/consumer
128     connection.setDataValue(request.getRequestItems().get(0).getAddress(), value);

131

133     PlcWriteResponse response = cf.get();
134     Assertions.assertNotNull(response);
135     T writtenData = (T) connection.getDataValue(request.getRequestItems().get(0).getAddress());
136     if(writtenData.getClass().isArray()) {
137       writtenData = (T) Array.get(writtenData, 0);
138     }
139     Assertions.assertEquals(value, writtenData);

339   static <T> void checkSupplier(MockConnection connection, Address address, Supplier<T> supplier, Object ... values) throws Exception {
340     checkSupplier(0, connection, address, supplier, values);
341   }
342   private static <T> void checkSupplier(int readFailureCountTrigger, MockConnection connection, Address address, Supplier<T> supplier, Object ... values) throws Exception {
343     // verify that a read failure doesn't kill the consumer
344     // it logs (not verified) but returns null (as designed) and keeps working for the subsequent reads
345     connection.setReadException(readFailureCountTrigger, "This is a mock read exception");
346     int readCount = 0;
347     for (Object value : values) {
348       connection.setDataValue(address, value);
349       T readData = supplier.get();
350       // System.out.println("checkSupplier"+(readFailureCountTrigger > 0 ? "NEG" : "")+": value:"+value+" readData:"+readData);
351       if (readFailureCountTrigger <= 0)
352         Assertions.assertEquals(value, readData);
353       else {
354         if (++readCount != readFailureCountTrigger)
355           Assertions.assertEquals(value, readData);
356         else
357           Assertions.assertNull(readData);
358       }
359     }
360   }

508     if (writeFailureCountTrigger > 0)
509       connection.setWriteException(writeFailureCountTrigger, "This is a mock write exception");
510     int writeCount = 0;
511     Object previousValue = null;
512     for (Object value : values) {
513       
514       // build the JsonObject to consume
515       JsonObject jo = new JsonObject();
516       jo.addProperty("address", address.getAddress());
517       if (value instanceof Boolean)
518         jo.addProperty("value", (Boolean)value);
519       else if (value instanceof Number)
520         jo.addProperty("value", (Number)value);
521       else if (value instanceof String)
522         jo.addProperty("value", (String)value);
523       
524       consumer.accept(jo);
525       
526       @SuppressWarnings("unchecked")
527       T writtenData = (T) connection.getDataValue(address);
528       if(writtenData.getClass().isArray()) {
529         writtenData = (T) Array.get(writtenData, 0);
