19  package org.apache.plc4x.java.ads;
20  
21  import org.apache.plc4x.java.PlcDriverManager;
22  import org.apache.plc4x.java.api.connection.PlcConnection;
23  import org.apache.plc4x.java.api.connection.PlcReader;
24  import org.apache.plc4x.java.api.connection.PlcSubscriber;
25  import org.apache.plc4x.java.api.messages.PlcReadResponse;
26  import org.apache.plc4x.java.api.messages.PlcSubscriptionResponse;
27  import org.apache.plc4x.java.api.messages.PlcUnsubscriptionResponse;
28  import org.apache.plc4x.java.api.model.PlcConsumerRegistration;
29  
30  import java.util.Collection;
31  import java.util.concurrent.CompletableFuture;
32  import java.util.concurrent.TimeUnit;
33  
34  public class ManualPlc4XAdsTest {
35  
36      public static void main(String... args) throws Exception {
37          String connectionUrl;
38          if (args.length > 0 && "serial".equalsIgnoreCase(args[0])) {
39              System.out.println("Using serial");
40              connectionUrl = "ads:serial:///dev/ttys003/10.10.64.40.1.1:851/10.10.56.23.1.1:30000";
41          } else {
42              System.out.println("Using tcp");
43              connectionUrl = "ads:tcp://10.10.64.40/10.10.64.40.1.1:851/192.168.113.1.1.1:30000";
44          }
45          try (PlcConnection plcConnection = new PlcDriverManager().getConnection(connectionUrl)) {
46              System.out.println("PlcConnection " + plcConnection);
47  
48              PlcReader reader = plcConnection.getReader().orElseThrow(() -> new RuntimeException("No Reader found"));
49  
50              CompletableFuture<PlcReadResponse<?>> response = reader.read(builder -> builder.addItem("station", "Allgemein_S2.Station:BYTE"));
51              PlcReadResponse<?> readResponse = response.get();
52              System.out.println("Response " + readResponse);
53              Collection<Integer> stations = readResponse.getAllIntegers("station");
54              stations.forEach(System.out::println);
55  
56              PlcSubscriber plcSubscriber = plcConnection.getSubscriber().orElseThrow(() -> new RuntimeException("Subscribe not available"));
57  
58              CompletableFuture<PlcSubscriptionResponse> subscribeResponse = plcSubscriber.subscribe(builder -> builder.addChangeOfStateField("stationChange", "Allgemein_S2.Station:BYTE"));
59              PlcSubscriptionResponse plcSubscriptionResponse = subscribeResponse.get();
60  
61              PlcConsumerRegistration plcConsumerRegistration = plcSubscriber.register(System.out::println, plcSubscriptionResponse.getSubscriptionHandles());
62  
63              TimeUnit.SECONDS.sleep(5);
64  
65              plcSubscriber.unregister(plcConsumerRegistration);
66              CompletableFuture<PlcUnsubscriptionResponse> unsubscriptionResponse = plcSubscriber.unsubscribe(builder -> builder.addHandles(plcSubscriptionResponse.getSubscriptionHandles()));
67  
68              unsubscriptionResponse
69                  .get(5, TimeUnit.SECONDS);
70              System.out.println(unsubscriptionResponse);
71          }
72          System.exit(0);
73      }
74  }
