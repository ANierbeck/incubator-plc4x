90      @Test
91      @Tag("fast")
92      void s7MessageParameters() {
93          MessageType messageType = MessageType.USER_DATA;
94          short tpduReference = 1;
95          ArrayList<S7Parameter> s7Parameters = new ArrayList<>();
96          ArrayList<S7Payload> s7Payloads = new ArrayList<>();
97          ParameterType parameterType = ParameterType.READ_VAR;
98          ArrayList<VarParameterItem> parameterItems = new ArrayList<>();
99          SpecificationType specificationType = SpecificationType.VARIABLE_SPECIFICATION;
100         MemoryArea memoryArea = MemoryArea.DATA_BLOCKS;
101         TransportSize transportSize = TransportSize.INT;
102         short numElements = 1;
103         byte dataBlock = (byte) 0x1;
104         byte byteOffset = (byte) 0x10;
105         byte bitOffset = (byte) 0x0;
106 
107         parameterItems.add(new S7AnyVarParameterItem(specificationType, memoryArea, transportSize, numElements, dataBlock, byteOffset, bitOffset));
108 
109         VarParameter varParameter = new VarParameter(parameterType, parameterItems);
110 
111         s7Parameters.add(varParameter);
112 
113         S7RequestMessage message = new S7RequestMessage(messageType, tpduReference, s7Parameters, s7Payloads);
114 
115         assertTrue(message.getTpduReference() == tpduReference, "Unexpected tpdu value");
116         assertTrue(message.getMessageType() == MessageType.USER_DATA, "Unexpected message type");
117         assertTrue(message.getParameters().size() == 1, "Unexpected number of parameters");
118         assertTrue(message.getParameters().containsAll(s7Parameters), "Unexpected parameters");
119         assertTrue(message.getParameter(VarParameter.class).get().equals(varParameter), "Parameter missing");
120         assertTrue(!message.getParameter(CpuServicesParameter.class).isPresent(), "Contains unexpected parameter");
121         assertTrue(message.getPayloads().size() == 0, "Unexpected number of payloads");
122     }

124     @Test
125     @Tag("fast")
126     void s7MessagePayload() {
127         MessageType messageType = MessageType.USER_DATA;
128         short tpduReference = 1;
129         ArrayList<S7Parameter> s7Parameters = new ArrayList<>();
130         ArrayList<S7Payload> s7Payloads = new ArrayList<>();
131         ParameterType parameterType = ParameterType.WRITE_VAR;
132         ArrayList<VarPayloadItem> payloadItems = new ArrayList<>();
133         byte[] data = {(byte)0x79};
134         VarPayload varPayload;
135 
136         payloadItems.add(new VarPayloadItem(DataTransportErrorCode.OK, DataTransportSize.BIT, data));
137         varPayload = new VarPayload(parameterType, payloadItems);
138         s7Payloads.add(varPayload);
139 
140         S7RequestMessage message = new S7RequestMessage(messageType, tpduReference, s7Parameters, s7Payloads);
141 
142         assertTrue(message.getTpduReference() == tpduReference, "Unexpected tpdu value");
143         assertTrue(message.getMessageType() == MessageType.USER_DATA, "Unexpected message type");
144         assertTrue(message.getPayloads().size() == 1, "Unexpected number of payloads");
145         assertTrue(message.getPayloads().containsAll(s7Payloads), "Unexpected payloads");
146         assertTrue(message.getPayload(VarPayload.class).get().equals(varPayload), "Payload missing");
147         assertTrue(!message.getPayload(VarParameter.class).isPresent(), "Contains unexpected payload"); // No other parameter classes
148         assertTrue(message.getParameters().size() == 0, "Unexpected number of parameters");
149     }
