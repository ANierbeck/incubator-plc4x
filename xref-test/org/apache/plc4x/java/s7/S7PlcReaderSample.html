23  import org.apache.plc4x.java.api.connection.PlcReader;
24  import org.apache.plc4x.java.api.messages.specific.TypeSafePlcReadRequest;
25  import org.apache.plc4x.java.api.messages.specific.TypeSafePlcReadResponse;
26  import org.apache.plc4x.java.api.model.Address;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import java.util.Optional;
31  
32  public class S7PlcReaderSample {
33  
34      private static final Logger logger = LoggerFactory.getLogger(S7PlcReaderSample.class);
35  
36      /**
37       * Example code do demonstrate using the S7 Plc Driver.
38       *
39       * @param args ignored.
40       * @throws Exception something went wrong.
41       */
42      public static void main(String[] args) throws Exception {
43          // Create a connection to the S7 PLC (s7://{hostname/ip}/{racknumber}/{slotnumber})
44          logger.info("Connecting");
45          try (PlcConnection plcConnection = new PlcDriverManager().getConnection("s7://192.168.0.1/0/0")) {
46              logger.info("Connected");
47  
48              Optional<PlcReader> reader = plcConnection.getReader();
49              // Check if this connection support reading of data.
50              if (reader.isPresent()) {
51                  PlcReader plcReader = reader.get();
52  
53                  // Prepare some address object for accessing fields in the PLC.
54                  // ({memory-area}/{byte-offset}[/{bit-offset}]
55                  // "bit-offset is only specified if the requested type is "bit"
56                  // NOTICE: This format is probably only valid when using a S7 connection.
57                  Address inputs = plcConnection.parseAddress("INPUTS/0");
58                  Address outputs = plcConnection.parseAddress("OUTPUTS/0");
59  
60                  //////////////////////////////////////////////////////////
61                  // Read synchronously ...
62                  // NOTICE: the ".get()" immediately lets this thread pause till
63                  // the response is processed and available.
64                  TypeSafePlcReadResponse<Byte> plcReadResponse = plcReader.read(new TypeSafePlcReadRequest<>(Byte.class, inputs)).get();
65                  System.out.println("Inputs: " + plcReadResponse.getResponseItem()
66                      .orElseThrow(() -> new IllegalStateException("No response available"))
67                      .getValues().get(0));
68  
69                  //////////////////////////////////////////////////////////
70                  // Read asynchronously ...
71                  /*Calendar start = Calendar.getInstance();
72                  CompletableFuture<PlcReadResponse<Byte>> asyncResponse = plcReader.read(
73                      new BytePlcReadRequest(outputs));
74  
75                  asyncResponse.thenAccept(bytePlcReadResponse -> {
76                      Calendar end = Calendar.getInstance();
77                      Byte dataAsync = bytePlcReadResponse.getValue();
78                      System.out.println("Outputs: " + dataAsync + " (in " + (end.getTimeInMillis() - start.getTimeInMillis()) + "ms)");
79                  });
80  
81                  // Simulate doing something else ...
82                  while (true) {
83                      Thread.sleep(1);
84                      if (asyncResponse.isDone()) {
85                          break;
86                      }
87                  }*/
88              }
89          }
90          // Catch any exception or the application won't be able to finish if something goes wrong.
91          catch (Exception e) {
92              logger.error("S7 PLC reader sample", e);
93          }
94          // The application would cleanly terminate after several seconds ... this just speeds things up.
95          System.exit(0);
96      }
