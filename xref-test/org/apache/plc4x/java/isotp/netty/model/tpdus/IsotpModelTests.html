40      @Test
41      @Tag("Fast")
42      void errorTpdu() {
43          short destinationReference = 0x1;
44          RejectCause rejectCause = RejectCause.REASON_NOT_SPECIFIED;
45          List<Parameter> parameters = Collections.emptyList();
46          ByteBuf userData = Unpooled.buffer();
47  
48          userData.writeByte(0x7F);
49  
50          ErrorTpdu tpdu = new ErrorTpdu(destinationReference, rejectCause, parameters, userData);
51  
52          assertTrue(tpdu.getTpduCode() == TpduCode.TPDU_ERROR);
53          assertTrue(tpdu.getDestinationReference() == 0x1, "Unexpected destination reference");
54          assertTrue(tpdu.getRejectCause() == RejectCause.REASON_NOT_SPECIFIED);
55          assertTrue(tpdu.getParameters().isEmpty(), "Unexpected parameters");
56          assertTrue(tpdu.getUserData().readByte() == (byte) 0x7F, "Unexpected user data");
57      }
58  
59      @Test
60      @Tag("Fast")
61      void errorTpduParameter() {
62          short destinationReference = 0x1;
63          RejectCause rejectCause = RejectCause.REASON_NOT_SPECIFIED;
64          ArrayList<Parameter> parameters = new ArrayList<>();
65          ByteBuf userData = Unpooled.buffer();
66  
67          userData.writeByte(0x7F);
68  
69          ErrorTpdu tpdu = new ErrorTpdu(destinationReference, rejectCause, parameters, userData);
70  
71          parameters.add(new TpduSizeParameter(TpduSize.SIZE_1024));
72          parameters.add(new ChecksumParameter((byte) 0xFF));
73  
74          assertTrue(tpdu.getParameters().size() == 2, "Unexpected number of parameters");
75          assertTrue(tpdu.getParameters().containsAll(parameters), "Unexpected parameter");
76          assertTrue(tpdu.getParameter(ChecksumParameter.class).isPresent(), "Checksum parameter should exist");
77          assertTrue(!tpdu.getParameter(CallingTsapParameter.class).isPresent(), "CallingTsapParameter parameter should not exist");
78      }
79  
80      @Test
81      @Tag("Fast")
82      void dataTpdu() {
83          List<Parameter> parameters = Collections.emptyList();
84          ByteBuf userData = Unpooled.buffer();
85  
86          userData.writeByte(0x66);
87  
88          DataTpdu tpdu = new DataTpdu(true, (byte) 0x7F, parameters, userData);
89  
90          assertTrue(tpdu.getTpduCode() == TpduCode.DATA);
91          assertTrue(tpdu.isEot(), "Unexpected eot reference");
92          assertTrue(tpdu.getTpduRef() == 0x7F);
93          assertTrue(tpdu.getParameters().isEmpty(), "Unexpected parameters");
94          assertTrue(tpdu.getUserData().readByte() == (byte) 0x66, "Unexpected user data");
95      }
