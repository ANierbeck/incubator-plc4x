688     protected void dumpAndExit(ByteBuf byteBuf, int endOfLastBlock, int lastBlockSize, String message) {
689         int errorPos = byteBuf.readerIndex();
690         int lastBlockStart = errorPos - endOfLastBlock;
691         byteBuf.resetReaderIndex();
692         System.out.println("-------------------- ERROR --------------------");
693         String packetAsHexString = Hex.encodeHexString(byteBuf.array()).replaceAll("(.{2})", "$1 ").replaceAll("(.{48})", "$1\n");
694         StringTokenizer stringTokenizer = new StringTokenizer(packetAsHexString, "\n");
695         while (stringTokenizer.hasMoreElements()) {
696             String line = stringTokenizer.nextToken();
697             System.out.println(line);
698             if((errorPos < 16) && (errorPos >= 0)) {
699                 StringBuffer sb = new StringBuffer();
700                 for(int i = 0; i < errorPos - 1; i++) {
701                     sb.append("---");
702                 }
703                 sb.append("^");
704                 System.out.println(sb);
705                 System.out.println("Last block started: " + lastBlockStart + " bytes before error and had a size of: " + lastBlockSize);
706                 System.out.println(message);
707                 System.out.println("\n");
708             }
709             errorPos -= 16;
710         }
711         throw new RuntimeException("Error");
712     }

722             case (byte) 0x80: // 10000000
