19  package org.apache.plc4x.java.modbus.netty;
20  
21  import org.apache.commons.lang3.reflect.MethodUtils;
22  import org.junit.Test;
23  
24  import java.util.Arrays;
25  import java.util.BitSet;
26  
27  import static org.junit.Assert.assertEquals;
28  import static org.junit.Assert.assertSame;
29  
30  public class Plc4XModbusProtocolTest {
31  
32      @Test
33      public void coilEncoding() throws Exception {
34          byte[] bytes = (byte[]) MethodUtils.invokeMethod(new Plc4XModbusProtocol(), true, "produceCoilValue", Arrays.asList(false, false, true, false, false, false, true, false));
35          assertSame(1, bytes.length);
36          String s1 = String.format("%8s", Integer.toBinaryString(bytes[0] & 0xFF)).replace(' ', '0');
37          assertEquals("00100010", s1);
38  
39          // TODO: just for reference we could use a bitset too
40          BitSet bitSet = new BitSet(8);
41          bitSet.set(1);
42          bitSet.set(5);
43          byte[] bytes1 = bitSet.toByteArray();
44          assertSame(1, bytes1.length);
45  
46          String s2 = String.format("%8s", Integer.toBinaryString(bytes1[0] & 0xFF)).replace(' ', '0');
47          assertEquals("00100010", s2);
48      }
49  }
