209     @Test
210     @Tag("fast")
211     void bulkPlcWriteResponseGetValue() {
212         PlcWriteRequest plcWriteRequest = new PlcWriteRequest();
213         List<WriteResponseItem<?>> responseItems = new ArrayList<>();
214         MockAddress address = new MockAddress("mock:/DATA");
215         WriteRequestItem<Byte> writeRequestItem1 = new WriteRequestItem<>(Byte.class, address, (byte) 1);
216         WriteRequestItem<Byte> writeRequestItem2 = new WriteRequestItem<>(Byte.class, address, (byte) 1);
217         WriteResponseItem<Byte> writeResponseItem1 = new WriteResponseItem<>(writeRequestItem1, ResponseCode.OK);
218         WriteResponseItem<Byte> writeResponseItem2 = new WriteResponseItem<>(writeRequestItem2, ResponseCode.OK);
219         responseItems.add(writeResponseItem1);
220         responseItems.add(writeResponseItem2);
221         PlcWriteResponse plcWriteResponse = new PlcWriteResponse(plcWriteRequest, responseItems);
222         Optional<WriteResponseItem<Byte>> responseValue1 = plcWriteResponse.getValue(writeRequestItem1);
223         Optional<WriteResponseItem<Byte>> responseValue2 = plcWriteResponse.getValue(writeRequestItem2);
224         assertEquals(Optional.of(writeResponseItem1), responseValue1, "Unexpected items in response items");
225         assertEquals(Optional.of(writeResponseItem2), responseValue2, "Unexpected items in response items");
226     }
