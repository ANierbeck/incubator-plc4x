214     @Test
215     @Category(FastTests.class)
216     public void bulkPlcWriteResponseGetValue() {
217         PlcWriteRequest plcWriteRequest = new PlcWriteRequest();
218         List<WriteResponseItem<?>> responseItems = new ArrayList<>();
219         MockAddress address = new MockAddress("mock:/DATA");
220         WriteRequestItem<Byte> writeRequestItem1 = new WriteRequestItem<>(Byte.class, address, (byte) 1);
221         WriteRequestItem<Byte> writeRequestItem2 = new WriteRequestItem<>(Byte.class, address, (byte) 1);
222         WriteResponseItem<Byte> writeResponseItem1 = new WriteResponseItem<>(writeRequestItem1, ResponseCode.OK);
223         WriteResponseItem<Byte> writeResponseItem2 = new WriteResponseItem<>(writeRequestItem2, ResponseCode.OK);
224         responseItems.add(writeResponseItem1);
225         responseItems.add(writeResponseItem2);
226         PlcWriteResponse plcWriteResponse = new PlcWriteResponse(plcWriteRequest, responseItems);
227         Optional<WriteResponseItem<Byte>> responseValue1 = plcWriteResponse.getValue(writeRequestItem1);
228         Optional<WriteResponseItem<Byte>> responseValue2 = plcWriteResponse.getValue(writeRequestItem2);
229         assertThat("Unexpected items in response items", responseValue1, equalTo(Optional.of(writeResponseItem1)));
230         assertThat("Unexpected items in response items", responseValue2, equalTo(Optional.of(writeResponseItem2)));
231     }
