117     private <T> void checkRead(MockConnection connection, TypeSafePlcReadRequest<T> request, T value) throws InterruptedException, ExecutionException {
118         // this is really a tests of our mock tooling but knowing it's behaving as expected
119         // will help identify problems in the adapter/supplier/consumer
120         connection.setDataValue(request.getCheckedReadRequestItems().get(0).getAddress(), value);
121 
122         CompletableFuture<TypeSafePlcReadResponse<T>> cf = connection.read(request);
123 
124         Assertions.assertTrue(cf.isDone());
125         TypeSafePlcReadResponse<T> response = cf.get();
126         Assertions.assertEquals(value, response.getResponseItems().get(0).getValues().get(0));
127     }

129     @SuppressWarnings("unchecked")
130     private <T> void checkWrite(MockConnection connection, TypeSafePlcWriteRequest<T> request, T value) throws InterruptedException, ExecutionException {
131         // this is really a tests of our mock tooling but knowing it's behaving as expected
132         // will help identify problems in the adapter/supplier/consumer
133         connection.setDataValue(request.getRequestItems().get(0).getAddress(), value);
134 
135         CompletableFuture<TypeSafePlcWriteResponse<T>> cf = connection.write(request);
136 
137         Assertions.assertTrue(cf.isDone());
138         PlcWriteResponse response = cf.get();
139         Assertions.assertNotNull(response);
140         T writtenData = (T) connection.getDataValue(request.getRequestItems().get(0).getAddress());
141         if (writtenData.getClass().isArray()) {
142             writtenData = (T) Array.get(writtenData, 0);
143         }
144         if (List.class.isAssignableFrom(writtenData.getClass())) {
145             writtenData = (T) ((List) writtenData).get(0);
146         }
147         Assertions.assertEquals(value, writtenData);
148     }

362     private static <T> void checkSupplier(int readFailureCountTrigger, MockConnection connection, Address address, Supplier<T> supplier, Object... values) throws Exception {
363         // verify that a read failure doesn't kill the consumer
364         // it logs (not verified) but returns null (as designed) and keeps working for the subsequent reads
365         connection.setReadException(readFailureCountTrigger, "This is a mock read exception");
366         int readCount = 0;
367         for (Object value : values) {
368             connection.setDataValue(address, value);
369             T readData = supplier.get();
370             // System.out.println("checkSupplier"+(readFailureCountTrigger > 0 ? "NEG" : "")+": value:"+value+" readData:"+readData);
371             if (readFailureCountTrigger <= 0)
372                 Assertions.assertEquals(value, readData);
373             else {
374                 if (++readCount != readFailureCountTrigger)
375                     Assertions.assertEquals(value, readData);
376                 else
377                     Assertions.assertNull(readData);
378             }
379         }
380     }

442     @SuppressWarnings("unchecked")
443     private static <T> void checkConsumer(int writeFailureCountTrigger, MockConnection connection, Address address, Consumer<T> consumer, Object... values) throws Exception {
444         // verify that a write failure doesn't kill the consumer
445         // it logs (not verified) but keeps working for the subsequent writes
446         connection.setWriteException(writeFailureCountTrigger, "This is a mock write exception");
447         int writeCount = 0;
448         Object previousValue = null;
449         for (Object value : values) {
450             consumer.accept((T) value);
451             T writtenData = (T) connection.getDataValue(address);
452             if (List.class.isAssignableFrom(writtenData.getClass())) {
453                 writtenData = (T) ((List) writtenData).get(0);
454             }
455             if (writtenData.getClass().isArray()) {
456                 writtenData = (T) Array.get(writtenData, 0);
457             }
458             // System.out.println("checkConsumer"+(writeFailureCountTrigger > 0 ? "NEG" : "")+": value:"+value+" writtenData:"+writtenData);
459             if (writeFailureCountTrigger <= 0)
460                 Assertions.assertEquals(value, writtenData);
461             else {
462                 if (++writeCount != writeFailureCountTrigger)
463                     Assertions.assertEquals(value, writtenData);
464                 else
465                     Assertions.assertEquals(previousValue, writtenData);
466             }
467             previousValue = value;
468         }
469     }

532     private static <T> void checkConsumerJson(int writeFailureCountTrigger, MockConnection connection, MockAddress address, Consumer<JsonObject> consumer, Object... values) throws Exception {
533         if (writeFailureCountTrigger > 0)
534             connection.setWriteException(writeFailureCountTrigger, "This is a mock write exception");
535         int writeCount = 0;
536         Object previousValue = null;
537         for (Object value : values) {
538 
539             // build the JsonObject to consume
540             JsonObject jo = new JsonObject();
541             jo.addProperty("address", address.getAddress());
542             if (value instanceof Boolean)
543                 jo.addProperty("value", (Boolean) value);
544             else if (value instanceof Number)
545                 jo.addProperty("value", (Number) value);
546             else if (value instanceof String)
547                 jo.addProperty("value", (String) value);
548 
549             consumer.accept(jo);
550 
551             @SuppressWarnings("unchecked")
552             T writtenData = (T) connection.getDataValue(address);
553             if (writtenData.getClass().isArray()) {
554                 writtenData = (T) Array.get(writtenData, 0);
555             }
556             if (List.class.isAssignableFrom(writtenData.getClass())) {
557                 writtenData = (T) ((List) writtenData).get(0);
558             }
559             // System.out.println("checkConsumerJson"+(writeFailureCountTrigger > 0 ? "NEG" : "")+": value:"+value+" writtenData:"+writtenData);
560             if (writeFailureCountTrigger <= 0)
561                 Assertions.assertEquals(value, writtenData);
562             else {
563                 if (++writeCount != writeFailureCountTrigger)
564                     Assertions.assertEquals(value, writtenData);
565                 else
566                     Assertions.assertEquals(previousValue, writtenData);
567             }
568             previousValue = value;
569         }
570     }
