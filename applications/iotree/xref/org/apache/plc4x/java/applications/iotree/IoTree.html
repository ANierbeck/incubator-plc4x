19  package org.apache.plc4x.java.applications.iotree;
20  
21  import org.apache.edgent.function.Consumer;
22  import org.apache.edgent.providers.direct.DirectProvider;
23  import org.apache.edgent.topology.TStream;
24  import org.apache.edgent.topology.Topology;
25  import org.apache.plc4x.edgent.PlcConnectionAdapter;
26  import org.apache.plc4x.edgent.PlcFunctions;
27  
28  import java.util.concurrent.TimeUnit;
29  import java.util.concurrent.atomic.AtomicInteger;
30  
31  public class IoTree {
32  
33      private final PlcConnectionAdapter plcAdapter;
34  
35      private AtomicInteger internalVariables;
36      private AtomicInteger digitalInput;
37      private AtomicInteger analogInput;
38      private AtomicInteger amplification;
39      private AtomicInteger maxValue;
40      private AtomicInteger digitalOutput;
41  
42      private IoTree(PlcConnectionAdapter plcAdapter) throws Exception {
43          this.plcAdapter = plcAdapter;
44          internalVariables = new AtomicInteger(0);
45          digitalInput = new AtomicInteger(0);
46          analogInput = new AtomicInteger(0);
47          amplification = new AtomicInteger(0);
48          maxValue = new AtomicInteger(0);
49          digitalOutput = new AtomicInteger(0);
50      }
51  
52      private void run() throws Exception {
53          DirectProvider dp = new DirectProvider();
54          Topology top = dp.newTopology();
55          // Automatically update the internal variable values ...
56          TStream<Byte> internalVariableStream = top.poll(PlcFunctions.byteSupplier(plcAdapter, "DATA_BLOCKS/0"), 10, TimeUnit.MILLISECONDS);
57          internalVariableStream.sink((Consumer<Byte>) inputs -> internalVariables.set(inputs));
58          // Automatically update the digital input values ...
59          TStream<Byte> digitalInputStream = top.poll(PlcFunctions.byteSupplier(plcAdapter, "INPUTS/0"), 10, TimeUnit.MILLISECONDS);
60          digitalInputStream.sink((Consumer<Byte>) inputs -> digitalInput.set(inputs));
61          // Automatically update the analog input values ...
62          TStream<Short> analogInputStream = top.poll(PlcFunctions.shortSupplier(plcAdapter, "INPUTS/66"), 10, TimeUnit.MILLISECONDS);
63          analogInputStream.sink((Consumer<Short>) inputs -> analogInput.set(inputs));
64          // Automatically update the amplification ...
65          TStream<Float> amplificationStream = top.poll(PlcFunctions.floatSupplier(plcAdapter, "DATA_BLOCKS/1/0"), 10, TimeUnit.MILLISECONDS);
66          amplificationStream.sink((Consumer<Float>) inputs -> amplification.set(inputs.intValue()));
67          // Automatically update the maxValue ...
68          TStream<Integer> maxValueStream = top.poll(PlcFunctions.integerSupplier(plcAdapter, "DATA_BLOCKS/1/4"), 10, TimeUnit.MILLISECONDS);
69          maxValueStream.sink((Consumer<Integer>) inputs -> maxValue.set(inputs));
70          // Automatically update the digital output values ...
71          TStream<Byte> digitalOutputStream = top.poll(PlcFunctions.byteSupplier(plcAdapter, "OUTPUTS/0"), 10, TimeUnit.MILLISECONDS);
72          digitalOutputStream.sink((Consumer<Byte>) outputs -> digitalOutput.set(outputs));
73  
74          // Start logging ...
75          dp.submit(top);
76  
77          while (true) {
78              Thread.sleep(1000);
79              System.out.println(String.format("Internal Variables: %d Digital In: %d Analog In: %d Amplifictaion: %d Max Value: %d Digital Out: %d",
80                  internalVariables.get(), digitalInput.get(), analogInput.get(), amplification.get(), maxValue.get(), digitalOutput.get()));
81          }
82      }
83  
84      public static void main(String[] args) {
85          if((args == null) || (args.length != 1)) {
86              System.out.println("Usage: IoTree {connection-string}");
87              System.out.println("Example: IoTree s7://192.168.0.1/0/0");
88              System.exit(1);
89          }
90  
91          try (PlcConnectionAdapter plcAdapter = new PlcConnectionAdapter(args[0])) {
92              // Initialize the tree itself
93              IoTree tree = new IoTree(plcAdapter);
94              // Start the tree ...
95              tree.run();
96              // Yeah ... well prevent the application from exiting ;-)
97              while (true) {
98                  Thread.sleep(1000);
99              }
100         } catch (Exception e) {
101             e.printStackTrace();
102         }
103     }
