19  package org.apache.plc4x.java.applications.iotree;
20  
21  import org.apache.edgent.function.Consumer;
22  import org.apache.edgent.providers.direct.DirectProvider;
23  import org.apache.edgent.topology.TStream;
24  import org.apache.edgent.topology.Topology;
25  import org.apache.plc4x.edgent.PlcConnectionAdapter;
26  import org.apache.plc4x.edgent.PlcFunctions;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import java.io.IOException;
31  import java.util.concurrent.TimeUnit;
32  import java.util.concurrent.atomic.AtomicInteger;
33  
34  public class IoTree {
35  
36      private static final Logger logger = LoggerFactory.getLogger(IoTree.class);
37  
38      private final PlcConnectionAdapter plcAdapter;
39  
40      private AtomicInteger internalVariables;
41      private AtomicInteger digitalInput;
42      private AtomicInteger analogInput;
43      private AtomicInteger amplification;
44      private AtomicInteger maxValue;
45      private AtomicInteger digitalOutput;
46  
47      private IoTree(PlcConnectionAdapter plcAdapter) {
48          this.plcAdapter = plcAdapter;
49          internalVariables = new AtomicInteger(0);
50          digitalInput = new AtomicInteger(0);
51          analogInput = new AtomicInteger(0);
52          amplification = new AtomicInteger(0);
53          maxValue = new AtomicInteger(0);
54          digitalOutput = new AtomicInteger(0);
55      }
56  
57      private void run() throws InterruptedException, IOException {
58          DirectProvider dp = new DirectProvider();
59          Topology top = dp.newTopology();
60          // Automatically update the internal variable values ...
61          TStream<Byte> internalVariableStream = top.poll(PlcFunctions.byteSupplier(plcAdapter, "DATA_BLOCKS/0"), 10, TimeUnit.MILLISECONDS);
62          internalVariableStream.sink((Consumer<Byte>) inputs -> internalVariables.set(inputs));
63          // Automatically update the digital input values ...
64          TStream<Byte> digitalInputStream = top.poll(PlcFunctions.byteSupplier(plcAdapter, "INPUTS/0"), 10, TimeUnit.MILLISECONDS);
65          digitalInputStream.sink((Consumer<Byte>) inputs -> digitalInput.set(inputs));
66          // Automatically update the analog input values ...
67          TStream<Short> analogInputStream = top.poll(PlcFunctions.shortSupplier(plcAdapter, "INPUTS/66"), 10, TimeUnit.MILLISECONDS);
68          analogInputStream.sink((Consumer<Short>) inputs -> analogInput.set(inputs));
69          // Automatically update the amplification ...
70          TStream<Float> amplificationStream = top.poll(PlcFunctions.floatSupplier(plcAdapter, "DATA_BLOCKS/1/0"), 10, TimeUnit.MILLISECONDS);
71          amplificationStream.sink((Consumer<Float>) inputs -> amplification.set(inputs.intValue()));
72          // Automatically update the maxValue ...
73          TStream<Integer> maxValueStream = top.poll(PlcFunctions.integerSupplier(plcAdapter, "DATA_BLOCKS/1/4"), 10, TimeUnit.MILLISECONDS);
74          maxValueStream.sink((Consumer<Integer>) inputs -> maxValue.set(inputs));
75          // Automatically update the digital output values ...
76          TStream<Byte> digitalOutputStream = top.poll(PlcFunctions.byteSupplier(plcAdapter, "OUTPUTS/0"), 10, TimeUnit.MILLISECONDS);
77          digitalOutputStream.sink((Consumer<Byte>) outputs -> digitalOutput.set(outputs));
78  
79          // Start logging ...
80          dp.submit(top);
81  
82          while (System.in.available() == 0) {
83              Thread.sleep(1000);
84              System.out.println(String.format("Internal Variables: %d Digital In: %d Analog In: %d Amplifictaion: %d Max Value: %d Digital Out: %d",
85                  internalVariables.get(), digitalInput.get(), analogInput.get(), amplification.get(), maxValue.get(), digitalOutput.get()));
86          }
87      }
88  
89      public static void main(String[] args) {
90          if((args == null) || (args.length != 1)) {
91              System.out.println("Usage: IoTree {connection-string}");
92              System.out.println("Example: IoTree s7://192.168.0.1/0/0");
93              System.exit(1);
94          }
95  
96          try (PlcConnectionAdapter plcAdapter = new PlcConnectionAdapter(args[0])) {
97              // Initialize the tree itself
98              IoTree tree = new IoTree(plcAdapter);
99  
100             // Start the tree ...
101             tree.run();
102 
103             System.exit(0);
104         } catch (Exception e) {
105             logger.error("Caught exception", e);
106         }
107     }
108 
109 }
