19  package org.apache.plc4x.java.examples.dummydriver.connection;
20  
21  import io.netty.bootstrap.Bootstrap;
22  import io.netty.channel.*;
23  import org.apache.plc4x.java.api.connection.AbstractPlcConnection;
24  import org.apache.plc4x.java.api.connection.PlcReader;
25  import org.apache.plc4x.java.api.connection.PlcWriter;
26  import org.apache.plc4x.java.api.exceptions.PlcConnectionException;
27  import org.apache.plc4x.java.api.messages.*;
28  import org.apache.plc4x.java.api.model.Address;
29  import org.apache.plc4x.java.examples.dummydriver.model.DummyAddress;
30  import org.apache.plc4x.java.examples.dummydriver.netty.DummyProtocol;
31  import org.apache.plc4x.java.utils.rawsockets.netty.RawSocketAddress;
32  import org.apache.plc4x.java.utils.rawsockets.netty.RawSocketChannel;
33  import org.slf4j.Logger;
34  import org.slf4j.LoggerFactory;
35  
36  import java.util.concurrent.CompletableFuture;
37  import java.util.concurrent.ExecutionException;
38  
39  public class DummyConnection extends AbstractPlcConnection implements PlcReader, PlcWriter {
40  
41      private static final Logger logger = LoggerFactory.getLogger(DummyConnection.class);
42  
43      private final String hostName;
44  
45      private EventLoopGroup workerGroup;
46      private Channel channel;
47      private boolean connected;
48  
49      public DummyConnection(String hostName) {
50          this.hostName = hostName;
51          this.connected = false;
52      }
53  
54      public String getHostName() {
55          return hostName;
56      }
57  
58      @Override
59      public void connect() throws PlcConnectionException {
60          workerGroup = new DefaultEventLoopGroup() {
61          };
62  
63          try {
64              // As we don't just want to wait till the connection is established,
65              // define a future we can use to signal back that the s7 session is
66              // finished initializing.
67              CompletableFuture<Void> sessionSetupCompleteFuture = new CompletableFuture<>();
68  
69              RawSocketAddress serverSocketAddress = new RawSocketAddress(hostName);
70  
71              Bootstrap bootstrap = new Bootstrap();
72              bootstrap.group(workerGroup);
73              bootstrap.channel(RawSocketChannel.class);
74              bootstrap.handler(new ChannelInitializer() {
75                  @Override
76                  protected void initChannel(Channel channel) throws Exception {
77                      ChannelPipeline pipeline = channel.pipeline();
78                      pipeline.addLast(new DummyProtocol());
79                  }
80              });
81              // Start the client.
82              ChannelFuture f = bootstrap.connect(serverSocketAddress).sync();
83              f.awaitUninterruptibly();
84              // Wait till the session is finished initializing.
85              channel = f.channel();
86  
87              sessionSetupCompleteFuture.get();
88  
89              connected = true;
90          } catch (InterruptedException e) {
91              Thread.currentThread().interrupt();
92              throw new PlcConnectionException(e);
93          }
94          catch (ExecutionException e) {
95              throw new PlcConnectionException(e);
96          }
97      }
98  
99      @Override
100     public void close() throws Exception {
101         if((channel != null) && channel.isOpen()) {
102             channel.closeFuture().await();
103         }
104 
105         if (workerGroup != null) {
106             workerGroup.shutdownGracefully();
107         }
108 
109         connected = false;
110     }
111 
112     @Override
113     public boolean isConnected() {
114         return true;
115     }
116 
117     @Override
118     public Address parseAddress(String addressString) {
119         return new DummyAddress(Integer.parseInt(addressString));
120     }
121 
122     @Override
123     public CompletableFuture<PlcReadResponse> read(PlcReadRequest readRequest) {
124         CompletableFuture<PlcReadResponse> readFuture = new CompletableFuture<>();
125         PlcRequestContainer<PlcReadRequest, PlcReadResponse> container =
126             new PlcRequestContainer<>(readRequest, readFuture);
127         channel.writeAndFlush(container);
128         return readFuture;
129     }
130 
131     @Override
132     public CompletableFuture<PlcWriteResponse> write(PlcWriteRequest writeRequest) {
133         CompletableFuture<PlcWriteResponse> writeFuture = new CompletableFuture<>();
134         PlcRequestContainer<PlcWriteRequest, PlcWriteResponse> container =
135             new PlcRequestContainer<>(writeRequest, writeFuture);
136         channel.writeAndFlush(container);
137         return writeFuture;
138     }
139 
140 }
